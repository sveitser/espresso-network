54/// [SubmitPublicUrlsToScrapeTask] is a task that is capable of submitting
55/// public urls to a url sender at a regular interval.  This task will
56/// submit the provided urls to the url sender every 5 minutes.
57pub struct SubmitPublicUrlsToScrapeTask {
58    pub task_handle: Option<JoinHandle<()>>,
59}
60
61const PUBLIC_URL_RESUBMIT_INTERVAL: Duration = Duration::from_secs(300);
62
63impl SubmitPublicUrlsToScrapeTask {
64    pub fn new<S>(url_sender: S, urls: Vec<Url>) -> Self
65    where
66        S: Sink<Url, Error = SendError> + Send + Unpin + 'static,
67    {
68        let task_handle = spawn(Self::submit_urls(url_sender, urls));
69
70        Self {
71            task_handle: Some(task_handle),
72        }
73    }
74
75    pub async fn submit_urls<S>(url_sender: S, urls: Vec<Url>)
76    where
77        S: Sink<Url, Error = SendError> + Unpin + 'static,
78    {
79        if urls.is_empty() {
80            tracing::warn!("no urls to send to url sender");
81            return;
82        }
83
84        let mut url_sender = url_sender;
85        tracing::debug!("sending initial urls to url sender to process node identity");
86        loop {
87            for url in urls.iter() {
88                let send_result = url_sender.send(url.clone()).await;
89                if let Err(err) = send_result {
90                    tracing::error!("url sender closed: {}", err);
91                    panic!("SubmitPublicUrlsToScrapeTask url sender is closed, unrecoverable, the node state will stagnate.");
92                }
93            }
94
95            // Sleep for 5 minutes before sending the urls again
96            tokio::time::sleep(PUBLIC_URL_RESUBMIT_INTERVAL).await;
97        }
98    }
99}

101/**
102 * create_node_validator_processing is a function that creates a node validator
103 * processing environment.  This function will create a number of tasks that
104 * will be responsible for processing the data streams that are coming in from
105 * the various sources.  This function will also create the data state that
106 * will be used to store the state of the network.
107 */
108pub async fn create_node_validator_processing(
109    config: NodeValidatorConfig,
110    internal_client_message_receiver: Receiver<InternalClientMessage<Sender<ServerMessage>>>,
111    leaf_and_block_pair_receiver: Receiver<LeafAndBlock<SeqTypes>>,
112) -> Result<NodeValidatorAPI<Sender<Url>>, CreateNodeValidatorProcessingError> {
113    let client_thread_state = ClientThreadState::<Sender<ServerMessage>>::new(
114        Default::default(),
115        Default::default(),
116        Default::default(),
117        Default::default(),
118        ClientId::from_count(1),
119    );
120
121    let hotshot_client = surf_disco::Client::new(config.stake_table_url_base.clone());
122
123    let hotshot_config = get_config_stake_table_from_sequencer(hotshot_client.clone())
124        .await
125        .map_err(CreateNodeValidatorProcessingError::FailedToGetStakeTable)?;
126    let mut stake_table = hotshot_config.known_nodes_with_stake.clone();
127
128    if let (Some(epoch_starting_block), Some(num_blocks_per_epoch)) = (
129        hotshot_config.epoch_start_block,
130        hotshot_config.epoch_height,
131    ) {
132        tracing::info!(
133            "epoch starting block: {}, num blocks per epoch: {}",
134            epoch_starting_block,
135            num_blocks_per_epoch
136        );
137        let epoch = epoch_from_block_number(config.starting_block_height, num_blocks_per_epoch);
138        if epoch > 1 {
139            // Let's fetch our initial stake table that is not derived from the
140            // initial configuration.
141
142            let node_stake_table =
143                get_node_stake_table_from_sequencer(hotshot_client.clone(), epoch)
144                    .await
145                    .map_err(CreateNodeValidatorProcessingError::FailedToGetStakeTable)?;
146
147            stake_table = node_stake_table;
148        }
149    } else {
150        tracing::warn!(
151            "epoch starting block or num blocks per epoch not found in retrieved hotshot config"
152        );
153    }
154
155    let data_state = DataState::new(Default::default(), Default::default(), stake_table);
156
157    let data_state = Arc::new(RwLock::new(data_state));
158    let client_thread_state = Arc::new(RwLock::new(client_thread_state));
159    let (block_detail_sender, block_detail_receiver) = mpsc::channel(32);
160    let (node_identity_sender_1, node_identity_receiver_1) = mpsc::channel(32);
161    let (node_identity_sender_2, node_identity_receiver_2) = mpsc::channel(32);
162    let (voters_sender, voters_receiver) = mpsc::channel(32);
163    let (url_sender, url_receiver) = mpsc::channel(32);
164
165    let process_internal_client_message_handle = InternalClientMessageProcessingTask::new(
166        internal_client_message_receiver,
167        data_state.clone(),
168        client_thread_state.clone(),
169    );
170
171    let process_distribute_block_detail_handle = ProcessDistributeBlockDetailHandlingTask::new(
172        client_thread_state.clone(),
173        block_detail_receiver,
174    );
175
176    let process_distribute_node_identity_handle = ProcessDistributeNodeIdentityHandlingTask::new(
177        client_thread_state.clone(),
178        node_identity_receiver_2,
179    );
180
181    let process_distribute_voters_handle =
182        ProcessDistributeVotersHandlingTask::new(client_thread_state.clone(), voters_receiver);
183
184    let process_leaf_stream_handle = ProcessLeafAndBlockPairStreamTask::new(
185        leaf_and_block_pair_receiver,
186        data_state.clone(),
187        hotshot_client,
188        hotshot_config,
189        block_detail_sender,
190        voters_sender,
191    );
192
193    let process_node_identity_stream_handle = ProcessNodeIdentityStreamTask::new(
194        node_identity_receiver_1,
195        data_state.clone(),
196        node_identity_sender_2,
197    );
198
199    let process_url_stream_handle =
200        ProcessNodeIdentityUrlStreamTask::new(url_receiver, node_identity_sender_1);
201
202    // Send any initial URLS to the url sender for immediate processing.
203    // These urls are supplied by the configuration of this function
204    let submit_public_urls_handle = SubmitPublicUrlsToScrapeTask::new(
205        url_sender.clone(),
206        config.initial_node_public_base_urls.clone(),
207    );
208
209    Ok(NodeValidatorAPI {
210        process_internal_client_message_handle: Some(process_internal_client_message_handle),
211        process_distribute_block_detail_handle: Some(process_distribute_block_detail_handle),
212        process_distribute_node_identity_handle: Some(process_distribute_node_identity_handle),
213        process_distribute_voters_handle: Some(process_distribute_voters_handle),
214        process_leaf_stream_handle: Some(process_leaf_stream_handle),
215        process_node_identity_stream_handle: Some(process_node_identity_stream_handle),
216        process_url_stream_handle: Some(process_url_stream_handle),
217        submit_public_urls_handle: Some(submit_public_urls_handle),
218        url_sender,
219    })
220}
