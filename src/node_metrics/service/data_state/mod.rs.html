99    // [stake_table_differences] is a helper function that will check the
100    // public key entry differences between the [old_stake_table] and the new
101    // [stake_table].
102    //
103    // This function will return a tuple of a list of of added public keys,
104    // and a list of removed public keys
105    fn stake_table_differences(
106        &self,
107        old_stake_table: &StakeTable<BLSPubKey, StateVerKey, CircuitField>,
108        stake_table: &StakeTable<BLSPubKey, StateVerKey, CircuitField>,
109    ) -> (Vec<BLSPubKey>, Vec<BLSPubKey>) {
110        let old_stake_table_set = match old_stake_table.try_iter(SnapshotVersion::Head) {
111            Ok(into_iter) => into_iter.map(|(key, ..)| key).collect::<HashSet<_>>(),
112            Err(err) => {
113                tracing::error!(
114                    "replace_stake_table: error getting old stake table iterator: {}",
115                    err
116                );
117
118                return (vec![], vec![]);
119            },
120        };
121
122        let new_stake_table_set = match stake_table.try_iter(SnapshotVersion::Head) {
123            Ok(into_iter) => into_iter.map(|(key, ..)| key).collect::<HashSet<_>>(),
124            Err(err) => {
125                tracing::error!(
126                    "replace_stake_table: error getting new stake table iterator: {}",
127                    err
128                );
129
130                return (vec![], vec![]);
131            },
132        };
133
134        let added_public_keys = new_stake_table_set
135            .difference(&old_stake_table_set)
136            .cloned()
137            .collect::<Vec<_>>();
138
139        let removed_public_keys = old_stake_table_set
140            .difference(&new_stake_table_set)
141            .cloned()
142            .collect::<Vec<_>>();
143
144        tracing::info!(
145            "new stake table added {:?} public keys",
146            added_public_keys
147                .iter()
148                .map(ToString::to_string)
149                .collect::<Vec<_>>()
150        );
151        tracing::info!(
152            "new stake table removed {:?} public keys",
153            removed_public_keys
154                .iter()
155                .map(ToString::to_string)
156                .collect::<Vec<_>>()
157        );
158
159        (added_public_keys, removed_public_keys)
160    }

162    pub fn replace_stake_table(
163        &mut self,
164        stake_table: StakeTable<BLSPubKey, StateVerKey, CircuitField>,
165    ) {
166        let old_stake_table = std::mem::replace(&mut self.stake_table, stake_table.clone());
167
168        let current_identity_set = self
169            .node_identity
170            .iter()
171            .map(|node_identity| *node_identity.public_key())
172            .collect::<HashSet<_>>();
173
174        // We want to make sure that we're accounting for this node identity
175        // information that we have.  In the case of any new public keys
176        // being added, we want to ensure we have an entry for them in our
177        // node identity list.
178
179        // TODO @ayiga: We need to figure out how to prune old nodes from
180        //      the list of node identities in such a way that does not
181        //      cause issues or lend to confusion.  For now, we will just
182        //      ignore these removals, and keep a potentially ever-growing
183        //      list of node identities.
184        let (new_public_keys, _) = self.stake_table_differences(&old_stake_table, &stake_table);
185
186        let missing_node_identity_entries = new_public_keys
187            .into_iter()
188            .filter(|key| !current_identity_set.contains(key));
189
190        self.node_identity
191            .extend(missing_node_identity_entries.map(NodeIdentity::from_public_key));
192    }

202    pub fn add_node_identity(&mut self, identity: NodeIdentity) {
203        // We need to check to see if this identity is already in the list,
204        // if it is, we will want to replace it.
205
206        let pub_key = identity.public_key();
207
208        let mut matching_public_keys = self
209            .node_identity
210            .iter()
211            // We want the index of the entry for easier editing
212            .enumerate()
213            .filter(|(_, node_identity)| node_identity.public_key() == pub_key);
214
215        // We only expect this have a single entry.
216        let existing_node_identity_option = matching_public_keys.next();
217
218        debug_assert_eq!(matching_public_keys.next(), None);
219
220        if let Some((index, _)) = existing_node_identity_option {
221            self.node_identity[index] = identity;
222            return;
223        }
224
225        // This entry doesn't appear in our table, so let's add it.
226        self.node_identity.push(identity);
227    }

265/// create_block_detail_from_block is a helper function that will create a
266/// [BlockDetail] from a [BlockQueryData].
267pub fn create_block_detail_from_block(block: &BlockQueryData<SeqTypes>) -> BlockDetail<SeqTypes> {
268    let block_header = block.header();
269    let block_payload = block.payload();
270    let num_transactions = block.num_transactions();
271    let encoded_bytes = block_payload.encode();
272    let total_payload_size = encoded_bytes.len() as u64;
273
274    BlockDetail::<SeqTypes> {
275        hash: block_header.commitment(),
276        height: block_header.height(),
277        time: Timestamp(
278            OffsetDateTime::from_unix_timestamp(block_header.timestamp() as i64)
279                .unwrap_or(OffsetDateTime::UNIX_EPOCH),
280        ),
281        proposer_id: block_header.proposer_id(),
282        num_transactions,
283        block_reward: vec![block_header.fee_info_balance().into()],
284        fee_recipient: block_header.fee_info_account(),
285        size: total_payload_size,
286    }
287}

289// epoch_number_helper is a utility function that will determine the
290// epoch number based on the given [block_height], [epoch_starting_block],
291// and [num_blocks_per_epoch].
292//
293// This function will ensure that the epoch number is zero if the given
294// [block_height] is less than the [epoch_starting_block]. Otherwise it
295// will return the [epoch_from_block_number] result.
296fn epoch_number_helper(
297    block_height: u64,
298    epoch_starting_block: u64,
299    num_blocks_per_epoch: u64,
300) -> u64 {
301    if block_height < epoch_starting_block {
302        return 0;
303    }
304
305    epoch_from_block_number(block_height, num_blocks_per_epoch)
306}

308async fn perform_stake_table_epoch_check_and_update(
309    data_state: Arc<RwLock<DataState>>,
310    client: surf_disco::Client<hotshot_query_service::Error, Version01>,
311    hotshot_config: &PublicHotShotConfig,
312    block_height: u64,
313) -> Result<(), ProcessLeafError> {
314    // Are we in a new epoch?
315    // Do we need to replace our stake table?
316    tracing::debug!("processing block height: {}", block_height);
317    if let (Some(epoch_starting_block), Some(num_blocks_per_epoch)) = (
318        hotshot_config.epoch_start_block,
319        hotshot_config.epoch_height,
320    ) {
321        let previous_epoch = epoch_number_helper(
322            block_height.saturating_sub(1),
323            epoch_starting_block,
324            num_blocks_per_epoch,
325        );
326        let upcoming_epoch =
327            epoch_number_helper(block_height, epoch_starting_block, num_blocks_per_epoch);
328
329        if upcoming_epoch != previous_epoch {
330            tracing::debug!(
331                "new epoch detected: {} -> {} for blocks {}, and {}",
332                previous_epoch,
333                upcoming_epoch,
334                block_height.saturating_sub(1),
335                block_height,
336            );
337            // We're in a new epoch, so we'll need to update our stake table
338            let next_stake_table = match get_node_stake_table_from_sequencer(client, upcoming_epoch)
339                .await
340            {
341                Ok(stake_table) => stake_table,
342                Err(err) => {
343                    tracing::error!("process_incoming_leaf_and_block: error getting stake table from sequencer: {}", err);
344                    return Err(ProcessLeafError::FailedToGetNewStakeTable);
345                },
346            };
347
348            {
349                tracing::debug!(
350                    "replacing stake table for epoch {}, new table contains {} entries",
351                    upcoming_epoch,
352                    next_stake_table.len(SnapshotVersion::Head).unwrap_or(0),
353                );
354                // Update the stake table
355                let mut data_state_write_lock_guard = data_state.write().await;
356                data_state_write_lock_guard.replace_stake_table(next_stake_table);
357            }
358        }
359    }
360
361    Ok(())
362}

364/// [process_incoming_leaf_and_block] is a helper function that will process
365/// an incoming [Leaf] and update the [DataState] with the new information.
366/// Additionally, the block that is contained within the [Leaf] will be
367/// computed into a [BlockDetail] and sent to the [Sink] so that it can be
368/// processed for real-time considerations.
369async fn process_incoming_leaf_and_block<BDSink, BVSink>(
370    leaf: Leaf1QueryData<SeqTypes>,
371    block: BlockQueryData<SeqTypes>,
372    data_state: Arc<RwLock<DataState>>,
373    client: surf_disco::Client<hotshot_query_service::Error, Version01>,
374    hotshot_config: &PublicHotShotConfig,
375    mut block_sender: BDSink,
376    mut voters_sender: BVSink,
377) -> Result<(), ProcessLeafError>
378where
379    Header: BlockHeader<SeqTypes> + QueryableHeader<SeqTypes> + ExplorerHeader<SeqTypes>,
380    Payload: BlockPayload<SeqTypes>,
381    BDSink: Sink<BlockDetail<SeqTypes>, Error = SendError> + Unpin,
382    BVSink: Sink<BitVec<u16>, Error = SendError> + Unpin,
383{
384    let block_detail = create_block_detail_from_block(&block);
385    let block_detail_copy = create_block_detail_from_block(&block);
386
387    let certificate = leaf.leaf().justify_qc();
388    let signatures = &certificate.signatures;
389
390    // Let's take a look at the quorum certificate signatures.
391    // It looks like all of these blocks are being decided by the
392    // same Quorum Certificate.
393
394    // Where's the stake table?
395    let signatures = signatures.as_ref();
396
397    // Let's determine the participants of the voter participants
398    // in the Quorum Certificate.
399
400    // We shouldn't ever have a BitVec that is empty, with the possible
401    // exception of the genesis block.
402    let stake_table_voters_bit_vec = signatures.map_or(Default::default(), |sig| sig.1.clone());
403
404    // This BitVec should be in the same order as the Stake Table.
405    // The StakeTable will be able to change its order between epochs,
406    // which means that its order can change between blocks.
407    // However, the BitVec is a really nice size in order for storing
408    // information.  We should be able to remap the BitVec order from
409    // the StakeTable order to our installed order representation.  This
410    // should allow us to still store as a BitVec while containing our
411    // out order of the voters.
412    // We will need to recompute these BitVecs if the node information that
413    // is stored shrinks instead of growing.
414
415    let block_height = block.header().height();
416
417    // We want to check to see if we need a new stake table before we process
418    // the block and leaf.  Otherwise we might have an outdated stake-table
419    // and may potentially result in miss-mapped staking entries.
420    perform_stake_table_epoch_check_and_update(
421        data_state.clone(),
422        client.clone(),
423        hotshot_config,
424        block_height,
425    )
426    .await?;
427
428    let mut data_state_write_lock_guard = data_state.write().await;
429
430    let stake_table = &data_state_write_lock_guard.stake_table;
431    let stable_table_entries_vec = stake_table
432        .try_iter(SnapshotVersion::Head)
433        .map_or(vec![], |into_iter| into_iter.collect::<Vec<_>>());
434
435    // We have a BitVec of voters who signed the QC.
436    // We can use this to determine the weight of the QC
437    let stake_table_entry_voter_participation_and_entries_pairs =
438        zip(stake_table_voters_bit_vec, stable_table_entries_vec);
439    let stake_table_keys_that_voted = stake_table_entry_voter_participation_and_entries_pairs
440        .filter(|(bit_ref, _)| *bit_ref)
441        .map(|(_, entry)| {
442            // Alright this is our entry that we care about.
443            // In this case, we just want to determine who voted for this
444            // Leaf.
445
446            let (key, ..): (BLSPubKey, _, _) = entry;
447            key
448        });
449
450    let voters_set: HashSet<BLSPubKey> = stake_table_keys_that_voted.collect();
451
452    let voters_bitvec = data_state_write_lock_guard.node_identity.iter().fold(
453        BitVec::with_capacity(data_state_write_lock_guard.node_identity.len()),
454        |mut acc, node_identity| {
455            acc.push(voters_set.contains(node_identity.public_key()));
456            acc
457        },
458    );
459
460    data_state_write_lock_guard
461        .latest_blocks
462        .push_back(block_detail);
463    data_state_write_lock_guard
464        .latest_voters
465        .push_back(voters_bitvec.clone());
466
467    drop(data_state_write_lock_guard);
468
469    if let Err(err) = block_sender.send(block_detail_copy).await {
470        // We have an error that prevents us from continuing
471        return Err(ProcessLeafError::BlockSendError(err));
472    }
473
474    if let Err(err) = voters_sender.send(voters_bitvec).await {
475        // We have an error that prevents us from continuing
476        return Err(ProcessLeafError::VotersSendError(err));
477    }
478
479    Ok(())
480}

482/// [ProcessLeafAndBlockPairStreamTask] represents the task that is responsible
483/// for processing a stream of incoming pairs of [Leaf]s and [BlockQueryData].
484pub struct ProcessLeafAndBlockPairStreamTask {
485    pub task_handle: Option<JoinHandle<()>>,
486}
487
488impl ProcessLeafAndBlockPairStreamTask {
489    /// [new] creates a new [ProcessLeafStreamTask] that will process a stream
490    /// of incoming [Leaf]s.
491    ///
492    /// Calling this function will create an asynchronous task that will start
493    /// processing immediately. The handle for the task will be stored within
494    /// the returned structure.
495    pub fn new<S, K1, K2>(
496        leaf_receiver: S,
497        data_state: Arc<RwLock<DataState>>,
498        client: surf_disco::Client<hotshot_query_service::Error, Version01>,
499        hotshot_config: PublicHotShotConfig,
500        block_detail_sender: K1,
501        voters_sender: K2,
502    ) -> Self
503    where
504        S: Stream<Item = LeafAndBlock<SeqTypes>> + Send + Sync + Unpin + 'static,
505        K1: Sink<BlockDetail<SeqTypes>, Error = SendError> + Clone + Send + Sync + Unpin + 'static,
506        K2: Sink<BitVec<u16>, Error = SendError> + Clone + Send + Sync + Unpin + 'static,
507    {
508        let task_handle = spawn(Self::process_leaf_stream(
509            leaf_receiver,
510            data_state.clone(),
511            client,
512            hotshot_config,
513            block_detail_sender,
514            voters_sender,
515        ));
516
517        Self {
518            task_handle: Some(task_handle),
519        }
520    }
521
522    /// [process_leaf_stream] allows for the consumption of a [Stream] when
523    /// attempting to process new incoming [Leaf]s.
524    async fn process_leaf_stream<S, BDSink, BVSink>(
525        mut stream: S,
526        data_state: Arc<RwLock<DataState>>,
527        client: surf_disco::Client<hotshot_query_service::Error, Version01>,
528        hotshot_config: PublicHotShotConfig,
529        block_sender: BDSink,
530        voters_senders: BVSink,
531    ) where
532        S: Stream<Item = LeafAndBlock<SeqTypes>> + Unpin,
533        Header: BlockHeader<SeqTypes> + QueryableHeader<SeqTypes> + ExplorerHeader<SeqTypes>,
534        Payload: BlockPayload<SeqTypes>,
535        BDSink: Sink<BlockDetail<SeqTypes>, Error = SendError> + Clone + Unpin,
536        BVSink: Sink<BitVec<u16>, Error = SendError> + Clone + Unpin,
537    {
538        loop {
539            let leaf_result = stream.next().await;
540            let (leaf, block) = if let Some(pair) = leaf_result {
541                pair
542            } else {
543                // We have reached the end of the stream
544                tracing::error!("process leaf stream: end of stream reached for leaf stream.");
545                return;
546            };
547
548            if let Err(err) = process_incoming_leaf_and_block(
549                leaf,
550                block,
551                data_state.clone(),
552                client.clone(),
553                &hotshot_config,
554                block_sender.clone(),
555                voters_senders.clone(),
556            )
557            .await
558            {
559                // We have an error that prevents us from continuing
560                tracing::error!("process leaf stream: error processing leaf: {}", err);
561
562                // At the moment, all underlying errors are due to `SendError`
563                // which will ultimately mean that further processing attempts
564                // will fail, and be fruitless.
565                match err {
566                    ProcessLeafError::BlockSendError(_) => {
567                        panic!("ProcessLeafStreamTask: process_incoming_leaf failed, underlying sink is closed, blocks will stagnate: {}", err)
568                    },
569                    ProcessLeafError::VotersSendError(_) => {
570                        panic!("ProcessLeafStreamTask: process_incoming_leaf failed, underlying sink is closed, voters will stagnate: {}", err)
571                    },
572                    ProcessLeafError::FailedToGetNewStakeTable => {
573                        panic!("ProcessLeafStreamTask: process_incoming_leaf failed, underlying stake table is closed, blocks will stagnate: {}", err)
574                    },
575                }
576            }
577        }
578    }
579}

623/// [process_incoming_node_identity] is a helper function that will process an
624/// incoming [NodeIdentity] and update the [DataState] with the new information.
625/// Additionally, the [NodeIdentity] will be sent to the [Sink] so that it can
626/// be processed for real-time considerations.
627async fn process_incoming_node_identity<NISink>(
628    node_identity: NodeIdentity,
629    data_state: Arc<RwLock<DataState>>,
630    mut node_identity_sender: NISink,
631) -> Result<(), ProcessNodeIdentityError>
632where
633    NISink: Sink<NodeIdentity, Error = SendError> + Unpin,
634{
635    let mut data_state_write_lock_guard = data_state.write().await;
636    data_state_write_lock_guard.add_node_identity(node_identity.clone());
637    node_identity_sender.send(node_identity).await?;
638
639    Ok(())
640}

642/// [ProcessNodeIdentityStreamTask] represents the task that is responsible for
643/// processing a stream of incoming [NodeIdentity]s and updating the [DataState]
644/// with the new information.
645pub struct ProcessNodeIdentityStreamTask {
646    pub task_handle: Option<JoinHandle<()>>,
647}
648
649impl ProcessNodeIdentityStreamTask {
650    /// [new] creates a new [ProcessNodeIdentityStreamTask] that will process a
651    /// stream of incoming [NodeIdentity]s.
652    ///
653    /// Calling this function will create an asynchronous task that will start
654    /// processing immediately. The handle for the task will be stored within
655    /// the returned structure.
656    pub fn new<S, K>(
657        node_identity_receiver: S,
658        data_state: Arc<RwLock<DataState>>,
659        node_identity_sender: K,
660    ) -> Self
661    where
662        S: Stream<Item = NodeIdentity> + Send + Sync + Unpin + 'static,
663        K: Sink<NodeIdentity, Error = SendError> + Clone + Send + Sync + Unpin + 'static,
664    {
665        let task_handle = spawn(Self::process_node_identity_stream(
666            node_identity_receiver,
667            data_state.clone(),
668            node_identity_sender,
669        ));
670
671        Self {
672            task_handle: Some(task_handle),
673        }
674    }
675
676    /// [process_node_identity_stream] allows for the consumption of a [Stream] when
677    /// attempting to process new incoming [NodeIdentity]s.
678    /// This function will process the incoming [NodeIdentity] and update the
679    /// [DataState] with the new information.
680    /// Additionally, the [NodeIdentity] will be sent to the [Sink] so that it can
681    /// be processed for real-time considerations.
682    async fn process_node_identity_stream<S, NISink>(
683        mut stream: S,
684        data_state: Arc<RwLock<DataState>>,
685        node_identity_sender: NISink,
686    ) where
687        S: Stream<Item = NodeIdentity> + Unpin,
688        NISink: Sink<NodeIdentity, Error = SendError> + Clone + Unpin,
689    {
690        loop {
691            let node_identity_result = stream.next().await;
692            let node_identity = if let Some(node_identity) = node_identity_result {
693                node_identity
694            } else {
695                // We have reached the end of the stream
696                tracing::info!(
697                    "process node identity stream: end of stream reached for node identity stream."
698                );
699                return;
700            };
701
702            if let Err(err) = process_incoming_node_identity(
703                node_identity,
704                data_state.clone(),
705                node_identity_sender.clone(),
706            )
707            .await
708            {
709                // We have an error that prevents us from continuing
710                tracing::error!(
711                    "process node identity stream: error processing node identity: {}",
712                    err
713                );
714
715                // The only underlying class of errors that can be returned from
716                // `process_incoming_node_identity` are due to `SendError` which
717                // will ultimately mean that further processing attempts will fail
718                // and be fruitless.
719                panic!("ProcessNodeIdentityStreamTask: process_incoming_node_identity failed, underlying sink is closed, node identities will stagnate: {}", err);
720            }
721        }
722    }
723}
724
725/// [Drop] implementation for [ProcessNodeIdentityStreamTask] that will cancel
726/// the task if it is dropped.
727impl Drop for ProcessNodeIdentityStreamTask {
728    fn drop(&mut self) {
729        let task_handle = self.task_handle.take();
730        if let Some(task_handle) = task_handle {
731            task_handle.abort();
732        }
733    }
734}
