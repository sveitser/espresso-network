62impl StakeTableEvents {
63    pub fn sort_events(self) -> anyhow::Result<Vec<(EventKey, StakeTableEvent)>> {
64        let mut events: Vec<(EventKey, StakeTableEvent)> = Vec::new();
65        let Self {
66            registrations,
67            deregistrations,
68            delegated,
69            undelegated,
70            keys,
71        } = self;
72
73        for (registration, log) in registrations {
74            events.push((
75                (
76                    log.block_number.context("block number")?,
77                    log.log_index.context("log index")?,
78                ),
79                registration.into(),
80            ));
81        }
82        for (dereg, log) in deregistrations {
83            events.push((
84                (
85                    log.block_number.context("block number")?,
86                    log.log_index.context("log index")?,
87                ),
88                dereg.into(),
89            ));
90        }
91        for (delegation, log) in delegated {
92            events.push((
93                (
94                    log.block_number.context("block number")?,
95                    log.log_index.context("log index")?,
96                ),
97                delegation.into(),
98            ));
99        }
100        for (undelegated, log) in undelegated {
101            events.push((
102                (
103                    log.block_number.context("block number")?,
104                    log.log_index.context("log index")?,
105                ),
106                undelegated.into(),
107            ));
108        }
109
110        for (update, log) in keys {
111            events.push((
112                (
113                    log.block_number.context("block number")?,
114                    log.log_index.context("log index")?,
115                ),
116                update.into(),
117            ));
118        }
119
120        events.sort_by_key(|(key, _)| (key.0, key.1));
121        Ok(events)
122    }
123}

125/// Extract all validators from L1 stake table events.
126pub fn validators_from_l1_events<I: Iterator<Item = StakeTableEvent>>(
127    events: I,
128) -> anyhow::Result<IndexMap<Address, Validator<BLSPubKey>>> {
129    let mut validators = IndexMap::new();
130    let mut bls_keys = HashSet::new();
131    let mut schnorr_keys = HashSet::new();
132    for event in events {
133        tracing::debug!("Processing stake table event: {:?}", event);
134        match event {
135            StakeTableEvent::Register(ValidatorRegistered {
136                account,
137                blsVk,
138                schnorrVk,
139                commission,
140            }) => {
141                // TODO(abdul): BLS and Schnorr signature keys verification
142                let stake_table_key: BLSPubKey = blsVk.clone().into();
143                let state_ver_key: SchnorrPubKey = schnorrVk.clone().into();
144                // TODO(MA): The stake table contract currently enforces that each bls key is only used once. We will
145                // move this check to the confirmation layer and remove it from the contract. Once we have the signature
146                // check in this functions we can skip if a BLS key, or Schnorr key was previously used.
147                if bls_keys.contains(&stake_table_key) {
148                    bail!("bls key {} already used", stake_table_key.to_string());
149                };
150
151                // The contract does *not* enforce that each schnorr key is only used once.
152                if schnorr_keys.contains(&state_ver_key) {
153                    tracing::warn!("schnorr key {} already used", state_ver_key.to_string());
154                };
155
156                bls_keys.insert(stake_table_key);
157                schnorr_keys.insert(state_ver_key.clone());
158
159                match validators.entry(account) {
160                    indexmap::map::Entry::Occupied(_occupied_entry) => {
161                        bail!("validator {:#x} already registered", *account)
162                    },
163                    indexmap::map::Entry::Vacant(vacant_entry) => vacant_entry.insert(Validator {
164                        account,
165                        stake_table_key,
166                        state_ver_key,
167                        stake: U256::from(0_u64),
168                        commission,
169                        delegators: HashMap::default(),
170                    }),
171                };
172            },
173            StakeTableEvent::Deregister(exit) => {
174                validators
175                    .shift_remove(&exit.validator)
176                    .with_context(|| format!("validator {:#x} not found", exit.validator))?;
177            },
178            StakeTableEvent::Delegate(delegated) => {
179                let Delegated {
180                    delegator,
181                    validator,
182                    amount,
183                } = delegated;
184                let validator_entry = validators
185                    .get_mut(&validator)
186                    .with_context(|| format!("validator {validator:#x} not found"))?;
187
188                if amount.is_zero() {
189                    tracing::warn!("delegator {delegator:?} has 0 stake");
190                    continue;
191                }
192                // Increase stake
193                validator_entry.stake += amount;
194                // Insert the delegator with the given stake
195                // or increase the stake if already present
196                validator_entry
197                    .delegators
198                    .entry(delegator)
199                    .and_modify(|stake| *stake += amount)
200                    .or_insert(amount);
201            },
202            StakeTableEvent::Undelegate(undelegated) => {
203                let Undelegated {
204                    delegator,
205                    validator,
206                    amount,
207                } = undelegated;
208                let validator_entry = validators
209                    .get_mut(&validator)
210                    .with_context(|| format!("validator {validator:#x} not found"))?;
211
212                validator_entry.stake = validator_entry
213                    .stake
214                    .checked_sub(amount)
215                    .with_context(|| "stake is less than undelegated amount")?;
216
217                let delegator_stake = validator_entry
218                    .delegators
219                    .get_mut(&delegator)
220                    .with_context(|| format!("delegator {delegator:#x} not found"))?;
221                *delegator_stake = delegator_stake
222                    .checked_sub(amount)
223                    .with_context(|| "delegator_stake is less than undelegated amount")?;
224
225                if delegator_stake.is_zero() {
226                    // if delegator stake is 0, remove from set
227                    validator_entry.delegators.remove(&delegator);
228                }
229            },
230            StakeTableEvent::KeyUpdate(update) => {
231                let ConsensusKeysUpdated {
232                    account,
233                    blsVK,
234                    schnorrVK,
235                } = update;
236                let validator = validators
237                    .get_mut(&account)
238                    .with_context(|| "validator {account:#x} not found")?;
239                let bls = blsVK.into();
240                let state_ver_key = schnorrVK.into();
241
242                validator.stake_table_key = bls;
243                validator.state_ver_key = state_ver_key;
244            },
245        }
246    }
247
248    Ok(validators)
249}

251/// Select active validators
252///
253/// Removes the validators without stake and selects the top 100 staked validators.
254pub(crate) fn select_active_validator_set(
255    validators: &mut IndexMap<Address, Validator<BLSPubKey>>,
256) -> anyhow::Result<()> {
257    // Remove invalid validators first
258    validators.retain(|address, validator| {
259        if validator.delegators.is_empty() {
260            tracing::info!("Validator {address:?} does not have any delegator");
261            return false;
262        }
263
264        if validator.stake.is_zero() {
265            tracing::info!("Validator {address:?} does not have any stake");
266            return false;
267        }
268
269        true
270    });
271
272    if validators.is_empty() {
273        bail!("No valid validators found");
274    }
275
276    // Find the maximum stake
277    let maximum_stake = validators
278        .values()
279        .map(|v| v.stake)
280        .max()
281        .context("Failed to determine max stake")?;
282
283    let minimum_stake = maximum_stake
284        .checked_div(U256::from(VID_TARGET_TOTAL_STAKE))
285        .context("div err")?;
286
287    // Collect validators that meet the minimum stake criteria
288    let mut valid_stakers: Vec<_> = validators
289        .iter()
290        .filter(|(_, v)| v.stake >= minimum_stake)
291        .map(|(addr, v)| (*addr, v.stake))
292        .collect();
293
294    // Sort by stake (descending order)
295    valid_stakers.sort_by_key(|(_, stake)| std::cmp::Reverse(*stake));
296
297    // Keep only the top 100 stakers
298    if valid_stakers.len() > 100 {
299        valid_stakers.truncate(100);
300    }
301
302    // Retain only the selected validators
303    let selected_addresses: HashSet<_> = valid_stakers.iter().map(|(addr, _)| *addr).collect();
304    validators.retain(|address, _| selected_addresses.contains(address));
305
306    Ok(())
307}

309/// Extract the active validator set from the L1 stake table events.
310pub(crate) fn active_validator_set_from_l1_events<I: Iterator<Item = StakeTableEvent>>(
311    events: I,
312) -> anyhow::Result<IndexMap<Address, Validator<BLSPubKey>>> {
313    let mut validators = validators_from_l1_events(events)?;
314    select_active_validator_set(&mut validators)?;
315    Ok(validators)
316}

359    pub async fn spawn_update_loop(&self) {
360        let mut update_task = self.update_task.0.lock().await;
361        if update_task.is_none() {
362            *update_task = Some(spawn(self.update_loop()));
363        }
364    }

366    /// Periodically updates the stake table from the L1 contract.
367    /// This function polls the finalized block number from the L1 client at an interval
368    /// and fetches stake table from contract
369    /// and updates the persistence
370    fn update_loop(&self) -> impl Future<Output = ()> {
371        let span = tracing::warn_span!("Stake table update loop");
372        let self_clone = self.clone();
373        let state = self.l1_client.state.clone();
374        let l1_retry = self.l1_client.options().l1_retry_delay;
375        let update_delay = self.l1_client.options().stake_table_update_interval;
376        let chain_config = self.chain_config.clone();
377
378        async move {
379            // Get the stake table contract address from the chain config.
380            // This may not contain a stake table address if we are on a pre-epoch version.
381            // It keeps retrying until the chain config is upgraded
382            // after a successful upgrade to an epoch version.
383            let stake_contract_address = loop {
384                match chain_config.lock().await.stake_table_contract {
385                    Some(addr) => break addr,
386                    None => {
387                        tracing::debug!(
388                            "Stake table contract address not found. Retrying in {l1_retry:?}...",
389                        );
390                    },
391                }
392                sleep(l1_retry).await;
393            };
394
395            // Begin the main polling loop
396            loop {
397                let finalized_block = loop {
398                    if let Some(block) = state.lock().await.last_finalized {
399                        break block;
400                    }
401                    tracing::debug!(
402                        "Finalized block not yet available. Retrying in {l1_retry:?}",
403                    );
404                    sleep(l1_retry).await;
405                };
406
407                tracing::debug!(
408                    "Attempting to fetch stake table at L1 block {finalized_block:?}",
409                );
410
411                // Retry stake table fetch until it succeeds
412                loop {
413                    match self_clone
414                        .fetch_and_store_stake_table_events(stake_contract_address, finalized_block)
415                        .await
416                    {
417                        Ok(_) => {
418                            tracing::info!("Successfully fetched and stored stake table events at block={finalized_block:?}");
419                            break;
420                        },
421                        Err(e) => {
422                            tracing::error!(
423                                "Error fetching stake table at block {finalized_block:?}. err= {e:#}",
424                            );
425                            sleep(l1_retry).await;
426                        },
427                    }
428                }
429
430                tracing::debug!(
431                    "Waiting {update_delay:?} before next stake table update...",
432                );
433                sleep(update_delay).await;
434            }
435        }
436        .instrument(span)
437    }

439    pub async fn fetch_events(
440        &self,
441        contract: Address,
442        to_block: u64,
443    ) -> anyhow::Result<Vec<(EventKey, StakeTableEvent)>> {
444        let persistence_lock = self.persistence.lock().await;
445        let res = persistence_lock.load_events().await?;
446        drop(persistence_lock);
447
448        let from_block = res
449            .as_ref()
450            .map(|(block, _)| block + 1)
451            .filter(|from| *from <= to_block); // Only use from_block if it's less than to_block
452
453        tracing::info!("loaded events from storage from_block={from_block:?}");
454
455        let contract_events = Self::fetch_events_from_contract(
456            self.l1_client.clone(),
457            contract,
458            from_block,
459            to_block,
460        )
461        .await?;
462
463        tracing::info!("loading events from contract to_block={to_block:?}");
464
465        let contract_events = contract_events.sort_events()?;
466        let mut events = match (from_block, res) {
467            (Some(_), Some((_, persistence_events))) => persistence_events
468                .into_iter()
469                .chain(contract_events)
470                .collect(),
471            _ => contract_events,
472        };
473
474        // There are no duplicates because the RPC returns all events,
475        // which are stored directly in persistence as is.
476        // However, this step is taken as a precaution.
477        // The vector is already sorted above, so this should be fast.
478        let len_before_dedup = events.len();
479        events.dedup();
480        let len_after_dedup = events.len();
481        if len_before_dedup != len_after_dedup {
482            tracing::warn!("Duplicate events found and removed. This should not normally happen.")
483        }
484
485        Ok(events)
486    }

488    /// Fetch all stake table events from L1
489    pub async fn fetch_events_from_contract(
490        l1_client: L1Client,
491        contract: Address,
492        from_block: Option<u64>,
493        to_block: u64,
494    ) -> anyhow::Result<StakeTableEvents> {
495        let stake_table_contract = StakeTable::new(contract, l1_client.provider.clone());
496
497        // get the block number when the contract was initialized
498        // to avoid fetching events from block number 0
499        let from_block = match from_block {
500            Some(block) => block,
501            None => {
502                loop {
503                    match stake_table_contract.initializedAtBlock().call().await {
504                        Ok(init_block) => {
505                            break init_block._0.to::<u64>();
506                        },
507                        Err(err) => {
508                            // Retry fetching incase of an error
509                            tracing::warn!(%err, "Failed to retrieve initial block, retrying..");
510                            sleep(l1_client.options().l1_retry_delay).await;
511                        },
512                    }
513                }
514            },
515        };
516
517        // To avoid making large RPC calls, divide the range into smaller chunks.
518        // chunk size is from env "ESPRESSO_SEQUENCER_L1_EVENTS_MAX_BLOCK_RANGE
519        // default value  is `10000` if env variable is not set
520        let mut start = from_block;
521        let end = to_block;
522        let chunk_size = l1_client.options().l1_events_max_block_range;
523        let chunks = std::iter::from_fn(move || {
524            let chunk_end = min(start + chunk_size - 1, end);
525            if chunk_end < start {
526                return None;
527            }
528
529            let chunk = (start, chunk_end);
530            start = chunk_end + 1;
531            Some(chunk)
532        });
533
534        // fetch registered events
535        // retry if the call to the provider to fetch the events fails
536        let registered_events = stream::iter(chunks.clone()).then(|(from, to)| {
537            let retry_delay = l1_client.options().l1_retry_delay;
538            let stake_table_contract = stake_table_contract.clone();
539            async move {
540                tracing::debug!(from, to, "fetch ValidatorRegistered events in range");
541                loop {
542                    match stake_table_contract
543                        .clone()
544                        .ValidatorRegistered_filter()
545                        .from_block(from)
546                        .to_block(to)
547                        .query()
548                        .await
549                    {
550                        Ok(events) => break stream::iter(events),
551                        Err(err) => {
552                            tracing::warn!(from, to, %err, "ValidatorRegistered Error");
553                            sleep(retry_delay).await;
554                        },
555                    }
556                }
557            }
558        });
559
560        // fetch validator de registration events
561        let deregistered_events = stream::iter(chunks.clone()).then(|(from, to)| {
562            let retry_delay = l1_client.options().l1_retry_delay;
563            let stake_table_contract = stake_table_contract.clone();
564            async move {
565                tracing::debug!(from, to, "fetch ValidatorExit events in range");
566                loop {
567                    match stake_table_contract
568                        .ValidatorExit_filter()
569                        .from_block(from)
570                        .to_block(to)
571                        .query()
572                        .await
573                    {
574                        Ok(events) => break stream::iter(events),
575                        Err(err) => {
576                            tracing::warn!(from, to, %err, "ValidatorExit Error");
577                            sleep(retry_delay).await;
578                        },
579                    }
580                }
581            }
582        });
583
584        // fetch delegated events
585        let delegated_events = stream::iter(chunks.clone()).then(|(from, to)| {
586            let retry_delay = l1_client.options().l1_retry_delay;
587            let stake_table_contract = stake_table_contract.clone();
588            async move {
589                tracing::debug!(from, to, "fetch Delegated events in range");
590                loop {
591                    match stake_table_contract
592                        .Delegated_filter()
593                        .from_block(from)
594                        .to_block(to)
595                        .query()
596                        .await
597                    {
598                        Ok(events) => break stream::iter(events),
599                        Err(err) => {
600                            tracing::warn!(from, to, %err, "Delegated Error");
601                            sleep(retry_delay).await;
602                        },
603                    }
604                }
605            }
606        });
607        // fetch undelegated events
608        let undelegated_events = stream::iter(chunks.clone()).then(|(from, to)| {
609            let retry_delay = l1_client.options().l1_retry_delay;
610            let stake_table_contract = stake_table_contract.clone();
611            async move {
612                tracing::debug!(from, to, "fetch Undelegated events in range");
613                loop {
614                    match stake_table_contract
615                        .Undelegated_filter()
616                        .from_block(from)
617                        .to_block(to)
618                        .query()
619                        .await
620                    {
621                        Ok(events) => break stream::iter(events),
622                        Err(err) => {
623                            tracing::warn!(from, to, %err, "Undelegated Error");
624                            sleep(retry_delay).await;
625                        },
626                    }
627                }
628            }
629        });
630
631        // fetch consensus keys updated events
632        let keys_update_events = stream::iter(chunks).then(|(from, to)| {
633            let retry_delay = l1_client.options().l1_retry_delay;
634            let stake_table_contract = stake_table_contract.clone();
635            async move {
636                tracing::debug!(from, to, "fetch ConsensusKeysUpdated events in range");
637                loop {
638                    match stake_table_contract
639                        .ConsensusKeysUpdated_filter()
640                        .from_block(from)
641                        .to_block(to)
642                        .query()
643                        .await
644                    {
645                        Ok(events) => break stream::iter(events),
646                        Err(err) => {
647                            tracing::warn!(from, to, %err, "ConsensusKeysUpdated Error");
648                            sleep(retry_delay).await;
649                        },
650                    }
651                }
652            }
653        });
654
655        let registrations = registered_events.flatten().collect().await;
656        let deregistrations = deregistered_events.flatten().collect().await;
657        let delegated = delegated_events.flatten().collect().await;
658        let undelegated = undelegated_events.flatten().collect().await;
659        let keys = keys_update_events.flatten().collect().await;
660
661        Ok(StakeTableEvents {
662            registrations,
663            deregistrations,
664            delegated,
665            undelegated,
666            keys,
667        })
668    }

670    /// Get `StakeTable` at specific l1 block height.
671    /// This function fetches and processes various events (ValidatorRegistered, ValidatorExit,
672    /// Delegated, Undelegated, and ConsensusKeysUpdated) within the block range from the
673    /// contract's initialization block to the provided `to_block` value.
674    /// Events are fetched in chunks to and retries are implemented for failed requests.
675    pub async fn fetch_and_store_stake_table_events(
676        &self,
677        contract: Address,
678        to_block: u64,
679    ) -> anyhow::Result<Vec<(EventKey, StakeTableEvent)>> {
680        let events = self.fetch_events(contract, to_block).await?;
681
682        tracing::info!("storing events in storage to_block={to_block:?}");
683
684        {

687                .store_events(to_block, events.clone())
688                .await
689                .inspect_err(|e| tracing::error!("failed to store events. err={e}"))?;

707    pub async fn fetch(
708        &self,
709        epoch: Epoch,
710        header: Header,
711    ) -> Option<IndexMap<Address, Validator<BLSPubKey>>> {
712        let chain_config = self.get_chain_config(&header).await.ok()?;
713        // update chain config
714        *self.chain_config.lock().await = chain_config;
715
716        let Some(address) = chain_config.stake_table_contract else {
717            tracing::error!("No stake table contract address found in Chain config");
718            return None;
719        };
720
721        let Some(l1_finalized_block_info) = header.l1_finalized() else {
722            tracing::error!("The epoch root for epoch {} is missing the L1 finalized block info. This is a fatal error. Consensus is blocked and will not recover.", epoch);
723            return None;
724        };
725
726        let events = match self
727            .fetch_and_store_stake_table_events(address, l1_finalized_block_info.number())
728            .await
729            .map_err(GetStakeTablesError::L1ClientFetchError)
730        {
731            Ok(events) => events,
732            Err(e) => {
733                tracing::error!("failed to fetch stake table events {e:?}");
734                return None;
735            },
736        };
737
738        match active_validator_set_from_l1_events(events.into_iter().map(|(_, e)| e)) {
739            Ok(validators) => Some(validators),
740            Err(e) => {
741                tracing::error!("failed to construct stake table {e:?}");
742                None
743            },
744        }
745    }

749    pub(crate) async fn get_chain_config(&self, header: &Header) -> anyhow::Result<ChainConfig> {
750        let chain_config = self.chain_config.lock().await;
751        let peers = self.peers.clone();
752        let header_cf = header.chain_config();
753        if chain_config.commit() == header_cf.commit() {

755        }

759            None => peers

762                .map_err(|err| {
763                    tracing::error!("failed to get chain_config from peers. err: {err:?}");
764                    err
765                })?,
766        };
767
768        Ok(cf)

845                        stake_table_entry: BLSPubKey::stake_table_entry(

1277    async fn get_epoch_drb(
1278        membership: Arc<RwLock<Self>>,
1279        block_height: u64,
1280        epoch: Epoch,
1281    ) -> anyhow::Result<DrbResult> {
1282        let membership_reader = membership.read().await;
1283        let peers = membership_reader.fetcher.peers.clone();
1284        let stake_table = membership_reader.stake_table(Some(epoch)).clone();
1285        let success_threshold = membership_reader.success_threshold(Some(epoch));
1286        drop(membership_reader);
1287
1288        tracing::debug!(
1289            "Getting DRB for epoch {:?}, block height {:?}",
1290            epoch,
1291            block_height
1292        );
1293        let mut drb_leaf_chain = peers.try_fetch_leaves(1, block_height).await?;
1294
1295        drb_leaf_chain.sort_by_key(|l| l.view_number());
1296        let leaf_chain = drb_leaf_chain.into_iter().rev().collect();
1297        let drb_leaf = verify_leaf_chain(
1298            leaf_chain,
1299            stake_table.clone(),
1300            success_threshold,
1301            block_height,
1302            &UpgradeLock::<SeqTypes, SequencerVersions<EpochVersion, EpochVersion>>::new(),
1303        )
1304        .await?;
1305
1306        let Some(drb) = drb_leaf.next_drb_result else {
1307            tracing::error!(
1308          "We received a leaf that should contain a DRB result, but the DRB result is missing: {:?}",
1309          drb_leaf
1310        );
1311
1312            bail!("DRB leaf is missing the DRB result.");
1313        };
1314
1315        Ok(drb)
1316    }

1319        let Some(raw_stake_table) = self.state.get(&epoch) else {
1320            tracing::error!("add_drb_result({}, {:?}) was called, but we do not yet have the stake table for epoch {}", epoch, drb, epoch);
1321            return;
1322        };
1323
1324        let leaders = raw_stake_table
1325            .eligible_leaders
1326            .clone()
1327            .into_iter()
1328            .map(|peer_config| peer_config.stake_table_entry)
1329            .collect::<Vec<_>>();
1330        let randomized_committee = generate_stake_cdf(leaders, drb);
1331
1332        self.randomized_committees
1333            .insert(epoch, randomized_committee);
1334    }
1335
1336    fn set_first_epoch(&mut self, epoch: Epoch, initial_drb_result: DrbResult) {
1337        self.first_epoch = Some(epoch);
1338
1339        let epoch_committee = self.state.get(&Epoch::genesis()).unwrap().clone();
1340        self.state.insert(epoch, epoch_committee.clone());
1341        self.state.insert(epoch + 1, epoch_committee);
1342        self.add_drb_result(epoch, initial_drb_result);
1343        self.add_drb_result(epoch + 1, initial_drb_result);
1344    }
1345}
