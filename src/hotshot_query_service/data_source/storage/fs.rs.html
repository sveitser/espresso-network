90impl<Types> FileSystemStorageInner<Types>
91where
92    Types: NodeType,
93    Payload<Types>: QueryablePayload<Types>,
94{
95    fn get_block_index(&self, id: BlockId<Types>) -> QueryResult<usize> {
96        match id {
97            BlockId::Number(n) => Ok(n),
98            BlockId::Hash(h) => {
99                Ok(*self.index_by_block_hash.get(&h).context(NotFoundSnafu)? as usize)
100            },
101            BlockId::PayloadHash(h) => {
102                Ok(*self.index_by_payload_hash.get(&h).context(NotFoundSnafu)? as usize)
103            },
104        }
105    }
106
107    fn get_block(&self, id: BlockId<Types>) -> QueryResult<BlockQueryData<Types>> {
108        self.block_storage
109            .iter()
110            .nth(self.get_block_index(id)?)
111            .context(NotFoundSnafu)?
112            .context(MissingSnafu)
113    }
114
115    fn get_header(&self, id: BlockId<Types>) -> QueryResult<Header<Types>> {
116        self.get_block(id).map(|block| block.header)
117    }
118
119    fn get_block_range<R>(&self, range: R) -> QueryResult<Vec<QueryResult<BlockQueryData<Types>>>>
120    where
121        R: RangeBounds<usize> + Send,
122    {
123        Ok(range_iter(self.block_storage.iter(), range).collect())
124    }

158impl<Types: NodeType> FileSystemStorage<Types>
159where
160    Payload<Types>: QueryablePayload<Types>,
161    Header<Types>: QueryableHeader<Types>,
162{
163    /// Create a new [FileSystemStorage] with storage at `path`.
164    ///
165    /// If there is already data at `path`, it will be archived.
166    ///
167    /// The [FileSystemStorage] will manage its own persistence synchronization.
168    pub async fn create(path: &Path) -> Result<Self, PersistenceError> {
169        let mut loader = AtomicStoreLoader::create(path, "hotshot_data_source")?;
170        loader.retain_archives(1);
171        let data_source = Self::create_with_store(&mut loader).await?;
172        data_source.inner.write().await.top_storage = Some(AtomicStore::open(loader)?);
173        Ok(data_source)
174    }
175
176    /// Open an existing [FileSystemStorage] from storage at `path`.
177    ///
178    /// If there is no data at `path`, a new store will be created.
179    ///
180    /// The [FileSystemStorage] will manage its own persistence synchronization.
181    pub async fn open(path: &Path) -> Result<Self, PersistenceError> {
182        let mut loader = AtomicStoreLoader::load(path, "hotshot_data_source")?;
183        loader.retain_archives(1);
184        let data_source = Self::open_with_store(&mut loader).await?;
185        data_source.inner.write().await.top_storage = Some(AtomicStore::open(loader)?);
186        Ok(data_source)
187    }
188
189    /// Create a new [FileSystemStorage] using a persistent storage loader.
190    ///
191    /// If there is existing data corresponding to the [FileSystemStorage] data structures, it will
192    /// be archived.
193    ///
194    /// The [FileSystemStorage] will register its persistent data structures with `loader`. The
195    /// caller is responsible for creating an [AtomicStore] from `loader` and managing
196    /// synchronization of the store.
197    pub async fn create_with_store(
198        loader: &mut AtomicStoreLoader,
199    ) -> Result<Self, PersistenceError> {
200        Ok(Self {
201            inner: RwLock::new(FileSystemStorageInner {
202                index_by_leaf_hash: Default::default(),
203                index_by_block_hash: Default::default(),
204                index_by_payload_hash: Default::default(),
205                index_by_txn_hash: Default::default(),
206                index_by_time: Default::default(),
207                num_transactions: 0,
208                payload_size: 0,
209                top_storage: None,
210                leaf_storage: LedgerLog::create(loader, "leaves", CACHED_LEAVES_COUNT)?,
211                block_storage: LedgerLog::create(loader, "blocks", CACHED_BLOCKS_COUNT)?,
212                vid_storage: LedgerLog::create(loader, "vid_common", CACHED_VID_COMMON_COUNT)?,
213                state_cert_storage: LedgerLog::create(
214                    loader,
215                    "state_cert",
216                    CACHED_STATE_CERT_COUNT,
217                )?,
218            }),
219            metrics: Default::default(),
220        })
221    }
222
223    /// Open an existing [FileSystemStorage] using a persistent storage loader.
224    ///
225    /// If there is no existing data corresponding to the [FileSystemStorage] data structures, a new
226    /// store will be created.
227    ///
228    /// The [FileSystemStorage] will register its persistent data structures with `loader`. The
229    /// caller is responsible for creating an [AtomicStore] from `loader` and managing
230    /// synchronization of the store.
231    pub async fn open_with_store(loader: &mut AtomicStoreLoader) -> Result<Self, PersistenceError> {
232        let leaf_storage =
233            LedgerLog::<LeafQueryData<Types>>::open(loader, "leaves", CACHED_LEAVES_COUNT)?;
234        let block_storage =
235            LedgerLog::<BlockQueryData<Types>>::open(loader, "blocks", CACHED_BLOCKS_COUNT)?;
236        let vid_storage = LedgerLog::<(VidCommonQueryData<Types>, Option<VidShare>)>::open(
237            loader,
238            "vid_common",
239            CACHED_VID_COMMON_COUNT,
240        )?;
241        let state_cert_storage = LedgerLog::<StateCertQueryData<Types>>::open(
242            loader,
243            "state_cert",
244            CACHED_STATE_CERT_COUNT,
245        )?;
246
247        let mut index_by_block_hash = HashMap::new();
248        let mut index_by_payload_hash = HashMap::new();
249        let mut index_by_time = BTreeMap::<u64, Vec<u64>>::new();
250        let index_by_leaf_hash = leaf_storage
251            .iter()
252            .flatten()
253            .map(|leaf| {
254                update_index_by_hash(&mut index_by_block_hash, leaf.block_hash(), leaf.height());
255                update_index_by_hash(
256                    &mut index_by_payload_hash,
257                    leaf.payload_hash(),
258                    leaf.height(),
259                );
260                index_by_time
261                    .entry(leaf.header().timestamp())
262                    .or_default()
263                    .push(leaf.height());
264                (leaf.hash(), leaf.height())
265            })
266            .collect();
267
268        let mut index_by_txn_hash = HashMap::new();
269        let mut num_transactions = 0;
270        let mut payload_size = 0;
271        for block in block_storage.iter().flatten() {
272            num_transactions += block.len();
273            payload_size += block.size() as usize;
274
275            let height = block.height();
276            for (_, txn) in block.enumerate() {
277                update_index_by_hash(&mut index_by_txn_hash, txn.commit(), height);
278            }
279        }
280
281        Ok(Self {
282            inner: RwLock::new(FileSystemStorageInner {
283                index_by_leaf_hash,
284                index_by_block_hash,
285                index_by_payload_hash,
286                index_by_txn_hash,
287                index_by_time,
288                num_transactions,
289                payload_size,
290                leaf_storage,
291                block_storage,
292                vid_storage,
293                state_cert_storage,
294                top_storage: None,
295            }),
296            metrics: Default::default(),
297        })
298    }
299
300    /// Advance the version of the persistent store without committing changes to persistent state.
301    pub async fn skip_version(&self) -> Result<(), PersistenceError> {
302        let mut inner = self.inner.write().await;
303        inner.leaf_storage.skip_version()?;
304        inner.block_storage.skip_version()?;
305        inner.vid_storage.skip_version()?;
306        inner.state_cert_storage.skip_version()?;
307        if let Some(store) = &mut inner.top_storage {
308            store.commit_version()?;
309        }
310        Ok(())
311    }
312}

414fn range_iter<T>(
415    mut iter: Iter<'_, T>,
416    range: impl RangeBounds<usize>,
417) -> impl '_ + Iterator<Item = QueryResult<T>>
418where
419    T: Clone + Serialize + DeserializeOwned,
420{
421    let start = range.start_bound().cloned();
422    let end = range.end_bound().cloned();
423
424    // Advance the underlying iterator to the start of the range.
425    let pos = match start {
426        Bound::Included(n) => {
427            if n > 0 {
428                iter.nth(n - 1);
429            }
430            n
431        },
432        Bound::Excluded(n) => {
433            iter.nth(n);
434            n + 1
435        },
436        Bound::Unbounded => 0,
437    };
438
439    itertools::unfold((iter, end, pos), |(iter, end, pos)| {
440        // Check if we have reached the end of the range.
441        let reached_end = match end {
442            Bound::Included(n) => pos > n,
443            Bound::Excluded(n) => pos >= n,
444            Bound::Unbounded => false,
445        };
446        if reached_end {
447            return None;
448        }
449        let opt = iter.next()?;
450        *pos += 1;
451        Some(opt.context(MissingSnafu))
452    })
453}

622impl<Types: NodeType> UpdateAvailabilityStorage<Types>
623    for Transaction<RwLockWriteGuard<'_, FileSystemStorageInner<Types>>>
624where
625    Payload<Types>: QueryablePayload<Types>,
626    Header<Types>: QueryableHeader<Types>,
627{
628    async fn insert_leaf(&mut self, leaf: LeafQueryData<Types>) -> anyhow::Result<()> {
629        self.inner
630            .leaf_storage
631            .insert(leaf.height() as usize, leaf.clone())?;
632        self.inner
633            .index_by_leaf_hash
634            .insert(leaf.hash(), leaf.height());
635        update_index_by_hash(
636            &mut self.inner.index_by_block_hash,
637            leaf.block_hash(),
638            leaf.height(),
639        );
640        update_index_by_hash(
641            &mut self.inner.index_by_payload_hash,
642            leaf.payload_hash(),
643            leaf.height(),
644        );
645        self.inner
646            .index_by_time
647            .entry(leaf.header().timestamp())
648            .or_default()
649            .push(leaf.height());
650        Ok(())
651    }
652
653    async fn insert_block(&mut self, block: BlockQueryData<Types>) -> anyhow::Result<()> {
654        if !self
655            .inner
656            .block_storage
657            .insert(block.height() as usize, block.clone())?
658        {
659            // The block was already present.
660            return Ok(());
661        }
662        self.inner.num_transactions += block.len();
663        self.inner.payload_size += block.size() as usize;
664        for (_, txn) in block.enumerate() {
665            update_index_by_hash(
666                &mut self.inner.index_by_txn_hash,
667                txn.commit(),
668                block.height(),
669            );
670        }
671        Ok(())
672    }
673
674    async fn insert_vid(
675        &mut self,
676        common: VidCommonQueryData<Types>,
677        share: Option<VidShare>,
678    ) -> anyhow::Result<()> {
679        self.inner
680            .vid_storage
681            .insert(common.height() as usize, (common, share))?;
682        Ok(())
683    }
684
685    async fn insert_state_cert(
686        &mut self,
687        state_cert: StateCertQueryData<Types>,
688    ) -> anyhow::Result<()> {
689        self.inner
690            .state_cert_storage
691            .insert(state_cert.0.epoch.u64() as usize, state_cert)?;
692        Ok(())
693    }
694}
695
696/// Update an index mapping hashes of objects to their positions in the ledger.
697///
698/// This function will insert the mapping from `hash` to `pos` into `index`, _unless_ there is
699/// already an entry for `hash` at an earlier position in the ledger.
700fn update_index_by_hash<H: Eq + Hash, P: Ord>(index: &mut HashMap<H, P>, hash: H, pos: P) {
701    match index.entry(hash) {
702        Entry::Occupied(mut e) => {
703            if &pos < e.get() {
704                // Overwrite the existing entry if the new object was sequenced first.
705                e.insert(pos);
706            }
707        },
708        Entry::Vacant(e) => {
709            e.insert(pos);
710        },
711    }
712}

715impl<Types, T> NodeStorage<Types> for Transaction<T>
716where
717    Types: NodeType,
718    Payload<Types>: QueryablePayload<Types>,
719    Header<Types>: QueryableHeader<Types>,
720    T: Revert + Deref<Target = FileSystemStorageInner<Types>> + Send,
721{
722    async fn block_height(&mut self) -> QueryResult<usize> {
723        Ok(self.inner.leaf_storage.iter().len())
724    }
725
726    async fn count_transactions_in_range(
727        &mut self,
728        range: impl RangeBounds<usize> + Send,
729    ) -> QueryResult<usize> {
730        if !matches!(range.start_bound(), Bound::Unbounded | Bound::Included(0))
731            || !matches!(range.end_bound(), Bound::Unbounded)
732        {
733            return Err(QueryError::Error {
734                message: "partial aggregates are not supported with file system backend".into(),
735            });
736        }
737
738        Ok(self.inner.num_transactions)
739    }
740
741    async fn payload_size_in_range(
742        &mut self,
743        range: impl RangeBounds<usize> + Send,
744    ) -> QueryResult<usize> {
745        if !matches!(range.start_bound(), Bound::Unbounded | Bound::Included(0))
746            || !matches!(range.end_bound(), Bound::Unbounded)
747        {
748            return Err(QueryError::Error {
749                message: "partial aggregates are not supported with file system backend".into(),
750            });
751        }
752
753        Ok(self.inner.payload_size)
754    }
755
756    async fn vid_share<ID>(&mut self, id: ID) -> QueryResult<VidShare>
757    where
758        ID: Into<BlockId<Types>> + Send + Sync,
759    {
760        self.inner
761            .vid_storage
762            .iter()
763            .nth(self.inner.get_block_index(id.into())?)
764            .context(NotFoundSnafu)?
765            .context(MissingSnafu)?
766            .1
767            .context(MissingSnafu)
768    }
769
770    async fn sync_status(&mut self) -> QueryResult<SyncStatus> {
771        let height = self.inner.leaf_storage.iter().len();
772
773        // The number of missing VID common is just the number of completely missing VID
774        // entries, since every entry we have is guaranteed to have the common data.
775        let missing_vid = self.inner.vid_storage.missing(height);
776        // Missing shares includes the completely missing VID entries, plus any entry which
777        // is _not_ messing but which has a null share.
778        let null_vid_shares: usize = self
779            .inner
780            .vid_storage
781            .iter()
782            .map(|res| if matches!(res, Some((_, None))) { 1 } else { 0 })
783            .sum();
784        Ok(SyncStatus {
785            missing_blocks: self.inner.block_storage.missing(height),
786            missing_leaves: self.inner.leaf_storage.missing(height),
787            missing_vid_common: missing_vid,
788            missing_vid_shares: missing_vid + null_vid_shares,
789            pruned_height: None,
790        })
791    }
792
793    async fn get_header_window(
794        &mut self,
795        start: impl Into<WindowStart<Types>> + Send + Sync,
796        end: u64,
797        limit: usize,
798    ) -> QueryResult<TimeWindowQueryData<Header<Types>>> {
799        let first_block = match start.into() {
800            WindowStart::Height(h) => h,
801            WindowStart::Hash(h) => self.inner.get_header(h.into())?.block_number(),
802            WindowStart::Time(t) => {
803                // Find the minimum timestamp which is at least `t`, and all the blocks with
804                // that timestamp.
805                let blocks = self
806                    .inner
807                    .index_by_time
808                    .range(t..)
809                    .next()
810                    .context(NotFoundSnafu)?
811                    .1;
812                // Multiple blocks can have the same timestamp (when truncated to seconds);
813                // we want the first one. It is an invariant that any timestamp which has an
814                // entry in `index_by_time` has a non-empty list associated with it, so this
815                // indexing is safe.
816                blocks[0]
817            },
818        } as usize;
819
820        let mut res = TimeWindowQueryData::default();
821
822        // Include the block just before the start of the window, if there is one.
823        if first_block > 0 {
824            res.prev = Some(self.inner.get_header((first_block - 1).into())?);
825        }
826
827        // Add blocks to the window, starting from `first_block`, until we reach the end of
828        // the requested time window.
829        for block in self.inner.get_block_range(first_block..)? {
830            let header = block?.header().clone();
831            if header.timestamp() >= end {
832                res.next = Some(header);
833                break;
834            }
835            res.window.push(header);
836            if res.window.len() >= limit {
837                break;
838            }
839        }
840
841        Ok(res)
842    }
843}
