10/// Count transactions and bytes confirmed in a given namespace.
11#[derive(Debug, Parser)]
12pub struct Options {
13    /// Block to start counting from.
14    #[clap(short, long, default_value = "0")]
15    from_block: usize,
16
17    /// Last block to count (inclusive).
18    ///
19    /// If not specified, will count until the end of the chain at the moment the program is
20    /// invoked.
21    #[clap(short, long)]
22    to_block: Option<usize>,
23
24    /// Namespace to aggregate.
25    #[clap(short, long)]
26    namespace: u64,
27
28    /// Number of parallel tasks to run.
29    ///
30    /// Since the process is mostly I/O bound, increasing this can dramatically speed up
31    /// aggregation.
32    #[clap(short, long, default_value = "1")]
33    jobs: usize,
34
35    /// Espresso query service URL.
36    url: Url,
37}
38
39pub async fn run(opt: Options) -> anyhow::Result<()> {
40    let ns = NamespaceId::from(opt.namespace);
41    let client = surf_disco::Client::<hotshot_query_service::Error, SequencerApiVersion>::new(
42        opt.url.clone(),
43    );
44
45    // Convert optional closed [from, to] interval to a semi-open [start, end) interval, which makes
46    // arithmetic simpler later on.
47    let start = opt.from_block;
48    let end = match opt.to_block {
49        Some(to) => to + 1,
50        None => client.get("status/block-height").send().await?,
51    };
52    ensure!(end > start, "to-block < from-block");
53
54    let tasks = (0..opt.jobs).map(|i| {
55        let chunk_size = (end - start) / opt.jobs;
56        let chunk_start = start + i * chunk_size;
57        let chunk_end = min(chunk_start + chunk_size, end);
58        process_chunk(opt.url.clone(), ns, chunk_start, chunk_end)
59    });
60    let chunks = try_join_all(tasks).await?;
61
62    let mut num_txs = 0;
63    let mut bytes = 0;
64    for (chunk_txs, chunk_bytes) in chunks {
65        num_txs += chunk_txs;
66        bytes += chunk_bytes;
67    }
68
69    println!("Scanned range [{start}, {end}) for namespace {ns}");
70    println!("{num_txs} transactions");
71    println!("{bytes} bytes");
72
73    Ok(())
74}

76async fn process_chunk(
77    url: Url,
78    ns: NamespaceId,
79    start: usize,
80    end: usize,
81) -> anyhow::Result<(usize, usize)> {
82    let client = surf_disco::Client::<hotshot_query_service::Error, SequencerApiVersion>::new(url);
83    let mut headers = client
84        .socket(&format!("availability/stream/headers/{start}"))
85        .subscribe::<Header>()
86        .await?
87        .take(end - start);
88
89    let mut num_txs = 0;
90    let mut bytes = 0;
91    while let Some(header) = headers.next().await {
92        let header = header?;
93        let height = header.height();
94
95        if header.ns_table().find_ns_id(&ns).is_none() {
96            tracing::debug!(height, "trivial block");
97            continue;
98        }
99        tracing::info!(height, "non-trivial block");
100
101        let payload: NamespaceProofQueryData = client
102            .get(&format!("availability/block/{height}/namespace/{ns}",))
103            .send()
104            .await
105            .context(format!("requesting namespace for block {height}"))?;
106        num_txs += payload.transactions.len();
107        bytes += payload
108            .transactions
109            .iter()
110            .map(|tx| tx.payload().len())
111            .sum::<usize>();
112    }
113
114    Ok((num_txs, bytes))
115}
