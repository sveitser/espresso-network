187impl StateKeyPair {
188    /// Generate key pairs from private signing keys
189    #[must_use]
190    pub fn from_sign_key(sk: StateSignKey) -> Self {
191        Self(schnorr::KeyPair::<Config>::from(sk))
192    }
193
194    /// Generate key pairs from `thread_rng()`
195    #[must_use]
196    pub fn generate() -> StateKeyPair {
197        schnorr::KeyPair::generate(&mut rand::thread_rng()).into()
198    }
199
200    /// Generate key pairs from seed
201    #[must_use]
202    pub fn generate_from_seed(seed: [u8; 32]) -> StateKeyPair {
203        schnorr::KeyPair::generate(&mut ChaCha20Rng::from_seed(seed)).into()
204    }
205
206    /// Generate key pairs from an index and a seed
207    #[must_use]
208    pub fn generate_from_seed_indexed(seed: [u8; 32], index: u64) -> StateKeyPair {
209        let mut hasher = blake3::Hasher::new();
210        hasher.update(&seed);
211        hasher.update(&index.to_le_bytes());
212        let new_seed = *hasher.finalize().as_bytes();
213        Self::generate_from_seed(new_seed)
214    }
215}

311pub fn hash_bytes_to_field<F: RescueParameter>(bytes: &[u8]) -> Result<F, RescueError> {
312    // make sure that `mod_order` won't happen.
313    let bytes_len = (<F as PrimeField>::MODULUS_BIT_SIZE.div_ceil(8) - 1) as usize;
314    let elem = bytes
315        .chunks(bytes_len)
316        .map(F::from_le_bytes_mod_order)
317        .collect::<Vec<_>>();
318    Ok(VariableLengthRescueCRHF::<_, 1>::evaluate(elem)?[0])
319}

321/// This trait is for light client use. It converts the stake table items into
322/// field elements. These items will then be digested into a part of the light client state.
323pub trait ToFieldsLightClientCompat {
324    const SIZE: usize;
325    fn to_fields(&self) -> Vec<CircuitField>;
326}
327
328impl ToFieldsLightClientCompat for StateVerKey {
329    const SIZE: usize = 2;
330    /// This should be compatible with our legacy implementation.
331    fn to_fields(&self) -> Vec<CircuitField> {
332        let p = self.to_affine();
333        vec![p.x, p.y]
334    }
335}
336
337impl ToFieldsLightClientCompat for BLSPubKey {
338    const SIZE: usize = 3;
339    /// This should be compatible with our legacy implementation.
340    fn to_fields(&self) -> Vec<CircuitField> {
341        match to_bytes!(&self.to_affine()) {
342            Ok(bytes) => {
343                vec![
344                    CircuitField::from_le_bytes_mod_order(&bytes[..31]),
345                    CircuitField::from_le_bytes_mod_order(&bytes[31..62]),
346                    CircuitField::from_le_bytes_mod_order(&bytes[62..]),
347                ]
348            },
349            Err(_) => unreachable!(),
350        }
351    }
352}

354#[inline]
355/// A helper function to compute the quorum threshold given a total amount of stake.
356/// TODO: clean up <https://github.com/EspressoSystems/espresso-network/issues/2971>
357pub fn one_honest_threshold(total_stake: U256) -> U256 {
358    total_stake / U256::from(3) + U256::from(1)
359}
360
361#[inline]
362/// TODO: clean up <https://github.com/EspressoSystems/espresso-network/issues/2971>
363fn u256_to_field(amount: U256) -> CircuitField {
364    let amount_bytes: [u8; 32] = amount.to_le_bytes();
365    CircuitField::from_le_bytes_mod_order(&amount_bytes)
366}

368/// Given a list of stakers from `PeerConfig`, compute the stake table commitment
369pub fn compute_stake_table_commitment<TYPES: NodeType>(
370    known_nodes_with_stakes: &[PeerConfig<TYPES>],
371    stake_table_capacity: usize,
372) -> StakeTableState {
373    let padding_len = stake_table_capacity - known_nodes_with_stakes.len();
374    let mut bls_preimage = vec![];
375    let mut schnorr_preimage = vec![];
376    let mut amount_preimage = vec![];
377    let mut total_stake = U256::from(0);
378    for peer in known_nodes_with_stakes {
379        bls_preimage.extend(peer.stake_table_entry.public_key().to_fields());
380        schnorr_preimage.extend(peer.state_ver_key.to_fields());
381        amount_preimage.push(u256_to_field(peer.stake_table_entry.stake()));
382        total_stake += peer.stake_table_entry.stake();
383    }
384    bls_preimage.resize(
385        <TYPES::SignatureKey as ToFieldsLightClientCompat>::SIZE * stake_table_capacity,
386        CircuitField::default(),
387    );
388    // Nasty tech debt
389    schnorr_preimage
390        .extend(iter::repeat_n(SchnorrPubKey::default().to_fields(), padding_len).flatten());
391    amount_preimage.resize(stake_table_capacity, CircuitField::default());
392    let threshold = u256_to_field(one_honest_threshold(total_stake));
393    StakeTableState {
394        bls_key_comm: VariableLengthRescueCRHF::<CircuitField, 1>::evaluate(bls_preimage).unwrap()
395            [0],
396        schnorr_key_comm: VariableLengthRescueCRHF::<CircuitField, 1>::evaluate(schnorr_preimage)
397            .unwrap()[0],
398        amount_comm: VariableLengthRescueCRHF::<CircuitField, 1>::evaluate(amount_preimage)
399            .unwrap()[0],
400        threshold,
401    }
