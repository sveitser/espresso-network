98#[allow(clippy::type_complexity)]
99pub fn gen_node_lists<TYPES: NodeType>(
100    num_staked_nodes: u64,
101    num_da_nodes: u64,
102) -> (Vec<PeerConfig<TYPES>>, Vec<PeerConfig<TYPES>>) {
103    let mut staked_nodes = Vec::new();
104    let mut da_nodes = Vec::new();
105
106    for n in 0..num_staked_nodes {
107        let validator_config: ValidatorConfig<TYPES> = ValidatorConfig::generated_from_seed_indexed(
108            [0u8; 32],
109            n,
110            U256::from(1),
111            n < num_da_nodes,
112        );
113
114        let peer_config = validator_config.public_config();
115        staked_nodes.push(peer_config.clone());
116
117        if n < num_da_nodes {
118            da_nodes.push(peer_config)
119        }
120    }
121
122    (staked_nodes, da_nodes)
123}

168pub fn nonempty_block_threshold(threshold: (u64, u64)) -> TransactionValidator {
169    Arc::new(move |transactions| {
170        if matches!(threshold, (0, _)) {
171            return Ok(());
172        }
173
174        let blocks: Vec<_> = transactions.iter().filter(|(view, _)| *view != 0).collect();
175
176        let num_blocks = blocks.len() as u64;
177        let mut num_nonempty_blocks = 0;
178
179        ensure!(num_blocks > 0, "Failed to commit any non-genesis blocks");
180
181        for (_, num_transactions) in blocks {
182            if *num_transactions > 0 {
183                num_nonempty_blocks += 1;
184            }
185        }
186
187        ensure!(
188          // i.e. num_nonempty_blocks / num_blocks >= threshold.0 / threshold.1
189          num_nonempty_blocks * threshold.1 >= threshold.0 * num_blocks,
190          "Failed to meet nonempty block threshold of {}/{}; got {num_nonempty_blocks} nonempty blocks out of a total of {num_blocks}", threshold.0, threshold.1
191        );
192
193        Ok(())
194    })
195}

197pub fn nonempty_block_limit(limit: (u64, u64)) -> TransactionValidator {
198    Arc::new(move |transactions| {
199        if matches!(limit, (_, 0)) {
200            return Ok(());
201        }
202
203        let blocks: Vec<_> = transactions.iter().filter(|(view, _)| *view != 0).collect();
204
205        let num_blocks = blocks.len() as u64;
206        let mut num_nonempty_blocks = 0;
207
208        ensure!(num_blocks > 0, "Failed to commit any non-genesis blocks");
209
210        for (_, num_transactions) in blocks {
211            if *num_transactions > 0 {
212                num_nonempty_blocks += 1;
213            }
214        }
215
216        ensure!(
217          // i.e. num_nonempty_blocks / num_blocks <= limit.0 / limit.1
218          num_nonempty_blocks * limit.1 <= limit.0 * num_blocks,
219          "Exceeded nonempty block limit of {}/{}; got {num_nonempty_blocks} nonempty blocks out of a total of {num_blocks}", limit.0, limit.1
220        );
221
222        Ok(())
223    })

233pub async fn create_test_handle<
234    TYPES: NodeType<InstanceState = TestInstanceState>,
235    I: NodeImplementation<TYPES>,
236    V: Versions,
237>(
238    metadata: TestDescription<TYPES, I, V>,
239    node_id: u64,
240    network: Network<TYPES, I>,
241    memberships: Arc<RwLock<TYPES::Membership>>,
242    config: HotShotConfig<TYPES>,
243    storage: I::Storage,
244    marketplace_config: MarketplaceConfig<TYPES, I>,
245) -> SystemContextHandle<TYPES, I, V> {
246    let initializer = HotShotInitializer::<TYPES>::from_genesis::<V>(
247        TestInstanceState::new(metadata.async_delay_config),
248        metadata.test_config.epoch_height,
249        metadata.test_config.epoch_start_block,
250        vec![],
251    )
252    .await
253    .unwrap();
254
255    // See whether or not we should be DA
256    let is_da = node_id < config.da_staked_committee_size as u64;
257
258    let validator_config: ValidatorConfig<TYPES> =
259        ValidatorConfig::generated_from_seed_indexed([0u8; 32], node_id, U256::from(1), is_da);
260
261    // Get key pair for certificate aggregation
262    let private_key = validator_config.private_key.clone();
263    let public_key = validator_config.public_key.clone();
264    let state_private_key = validator_config.state_private_key.clone();
265    let membership_coordinator = EpochMembershipCoordinator::new(
266        memberships,
267        Some(storage_add_drb_result(storage.clone())),
268        config.epoch_height,
269    );
270
271    let behaviour = (metadata.behaviour)(node_id);
272    match behaviour {
273        Behaviour::ByzantineTwins(state) => {
274            let state = Box::leak(state);
275            let (left_handle, _right_handle) = state
276                .spawn_twin_handles(
277                    public_key,
278                    private_key,
279                    state_private_key,
280                    node_id,
281                    config,
282                    membership_coordinator,
283                    network,
284                    initializer,
285                    ConsensusMetricsValue::default(),
286                    storage,
287                    marketplace_config,
288                )
289                .await;
290
291            left_handle
292        },
293        Behaviour::Byzantine(state) => {
294            let state = Box::leak(state);
295            state
296                .spawn_handle(
297                    public_key,
298                    private_key,
299                    state_private_key,
300                    node_id,
301                    config,
302                    membership_coordinator,
303                    network,
304                    initializer,
305                    ConsensusMetricsValue::default(),
306                    storage,
307                    marketplace_config,
308                )
309                .await
310        },
311        Behaviour::Standard => {
312            let hotshot = SystemContext::<TYPES, I, V>::new(
313                public_key,
314                private_key,
315                state_private_key,
316                node_id,
317                config,
318                membership_coordinator,
319                network,
320                initializer,
321                ConsensusMetricsValue::default(),
322                storage,
323                marketplace_config,
324            )
325            .await;
326
327            hotshot.run_tasks().await
328        },
329    }
330}

534impl<
535        TYPES: NodeType<InstanceState = TestInstanceState>,
536        I: TestableNodeImplementation<TYPES>,
537        V: Versions,
538    > TestDescription<TYPES, I, V>
539where
540    I: NodeImplementation<TYPES, AuctionResultsProvider = TestAuctionResultsProvider<TYPES>>,
541{
542    /// turn a description of a test (e.g. a [`TestDescription`]) into
543    /// a [`TestLauncher`] that can be used to launch the test.
544    /// # Panics
545    /// if some of the configuration values are zero
546    pub fn gen_launcher(self) -> TestLauncher<TYPES, I, V> {
547        self.gen_launcher_with_tasks(vec![])
548    }
549
550    /// turn a description of a test (e.g. a [`TestDescription`]) into
551    /// a [`TestLauncher`] that can be used to launch the test, with
552    /// additional testing tasks to run in test harness
553    /// # Panics
554    /// if some of the configuration values are zero
555    #[must_use]
556    pub fn gen_launcher_with_tasks(
557        self,
558        additional_test_tasks: Vec<Box<dyn TestTaskStateSeed<TYPES, I, V>>>,
559    ) -> TestLauncher<TYPES, I, V> {
560        let TestDescription {
561            timing_data,
562            unreliable_network,
563            test_config,
564            ..
565        } = self.clone();
566
567        let num_nodes_with_stake = test_config.num_nodes_with_stake.into();
568        let num_bootstrap_nodes = test_config.num_bootstrap;
569        let da_staked_committee_size = test_config.da_staked_committee_size;
570
571        let validator_config = Rc::new(move |node_id| {
572            ValidatorConfig::<TYPES>::generated_from_seed_indexed(
573                [0u8; 32],
574                node_id,
575                U256::from(1),
576                // This is the config for node 0
577                node_id < test_config.da_staked_committee_size as u64,
578            )
579        });
580
581        let hotshot_config = Rc::new(move |_| test_config.clone());
582        let TimingData {
583            next_view_timeout,
584            builder_timeout,
585            data_request_delay,
586            secondary_network_delay,
587            view_sync_timeout,
588        } = timing_data;
589        // TODO this should really be using the timing config struct
590        let mod_hotshot_config = move |hotshot_config: &mut HotShotConfig<TYPES>| {
591            hotshot_config.next_view_timeout = next_view_timeout;
592            hotshot_config.builder_timeout = builder_timeout;
593            hotshot_config.data_request_delay = data_request_delay;
594            hotshot_config.view_sync_timeout = view_sync_timeout;
595        };
596
597        let metadata = self.clone();
598        TestLauncher {
599            resource_generators: ResourceGenerators {
600                channel_generator: <I as TestableNodeImplementation<TYPES>>::gen_networks(
601                    num_nodes_with_stake,
602                    num_bootstrap_nodes,
603                    da_staked_committee_size,
604                    unreliable_network,
605                    secondary_network_delay,
606                ),
607                storage: Rc::new(move |_| {
608                    let mut storage = TestStorage::<TYPES>::default();
609                    // update storage impl to use settings delay option
610                    storage.delay_config = metadata.async_delay_config.clone();
611                    storage
612                }),
613                hotshot_config,
614                validator_config,
615                marketplace_config: Rc::new(|_| MarketplaceConfig::<TYPES, I> {
616                    auction_results_provider: TestAuctionResultsProvider::<TYPES>::default().into(),
617                    fallback_builder_url: Url::parse("http://localhost:9999").unwrap(),
618                }),
619            },
620            metadata: self,
621            additional_test_tasks,
622        }
623        .map_hotshot_config(mod_hotshot_config)
624    }
