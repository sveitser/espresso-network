55/// Spinning task state
56pub struct SpinningTask<
57    TYPES: NodeType,
58    N: ConnectedNetwork<TYPES::SignatureKey>,
59    I: TestableNodeImplementation<TYPES>,
60    V: Versions,
61> {
62    /// epoch height
63    pub epoch_height: u64,
64    /// Epoch start block
65    pub epoch_start_block: u64,
66    /// Saved epoch information. This must be sorted ascending by epoch.
67    pub start_epoch_info: Vec<InitializerEpochInfo<TYPES>>,
68    /// handle to the nodes
69    pub(crate) handles: Arc<RwLock<Vec<Node<TYPES, I, V>>>>,
70    /// late start nodes
71    pub(crate) late_start: HashMap<u64, LateStartNode<TYPES, I, V>>,
72    /// time based changes
73    pub(crate) changes: BTreeMap<TYPES::View, Vec<ChangeNode>>,
74    /// most recent view seen by spinning task
75    pub(crate) latest_view: Option<TYPES::View>,
76    /// Last decided leaf that can be used as the anchor leaf to initialize the node.
77    pub(crate) last_decided_leaf: Leaf2<TYPES>,
78    /// Highest qc seen in the test for restarting nodes
79    pub(crate) high_qc: QuorumCertificate2<TYPES>,
80    /// Next epoch highest qc seen in the test for restarting nodes
81    pub(crate) next_epoch_high_qc: Option<NextEpochQuorumCertificate2<TYPES>>,
82    /// Add specified delay to async calls
83    pub(crate) async_delay_config: DelayConfig,
84    /// Context stored for nodes to be restarted with
85    pub(crate) restart_contexts: HashMap<usize, RestartContext<TYPES, N, I, V>>,
86    /// Generate network channel for restart nodes
87    pub(crate) channel_generator: AsyncGenerator<Network<TYPES, I>>,
88    /// The light client state update certificate
89    pub(crate) state_cert: Option<LightClientStateUpdateCertificate<TYPES>>,
90}

92#[async_trait]
93impl<
94        TYPES: NodeType<
95            InstanceState = TestInstanceState,
96            ValidatedState = TestValidatedState,
97            BlockHeader = TestBlockHeader,
98        >,
99        I: TestableNodeImplementation<TYPES>,
100        N: ConnectedNetwork<TYPES::SignatureKey>,
101        V: Versions,
102    > TestTaskState for SpinningTask<TYPES, N, I, V>
103where
104    I: TestableNodeImplementation<TYPES>,
105    I: NodeImplementation<
106        TYPES,
107        Network = N,
108        Storage = TestStorage<TYPES>,
109        AuctionResultsProvider = TestAuctionResultsProvider<TYPES>,
110    >,
111{
112    type Event = Event<TYPES>;
113    type Error = Error;
114
115    async fn handle_event(&mut self, (message, _id): (Self::Event, usize)) -> Result<()> {
116        let Event { view_number, event } = message;
117
118        if let EventType::Decide {
119            leaf_chain,
120            qc: _,
121            block_size: _,
122        } = event
123        {
124            let leaf = leaf_chain.first().unwrap().leaf.clone();
125            if leaf.view_number() > self.last_decided_leaf.view_number() {
126                self.last_decided_leaf = leaf;
127            }
128        } else if let EventType::QuorumProposal {
129            proposal,
130            sender: _,
131        } = event
132        {
133            if proposal.data.justify_qc().view_number() > self.high_qc.view_number() {
134                self.high_qc = proposal.data.justify_qc().clone();
135            }
136        }
137
138        let mut new_nodes = vec![];
139        let mut new_networks = vec![];
140        // if we have not seen this view before
141        if self.latest_view.is_none() || view_number > self.latest_view.unwrap() {
142            // perform operations on the nodes
143            if let Some(operations) = self.changes.remove(&view_number) {
144                for ChangeNode { idx, updown } in operations {
145                    match updown {
146                        NodeAction::Up => {
147                            let node_id = idx.try_into().unwrap();
148                            if let Some(node) = self.late_start.remove(&node_id) {
149                                tracing::error!("Node {} spinning up late", idx);
150                                let network = if let Some(network) = node.network {
151                                    network
152                                } else {
153                                    let generated_network = (self.channel_generator)(node_id).await;
154                                    generated_network.wait_for_ready().await;
155                                    generated_network
156                                };
157                                let node_id = idx.try_into().unwrap();
158                                let context = match node.context {
159                                    LateNodeContext::InitializedContext(context) => context,
160                                    // Node not initialized. Initialize it
161                                    // based on the received leaf.
162                                    LateNodeContext::UninitializedContext(late_context_params) => {
163                                        // We'll deconstruct the individual terms here.
164                                        let LateNodeContextParameters {
165                                            storage,
166                                            memberships,
167                                            config,
168                                            marketplace_config,
169                                        } = late_context_params;
170
171                                        let initializer = HotShotInitializer::<TYPES>::load(
172                                            TestInstanceState::new(self.async_delay_config.clone()),
173                                            self.epoch_height,
174                                            self.epoch_start_block,
175                                            self.start_epoch_info.clone(),
176                                            self.last_decided_leaf.clone(),
177                                            (
178                                                TYPES::View::genesis(),
179                                                genesis_epoch_from_version::<V, TYPES>(),
180                                            ),
181                                            (self.high_qc.clone(), self.next_epoch_high_qc.clone()),
182                                            BTreeMap::new(),
183                                            BTreeMap::new(),
184                                            None,
185                                            self.state_cert.clone(),
186                                        );
187                                        // We assign node's public key and stake value rather than read from config file since it's a test
188                                        let validator_config =
189                                            ValidatorConfig::generated_from_seed_indexed(
190                                                [0u8; 32],
191                                                node_id,
192                                                U256::from(1),
193                                                // For tests, make the node DA based on its index
194                                                node_id < config.da_staked_committee_size as u64,
195                                            );
196
197                                        TestRunner::add_node_with_config(
198                                            node_id,
199                                            network.clone(),
200                                            memberships,
201                                            initializer,
202                                            config,
203                                            validator_config,
204                                            storage,
205                                            marketplace_config,
206                                        )
207                                        .await
208                                    },
209                                    LateNodeContext::Restart => {
210                                        panic!("Cannot spin up a node with Restart context")
211                                    },
212                                };
213
214                                let handle = context.run_tasks().await;
215
216                                // Create the node and add it to the state, so we can shut them
217                                // down properly later to avoid the overflow error in the overall
218                                // safety task.
219                                let node = Node {
220                                    node_id,
221                                    network,
222                                    handle,
223                                };
224                                node.handle.hotshot.start_consensus().await;
225
226                                self.handles.write().await.push(node);
227                            }
228                        },
229                        NodeAction::Down => {
230                            if let Some(node) = self.handles.write().await.get_mut(idx) {
231                                tracing::error!("Node {} shutting down", idx);
232                                node.handle.shut_down().await;
233                            }
234                        },
235                        NodeAction::RestartDown(delay_views) => {
236                            let node_id = idx.try_into().unwrap();
237                            if let Some(node) = self.handles.write().await.get_mut(idx) {
238                                tracing::error!("Node {} shutting down", idx);
239                                node.handle.shut_down().await;
240                                // For restarted nodes generate the network on correct view
241                                let generated_network = (self.channel_generator)(node_id).await;
242
243                                let Some(LateStartNode {
244                                    network: _,
245                                    context: LateNodeContext::Restart,
246                                }) = self.late_start.get(&node_id)
247                                else {
248                                    panic!("Restarted Nodes must have an uninitialized context");
249                                };
250
251                                let storage = node.handle.storage().clone();
252                                let memberships = node.handle.membership_coordinator.clone();
253                                let config = node.handle.hotshot.config.clone();
254                                let marketplace_config =
255                                    node.handle.hotshot.marketplace_config.clone();
256
257                                let next_epoch_high_qc = storage.next_epoch_high_qc_cloned().await;
258                                let start_view = storage.last_actioned_view().await;
259                                let start_epoch = storage.last_actioned_epoch().await;
260                                let high_qc = storage.high_qc_cloned().await.unwrap_or(
261                                    QuorumCertificate2::genesis::<V>(
262                                        &TestValidatedState::default(),
263                                        &TestInstanceState::default(),
264                                    )
265                                    .await,
266                                );
267                                let state_cert = storage.state_cert_cloned().await;
268                                let saved_proposals = storage.proposals_cloned().await;
269                                let mut vid_shares = BTreeMap::new();
270                                for (view, hash_map) in storage.vids_cloned().await {
271                                    let mut converted_hash_map = HashMap::new();
272                                    for (key, proposal) in hash_map {
273                                        converted_hash_map
274                                            .entry(key)
275                                            .or_insert_with(BTreeMap::new)
276                                            .insert(
277                                                proposal.data.target_epoch,
278                                                convert_proposal(proposal),
279                                            );
280                                    }
281                                    vid_shares.insert(view, converted_hash_map);
282                                }
283                                let decided_upgrade_certificate =
284                                    storage.decided_upgrade_certificate().await;
285
286                                let initializer = HotShotInitializer::<TYPES>::load(
287                                    TestInstanceState::new(self.async_delay_config.clone()),
288                                    self.epoch_height,
289                                    self.epoch_start_block,
290                                    self.start_epoch_info.clone(),
291                                    self.last_decided_leaf.clone(),
292                                    (start_view, start_epoch),
293                                    (high_qc, next_epoch_high_qc),
294                                    saved_proposals,
295                                    vid_shares,
296                                    decided_upgrade_certificate,
297                                    state_cert,
298                                );
299                                // We assign node's public key and stake value rather than read from config file since it's a test
300                                let validator_config = ValidatorConfig::generated_from_seed_indexed(
301                                    [0u8; 32],
302                                    node_id,
303                                    U256::from(1),
304                                    // For tests, make the node DA based on its index
305                                    node_id < config.da_staked_committee_size as u64,
306                                );
307                                let internal_chan = broadcast(EVENT_CHANNEL_SIZE);
308                                let context =
309                                    TestRunner::<TYPES, I, V, N>::add_node_with_config_and_channels(
310                                        node_id,
311                                        generated_network.clone(),
312                                        Arc::clone(memberships.membership()),
313                                        initializer,
314                                        config,
315                                        validator_config,
316                                        storage.clone(),
317                                        marketplace_config.clone(),
318                                        internal_chan,
319                                        (
320                                            node.handle.external_channel_sender(),
321                                            node.handle.event_stream_known_impl().new_receiver(),
322                                        ),
323                                    )
324                                    .await;
325                                if delay_views == 0 {
326                                    new_nodes.push((context, idx));
327                                    new_networks.push(generated_network.clone());
328                                } else {
329                                    let up_view = view_number + delay_views;
330                                    let change = ChangeNode {
331                                        idx,
332                                        updown: NodeAction::RestartUp,
333                                    };
334                                    self.changes.entry(up_view).or_default().push(change);
335                                    let new_ctx = RestartContext {
336                                        context,
337                                        network: generated_network.clone(),
338                                    };
339                                    self.restart_contexts.insert(idx, new_ctx);
340                                }
341                            }
342                        },
343                        NodeAction::RestartUp => {
344                            if let Some(ctx) = self.restart_contexts.remove(&idx) {
345                                new_nodes.push((ctx.context, idx));
346                                new_networks.push(ctx.network.clone());
347                            }
348                        },
349                        NodeAction::NetworkUp => {
350                            if let Some(handle) = self.handles.write().await.get(idx) {
351                                tracing::error!("Node {} networks resuming", idx);
352                                handle.network.resume();
353                            }
354                        },
355                        NodeAction::NetworkDown => {
356                            if let Some(handle) = self.handles.write().await.get(idx) {
357                                tracing::error!("Node {} networks pausing", idx);
358                                handle.network.pause();
359                            }
360                        },
361                    }
362                }
363            }
364            let mut ready_futs = vec![];
365            while let Some(net) = new_networks.pop() {
366                ready_futs.push(async move {
367                    net.wait_for_ready().await;
368                });
369            }
370            join_all(ready_futs).await;
371
372            let mut start_futs = vec![];
373
374            while let Some((node, id)) = new_nodes.pop() {
375                let handles = self.handles.clone();
376                let fut = async move {
377                    tracing::info!("Starting node {} back up", id);
378                    let handle = node.run_tasks().await;
379
380                    // Create the node and add it to the state, so we can shut them
381                    // down properly later to avoid the overflow error in the overall
382                    // safety task.
383                    let node = Node {
384                        node_id: id.try_into().unwrap(),
385                        network: node.network.clone(),
386                        handle,
387                    };
388                    node.handle.hotshot.start_consensus().await;
389
390                    handles.write().await[id] = node;
391                };
392                start_futs.push(fut);
393            }
394            if !start_futs.is_empty() {
395                join_all(start_futs).await;
396                tracing::info!("Nodes all started");
397            }
398
399            // update our latest view
400            self.latest_view = Some(view_number);
401        }
402
403        Ok(())
404    }
405
406    async fn check(&self) -> TestResult {
407        TestResult::Pass
408    }
409}

422/// Spin the node up or down
423#[derive(Clone, Debug)]
424pub enum NodeAction {
425    /// spin the node up
426    Up,
427    /// spin the node down
428    Down,
429    /// spin the node's network up
430    NetworkUp,
431    /// spin the node's network down
432    NetworkDown,
433    /// Take a node down to be restarted after a number of views
434    RestartDown(u64),
435    /// Start a node up again after it's been shutdown for restart.  This
436    /// should only be created following a `RestartDown`
437    RestartUp,
438}

440/// denotes a change in node state
441#[derive(Clone, Debug)]
442pub struct ChangeNode {
443    /// the index of the node
444    pub idx: usize,
445    /// spin the node or node's network up or down
446    pub updown: NodeAction,
447}
