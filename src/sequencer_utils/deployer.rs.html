113        if let Some(addr) = deployed.light_client {
114            m.insert(Contract::LightClient, addr);
115        }
116        if let Some(addr) = deployed.light_client_v2 {
117            m.insert(Contract::LightClientV2, addr);
118        }
119        if let Some(addr) = deployed.light_client_proxy {
120            m.insert(Contract::LightClientProxy, addr);
121        }
122        if let Some(addr) = deployed.fee_contract {
123            m.insert(Contract::FeeContract, addr);
124        }
125        if let Some(addr) = deployed.fee_contract_proxy {
126            m.insert(Contract::FeeContractProxy, addr);
127        }
128        if let Some(addr) = deployed.esp_token {
129            m.insert(Contract::EspToken, addr);

131        if let Some(addr) = deployed.esp_token_proxy {

149    pub fn address(&self, contract: Contract) -> Option<Address> {
150        self.0.get(&contract).copied()
151    }

153    /// Deploy a contract (with logging and cached deployments)
154    ///
155    /// The deployment `tx` will be sent only if contract `name` is not already deployed;
156    /// otherwise this function will just return the predeployed address.
157    pub async fn deploy<T, P>(
158        &mut self,
159        name: Contract,
160        tx: RawCallBuilder<T, P>,
161    ) -> Result<Address>
162    where
163        P: Provider,
164    {
165        if let Some(addr) = self.0.get(&name) {
166            tracing::info!("skipping deployment of {name}, already deployed at {addr:#x}");
167            return Ok(*addr);
168        }
169        tracing::info!("deploying {name}");
170        let addr = tx.deploy().await?;
171        tracing::info!("deployed {name} at {addr:#x}");
172
173        self.0.insert(name, addr);
174        Ok(addr)
175    }
176
177    /// Write a .env file.
178    pub fn write(&self, mut w: impl Write) -> Result<()> {
179        for (contract, address) in &self.0 {
180            writeln!(w, "{contract}={address:#x}")?;
181        }
182        Ok(())
183    }
184}

186/// Default deployment function `LightClient.sol` or `LightClientMock.sol` with `mock: true`.
187///
188/// # NOTE:
189/// In most cases, you only need to use [`deploy_light_client_proxy()`]
190///
191/// # NOTE:
192/// currently, `LightClient.sol` follows upgradable contract, thus a follow-up
193/// call to `.initialize()` with proper genesis block (and other constructor args)
194/// are expected to be *delegatecall-ed through the proxy contract*.
195pub(crate) async fn deploy_light_client_contract(
196    provider: impl Provider,
197    contracts: &mut Contracts,
198    mock: bool,
199) -> Result<Address> {
200    // Deploy library contracts.
201    let plonk_verifier_addr = contracts
202        .deploy(
203            Contract::PlonkVerifier,
204            PlonkVerifier::deploy_builder(&provider),
205        )
206        .await?;
207
208    assert!(is_contract(&provider, plonk_verifier_addr).await?);
209
210    // when generate alloy's bindings, we supply a placeholder address, now we modify the actual
211    // bytecode with deployed address of the library.
212    let target_lc_bytecode = if mock {
213        LightClientMock::BYTECODE.encode_hex()
214    } else {
215        LightClient::BYTECODE.encode_hex()
216    };
217    let lc_linked_bytecode = {
218        match target_lc_bytecode
219            .matches(LIBRARY_PLACEHOLDER_ADDRESS)
220            .count()
221        {
222            0 => return Err(anyhow!("lib placeholder not found")),
223            1 => Bytes::from_hex(target_lc_bytecode.replacen(
224                LIBRARY_PLACEHOLDER_ADDRESS,
225                &plonk_verifier_addr.encode_hex(),
226                1,
227            ))?,
228            _ => {
229                return Err(anyhow!(
230                    "more than one lib placeholder found, consider using a different value"
231                ))
232            },
233        }
234    };
235
236    // Deploy the light client
237    let light_client_addr = if mock {
238        // for mock, we don't populate the `contracts` since it only track production-ready deployments
239        let addr = LightClientMock::deploy_builder(&provider)
240            .map(|req| req.with_deploy_code(lc_linked_bytecode))
241            .deploy()
242            .await?;
243        tracing::info!("deployed LightClientMock at {addr:#x}");
244        addr
245    } else {
246        contracts
247            .deploy(
248                Contract::LightClient,
249                LightClient::deploy_builder(&provider)
250                    .map(|req| req.with_deploy_code(lc_linked_bytecode)),
251            )
252            .await?
253    };
254    Ok(light_client_addr)
255}

257/// The primary logic for deploying and initializing an upgradable light client contract.
258///
259/// Deploy the upgradable proxy contract, point to an already deployed light client contract as its implementation, and invoke `initialize()` on it.
260/// This is run after `deploy_light_client_contract()`, returns the proxy address.
261/// This works for both mock and production light client proxy.
262pub async fn deploy_light_client_proxy(
263    provider: impl Provider,
264    contracts: &mut Contracts,
265    mock: bool,
266    genesis_state: LightClientStateSol,
267    genesis_stake: StakeTableStateSol,
268    admin: Address,
269    prover: Option<Address>,
270) -> Result<Address> {
271    // deploy the light client implementation contract
272    let impl_addr = deploy_light_client_contract(&provider, contracts, mock).await?;
273    let lc = LightClient::new(impl_addr, &provider);
274
275    // prepare the input arg for `initialize()`
276    let init_data = lc
277        .initialize(
278            genesis_state,
279            genesis_stake,
280            MAX_HISTORY_RETENTION_SECONDS,
281            admin,
282        )
283        .calldata()
284        .to_owned();
285    // deploy proxy and initialize
286    let lc_proxy_addr = contracts
287        .deploy(
288            Contract::LightClientProxy,
289            ERC1967Proxy::deploy_builder(&provider, impl_addr, init_data),
290        )
291        .await?;
292
293    // sanity check
294    if !is_proxy_contract(&provider, lc_proxy_addr).await? {
295        panic!("LightClientProxy detected not as a proxy, report error!");
296    }
297
298    // instantiate a proxy instance, cast as LightClient's ABI interface
299    let lc_proxy = LightClient::new(lc_proxy_addr, &provider);
300
301    // set permissioned prover
302    if let Some(prover) = prover {
303        tracing::info!(%lc_proxy_addr, %prover, "Set permissioned prover ");
304        lc_proxy
305            .setPermissionedProver(prover)
306            .send()
307            .await?
308            .get_receipt()
309            .await?;
310    }
311
312    // post deploy verification checks
313    assert_eq!(lc_proxy.getVersion().call().await?.majorVersion, 1);
314    assert_eq!(lc_proxy.owner().call().await?._0, admin);
315    if let Some(prover) = prover {
316        assert_eq!(lc_proxy.permissionedProver().call().await?._0, prover);
317    }
318    assert_eq!(
319        lc_proxy.stateHistoryRetentionPeriod().call().await?._0,
320        864000
321    );
322    assert_eq!(
323        lc_proxy.currentBlockNumber().call().await?._0,
324        U256::from(provider.get_block_number().await?)
325    );
326
327    Ok(lc_proxy_addr)
328}

330/// Upgrade the light client proxy to use LightClientV2.
331/// Internally, first detect existence of proxy, then deploy LCV2, then upgrade and initializeV2.
332/// Internal to "deploy LCV2", we deploy PlonkVerifierV2 whose address will be used at LCV2 init time.
333pub async fn upgrade_light_client_v2(
334    provider: impl Provider,
335    contracts: &mut Contracts,
336    is_mock: bool,
337    blocks_per_epoch: u64,
338    epoch_start_block: u64,
339) -> Result<TransactionReceipt> {
340    match contracts.address(Contract::LightClientProxy) {
341        // check if proxy already exists
342        None => Err(anyhow!("LightClientProxy not found, can't upgrade")),
343        Some(proxy_addr) => {
344            let proxy = LightClient::new(proxy_addr, &provider);
345            let state_history_retention_period =
346                proxy.stateHistoryRetentionPeriod().call().await?._0;
347            // first deploy PlonkVerifierV2.sol
348            let pv2_addr = contracts
349                .deploy(
350                    Contract::PlonkVerifierV2,
351                    PlonkVerifierV2::deploy_builder(&provider),
352                )
353                .await?;
354
355            assert!(is_contract(&provider, pv2_addr).await?);
356
357            // then deploy LightClientV2.sol
358            let target_lcv2_bytecode = if is_mock {
359                LightClientV2Mock::BYTECODE.encode_hex()
360            } else {
361                LightClientV2::BYTECODE.encode_hex()
362            };
363            let lcv2_linked_bytecode = {
364                match target_lcv2_bytecode
365                    .matches(LIBRARY_PLACEHOLDER_ADDRESS)
366                    .count()
367                {
368                    0 => return Err(anyhow!("lib placeholder not found")),
369                    1 => Bytes::from_hex(target_lcv2_bytecode.replacen(
370                        LIBRARY_PLACEHOLDER_ADDRESS,
371                        &pv2_addr.encode_hex(),
372                        1,
373                    ))?,
374                    _ => {
375                        return Err(anyhow!(
376                            "more than one lib placeholder found, consider using a different value"
377                        ))
378                    },
379                }
380            };
381            let lcv2_addr = if is_mock {
382                let addr = LightClientV2Mock::deploy_builder(&provider)
383                    .map(|req| req.with_deploy_code(lcv2_linked_bytecode))
384                    .deploy()
385                    .await?;
386                tracing::info!("deployed LightClientV2Mock at {addr:#x}");
387                addr
388            } else {
389                contracts
390                    .deploy(
391                        Contract::LightClientV2,
392                        LightClientV2::deploy_builder(&provider)
393                            .map(|req| req.with_deploy_code(lcv2_linked_bytecode)),
394                    )
395                    .await?
396            };
397
398            // prepare init calldata
399            let lcv2 = LightClientV2::new(lcv2_addr, &provider);
400            let init_data = lcv2
401                .initializeV2(blocks_per_epoch, epoch_start_block)
402                .calldata()
403                .to_owned();
404            // invoke upgrade on proxy
405            let receipt = proxy
406                .upgradeToAndCall(lcv2_addr, init_data)
407                .send()
408                .await?
409                .get_receipt()
410                .await?;
411            if receipt.inner.is_success() {
412                // post deploy verification checks
413                let proxy_as_v2 = LightClientV2::new(proxy_addr, &provider);
414                assert_eq!(proxy_as_v2.getVersion().call().await?.majorVersion, 2);
415                assert_eq!(
416                    proxy_as_v2.blocksPerEpoch().call().await?._0,
417                    blocks_per_epoch
418                );
419                assert_eq!(
420                    proxy_as_v2.epochStartBlock().call().await?._0,
421                    epoch_start_block
422                );
423                assert_eq!(
424                    proxy_as_v2.stateHistoryRetentionPeriod().call().await?._0,
425                    state_history_retention_period
426                );
427                assert_eq!(
428                    proxy_as_v2.currentBlockNumber().call().await?._0,
429                    U256::from(provider.get_block_number().await?)
430                );
431
432                tracing::info!(%lcv2_addr, "LightClientProxy successfully upgrade to: ")
433            } else {
434                tracing::error!("LightClientProxy upgrade failed: {:?}", receipt);
435            }
436
437            Ok(receipt)
438        },
439    }
440}

442/// The primary logic for deploying and initializing an upgradable fee contract.
443///
444/// Deploy the upgradable proxy contract, point to a deployed fee contract as its implementation, and invoke `initialize()` on it.
445/// - `admin`: is the new owner (e.g. a multisig address) of the proxy contract
446///
447/// Return the proxy address.
448pub async fn deploy_fee_contract_proxy(
449    provider: impl Provider,
450    contracts: &mut Contracts,
451    admin: Address,
452) -> Result<Address> {
453    // deploy the fee implementation contract
454    let fee_addr = contracts
455        .deploy(
456            Contract::FeeContract,
457            FeeContract::deploy_builder(&provider),
458        )
459        .await?;
460    let fee = FeeContract::new(fee_addr, &provider);
461
462    // prepare the input arg for `initialize()`
463    let init_data = fee.initialize(admin).calldata().to_owned();
464    // deploy proxy and initialize
465    let fee_proxy_addr = contracts
466        .deploy(
467            Contract::FeeContractProxy,
468            ERC1967Proxy::deploy_builder(&provider, fee_addr, init_data),
469        )
470        .await?;
471    // sanity check
472    if !is_proxy_contract(&provider, fee_proxy_addr).await? {
473        panic!("FeeContractProxy detected not as a proxy, report error!");
474    }
475
476    // post deploy verification checks
477    let fee_proxy = FeeContract::new(fee_proxy_addr, &provider);
478    assert_eq!(fee_proxy.getVersion().call().await?.majorVersion, 1);
479    assert_eq!(fee_proxy.owner().call().await?._0, admin);
480
481    Ok(fee_proxy_addr)
482}

506

509    }
510
511    // post deploy verification checks
512    let token_proxy = EspToken::new(token_proxy_addr, &provider);
513    assert_eq!(token_proxy.getVersion().call().await?.majorVersion, 1);
514    assert_eq!(token_proxy.owner().call().await?._0, owner);
515    assert_eq!(token_proxy.symbol().call().await?._0, "ESP");
516    assert_eq!(token_proxy.decimals().call().await?._0, 18);
517    assert_eq!(token_proxy.name().call().await?._0, "Espresso Token");
518    let total_supply = token_proxy.totalSupply().call().await?._0;
519    assert_eq!(
520        token_proxy.balanceOf(init_grant_recipient).call().await?._0,
521        total_supply
522    );
523
524    Ok(token_proxy_addr)

576/// Common logic for any Ownable contract to transfer ownership
577pub async fn transfer_ownership(
578    provider: impl Provider,
579    target: Contract,
580    addr: Address,
581    new_owner: Address,
582) -> Result<TransactionReceipt> {
583    let receipt = match target {
584        Contract::LightClient | Contract::LightClientProxy => {
585            tracing::info!(%addr, %new_owner, "Transfer LightClient ownership");
586            let lc = LightClient::new(addr, &provider);
587            lc.transferOwnership(new_owner)
588                .send()
589                .await?
590                .get_receipt()
591                .await?
592        },
593        Contract::FeeContract | Contract::FeeContractProxy => {
594            tracing::info!(%addr, %new_owner, "Transfer FeeContract ownership");
595            let fee = FeeContract::new(addr, &provider);
596            fee.transferOwnership(new_owner)
597                .send()
598                .await?
599                .get_receipt()
600                .await?
601        },
602        Contract::EspToken | Contract::EspTokenProxy => {
603            tracing::info!(%addr, %new_owner, "Transfer EspToken ownership");
604            let token = EspToken::new(addr, &provider);
605            token
606                .transferOwnership(new_owner)
607                .send()
608                .await?
609                .get_receipt()
610                .await?
611        },
612        Contract::StakeTable | Contract::StakeTableProxy => {
613            tracing::info!(%addr, %new_owner, "Transfer StakeTable ownership");
614            let stake_table = StakeTable::new(addr, &provider);
615            stake_table
616                .transferOwnership(new_owner)
617                .send()
618                .await?
619                .get_receipt()
620                .await?
621        },
622        _ => return Err(anyhow!("Not Ownable, can't transfer ownership!")),
623    };
624    Ok(receipt)
625}

627/// helper function to decide if the contract at given address `addr` is a proxy contract
628pub async fn is_proxy_contract(provider: impl Provider, addr: Address) -> Result<bool> {
629    // confirm that the proxy_address is a proxy
630    // using the implementation slot, 0x360894a13ba1a3210667c828492db98dca3e2076cc3735a920a3ca505d382bbc, which is the keccak-256 hash of "eip1967.proxy.implementation" subtracted by 1
631    let impl_slot = U256::from_str_radix(
632        "360894a13ba1a3210667c828492db98dca3e2076cc3735a920a3ca505d382bbc",
633        16,
634    )?;
635    let storage = provider.get_storage_at(addr, impl_slot).await?;
636    let impl_address = Address::from_slice(&storage.to_be_bytes_vec()[12..]);
637
638    // when the implementation address is not equal to zero, it's a proxy
639    Ok(impl_address != Address::default())
640}

642pub async fn is_contract(provider: impl Provider, address: Address) -> Result<bool> {
643    if address == Address::ZERO {
644        return Ok(false);
645    }
646
647    let code = provider.get_code_at(address).await?;
648    if code.is_empty() {
649        return Ok(false);
650    }
651
652    Ok(true)
653}
