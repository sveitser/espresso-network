91/// The underlying configuration for the request-response protocol
92#[derive(Clone)]
93pub struct RequestResponseConfig {
94    /// The timeout for incoming requests. Do not respond to a request after this threshold
95    /// has passed.
96    pub incoming_request_ttl: Duration,
97    /// The maximum amount of time we will spend trying to both derive a response for a request and
98    /// send the response over the wire.
99    pub response_send_timeout: Duration,
100    /// The maximum amount of time we will spend trying to validate a response. This is used to prevent
101    /// an attack where a malicious participant sends us a bunch of requests that take a long time to
102    /// validate.
103    pub response_validate_timeout: Duration,
104    /// The batch size for outgoing requests. This is the number of request messages that we will
105    /// send out at a time for a single request before waiting for the [`request_batch_interval`].
106    pub request_batch_size: usize,
107    /// The time to wait (per request) between sending out batches of request messages
108    pub request_batch_interval: Duration,
109    /// The maximum (global) number of outgoing responses that can be in flight at any given time
110    pub max_outgoing_responses: usize,
111    /// The maximum (global) number of incoming responses that can be processed at any given time.
112    /// We need this because responses coming in need to be validated [asynchronously] that they
113    /// satisfy the request they are responding to
114    pub max_incoming_responses: usize,
115}

117/// A protocol that allows for request-response communication. Is cheaply cloneable, so there is no
118/// need to wrap it in an `Arc`
119#[derive(Deref)]
120pub struct RequestResponse<
121    S: Sender<K>,
122    R: Receiver,
123    Req: Request,
124    RS: RecipientSource<Req, K>,
125    DS: DataSource<Req>,
126    K: SignatureKey + 'static,
127> {
128    #[deref]
129    /// The inner implementation of the request-response protocol
130    inner: Arc<RequestResponseInner<S, R, Req, RS, DS, K>>,
131    /// A handle to the receiving task. This will automatically get cancelled when the protocol is dropped
132    _receiving_task_handle: Arc<AbortOnDropHandle<()>>,
133}
134
135/// We need to manually implement the `Clone` trait for this type because deriving
136/// `Deref` will cause an issue where it tries to clone the inner field instead
137impl<
138        S: Sender<K>,
139        R: Receiver,
140        Req: Request,
141        RS: RecipientSource<Req, K>,
142        DS: DataSource<Req>,
143        K: SignatureKey + 'static,
144    > Clone for RequestResponse<S, R, Req, RS, DS, K>
145{
146    fn clone(&self) -> Self {
147        Self {
148            inner: Arc::clone(&self.inner),
149            _receiving_task_handle: Arc::clone(&self._receiving_task_handle),
150        }
151    }
152}

154impl<
155        S: Sender<K>,
156        R: Receiver,
157        Req: Request,
158        RS: RecipientSource<Req, K>,
159        DS: DataSource<Req>,
160        K: SignatureKey + 'static,
161    > RequestResponse<S, R, Req, RS, DS, K>
162{
163    /// Create a new [`RequestResponseProtocol`]
164    pub fn new(
165        // The configuration for the protocol
166        config: RequestResponseConfig,
167        // The network sender that [`RequestResponseProtocol`] will use to send messages
168        sender: S,
169        // The network receiver that [`RequestResponseProtocol`] will use to receive messages
170        receiver: R,
171        // The recipient source that [`RequestResponseProtocol`] will use to get the recipients
172        // that a specific message should expect responses from
173        recipient_source: RS,
174        // The [response] data source that [`RequestResponseProtocol`] will use to derive the
175        // response data for a specific request
176        data_source: DS,
177    ) -> Self {
178        // Create the active requests map
179        let active_requests = ActiveRequestsMap::default();
180
181        // Create the inner implementation
182        let inner = Arc::new(RequestResponseInner {
183            config,
184            sender,
185            recipient_source,
186            data_source,
187            active_requests,
188            phantom_data: PhantomData,
189        });
190
191        // Start the task that receives messages and handles them. This will automatically get cancelled
192        // when the protocol is dropped
193        let inner_clone = Arc::clone(&inner);
194        let receive_task_handle =
195            AbortOnDropHandle::new(tokio::spawn(inner_clone.receiving_task(receiver)));
196
197        // Return the protocol
198        Self {
199            inner,
200            _receiving_task_handle: Arc::new(receive_task_handle),
201        }
202    }
203}

216/// The inner implementation for the request-response protocol
217pub struct RequestResponseInner<
218    S: Sender<K>,
219    R: Receiver,
220    Req: Request,
221    RS: RecipientSource<Req, K>,
222    DS: DataSource<Req>,
223    K: SignatureKey + 'static,
224> {
225    /// The configuration of the protocol
226    config: RequestResponseConfig,
227    /// The sender to use for the protocol
228    pub sender: S,
229    /// The recipient source to use for the protocol
230    recipient_source: RS,
231    /// The data source to use for the protocol
232    data_source: DS,
233    /// The map of currently active requests
234    active_requests: ActiveRequestsMap<Req>,
235    /// Phantom data to help with type inference
236    phantom_data: PhantomData<(K, R, Req, DS)>,
237}
238impl<
239        S: Sender<K>,
240        R: Receiver,
241        Req: Request,
242        RS: RecipientSource<Req, K>,
243        DS: DataSource<Req>,
244        K: SignatureKey + 'static,
245    > RequestResponseInner<S, R, Req, RS, DS, K>
246{
247    /// Request something from the protocol indefinitely until we get a response
248    /// or there was a critical error (e.g. the request could not be signed)
249    ///
250    /// # Errors
251    /// - If the request was invalid
252    /// - If there was a critical error (e.g. the channel was closed)
253    pub async fn request_indefinitely<F, Fut, O>(
254        self: &Arc<Self>,
255        public_key: &K,
256        private_key: &K::PrivateKey,
257        // The estimated TTL of other participants. This is used to decide when to
258        // stop making requests and sign a new one
259        estimated_request_ttl: Duration,
260        // The request to make
261        request: Req,
262        // The response validation function
263        response_validation_fn: F,
264    ) -> std::result::Result<O, RequestError>
265    where
266        F: Fn(&Req, Req::Response) -> Fut + Send + Sync + 'static + Clone,
267        Fut: Future<Output = anyhow::Result<O>> + Send + Sync + 'static,
268        O: Send + Sync + 'static + Clone,
269    {
270        loop {
271            // Sign a request message
272            let request_message = RequestMessage::new_signed(public_key, private_key, &request)
273                .map_err(|e| {
274                    RequestError::InvalidRequest(anyhow::anyhow!(
275                        "failed to sign request message: {e}"
276                    ))
277                })?;
278
279            // Request the data, handling the errors appropriately
280            match self
281                .request(
282                    request_message,
283                    estimated_request_ttl,
284                    response_validation_fn.clone(),
285                )
286                .await
287            {
288                Ok(response) => return Ok(response),
289                Err(RequestError::Timeout) => continue,
290                Err(e) => return Err(e),
291            }
292        }
293    }
294
295    /// Request something from the protocol and wait for the response. This function
296    /// will join with an existing request for the same data (determined by `Blake3` hash),
297    /// however both will make requests until the timeout is reached
298    ///
299    /// # Errors
300    /// - If the request times out
301    /// - If the channel is closed (this is an internal error)
302    /// - If the request we sign is invalid
303    pub async fn request<F, Fut, O>(
304        self: &Arc<Self>,
305        request_message: RequestMessage<Req, K>,
306        timeout_duration: Duration,
307        response_validation_fn: F,
308    ) -> std::result::Result<O, RequestError>
309    where
310        F: Fn(&Req, Req::Response) -> Fut + Send + Sync + 'static + Clone,
311        Fut: Future<Output = anyhow::Result<O>> + Send + Sync + 'static,
312        O: Send + Sync + 'static + Clone,
313    {
314        timeout(timeout_duration, async move {
315            // Calculate the hash of the request
316            let request_hash = blake3::hash(&request_message.request.to_bytes().map_err(|e| {
317                RequestError::InvalidRequest(anyhow::anyhow!(
318                    "failed to serialize request message: {e}"
319                ))
320            })?);
321
322            let request = {
323                // Get a write lock on the active requests map
324                let mut active_requests_write = self.active_requests.write();
325
326                // Conditionally get the active request, creating a new one if it doesn't exist or if
327                // the existing one has been dropped and not yet removed
328                if let Some(active_request) = active_requests_write
329                    .get(&request_hash)
330                    .and_then(Weak::upgrade)
331                {
332                    ActiveRequest(active_request)
333                } else {
334                    // Create a new broadcast channel for the response
335                    let (sender, receiver) = async_broadcast::broadcast(1);
336
337                    // Modify the response validation function to return an `Arc<dyn Any>`
338                    let response_validation_fn =
339                        Box::new(move |request: &Req, response: Req::Response| {
340                            let fut = response_validation_fn(request, response);
341                            Box::pin(
342                                async move { fut.await.map(|ok| Arc::new(ok) as ThreadSafeAny) },
343                            ) as ResponseValidationFuture
344                        });
345
346                    // Create a new active request
347                    let active_request = ActiveRequest(Arc::new(ActiveRequestInner {
348                        sender,
349                        receiver,
350                        response_validation_fn,
351                        request: request_message.request.clone(),
352                        active_requests: Arc::clone(&self.active_requests),
353                        request_hash,
354                    }));
355
356                    // Write the new active request to the map
357                    active_requests_write.insert(request_hash, Arc::downgrade(&active_request.0));
358
359                    // Return the new active request
360                    active_request
361                }
362            };
363
364            // Get the recipients that the request should expect responses from. Shuffle them so
365            // that we don't always send to the same recipients in the same order
366            let mut recipients = self
367                .recipient_source
368                .get_expected_responders(&request_message.request)
369                .await
370                .map_err(|e| {
371                    RequestError::InvalidRequest(anyhow::anyhow!(
372                        "failed to get expected responders for request: {e}"
373                    ))
374                })?;
375            recipients.shuffle(&mut rand::thread_rng());
376
377            // Create a request message and serialize it
378            let message =
379                Bytes::from(Message::Request(request_message).to_bytes().map_err(|e| {
380                    RequestError::InvalidRequest(anyhow::anyhow!(
381                        "failed to serialize request message: {e}"
382                    ))
383                })?);
384
385            // Get the current time so we can check when the timeout has elapsed
386            let start_time = Instant::now();
387
388            // Spawn a task that sends out requests to the network
389            let self_clone = Arc::clone(self);
390            let _handle = AbortOnDropHandle::new(spawn(async move {
391                // Create a bounded queue for the outgoing requests. We use this to make sure
392                // we have less than [`config.request_batch_size`] requests in flight at any time.
393                //
394                // When newer requests are added, older ones are removed from the queue. Because we use
395                // `AbortOnDropHandle`, the older ones will automatically get cancelled
396                let mut outgoing_requests =
397                    BoundedVecDeque::new(self_clone.config.request_batch_size);
398
399                // While the timeout hasn't elapsed, send out requests to the network
400                while start_time.elapsed() < timeout_duration {
401                    // Send out requests to the network in their own separate tasks
402                    for recipient_batch in recipients.chunks(self_clone.config.request_batch_size) {
403                        for recipient in recipient_batch {
404                            // Clone ourselves, the message, and the recipient so they can be moved
405                            let self_clone = Arc::clone(&self_clone);
406                            let recipient_clone = recipient.clone();
407                            let message_clone = Arc::clone(&message);
408
409                            // Spawn the task that sends the request to the participant
410                            let individual_sending_task = spawn(async move {
411                                let _ = self_clone
412                                    .sender
413                                    .send_message(&message_clone, recipient_clone)
414                                    .await;
415                            });
416
417                            // Add the sending task to the queue
418                            outgoing_requests.push(AbortOnDropHandle::new(individual_sending_task));
419                        }
420
421                        // After we send the batch out, wait the [`config.request_batch_interval`]
422                        // before sending the next one
423                        sleep(self_clone.config.request_batch_interval).await;
424                    }
425                }
426            }));
427
428            // Wait for a response on the channel
429            request
430                .receiver
431                .clone()
432                .recv()
433                .await
434                .map_err(|_| RequestError::Other(anyhow!("channel was closed")))
435        })
436        .await
437        .map_err(|_| RequestError::Timeout)
438        .and_then(|result| result)
439        .and_then(|result| {
440            result.downcast::<O>().map_err(|e| {
441                RequestError::Other(anyhow::anyhow!(
442                    "failed to downcast response to expected type: {:?}",
443                    e
444                ))
445            })
446        })
447        .map(|result| Arc::unwrap_or_clone(result))
448    }

450    /// The task responsible for receiving messages from the receiver and handling them
451    async fn receiving_task(self: Arc<Self>, mut receiver: R) {
452        // Upper bound the number of outgoing and incoming responses
453        let mut outgoing_responses = BoundedVecDeque::new(self.config.max_outgoing_responses);
454        let mut incoming_responses = BoundedVecDeque::new(self.config.max_incoming_responses);
455
456        // While the receiver is open, we receive messages and handle them
457        loop {
458            // Try to receive a message
459            match receiver.receive_message().await {
460                Ok(message) => {
461                    // Deserialize the message, warning if it fails
462                    let message = match Message::from_bytes(&message) {
463                        Ok(message) => message,
464                        Err(e) => {
465                            warn!("Received invalid message: {e:#}");
466                            continue;
467                        },
468                    };
469
470                    // Handle the message based on its type
471                    match message {
472                        Message::Request(request_message) => {
473                            self.handle_request(request_message, &mut outgoing_responses);
474                        },
475                        Message::Response(response_message) => {
476                            self.handle_response(response_message, &mut incoming_responses);
477                        },
478                    }
479                },
480                // An error here means the receiver will _NEVER_ receive any more messages
481                Err(e) => {
482                    error!("Request/response receive task exited: {e:#}");
483                    return;
484                },
485            }
486        }
487    }

489    /// Handle a request sent to us
490    fn handle_request(
491        self: &Arc<Self>,
492        request_message: RequestMessage<Req, K>,
493        outgoing_responses: &mut OutgoingResponses,
494    ) {
495        // Spawn a task to:
496        // - Validate the request
497        // - Derive the response data (check if we have it)
498        // - Send the response to the requester
499        let self_clone = Arc::clone(self);
500        let response_task = AbortOnDropHandle::new(tokio::spawn(async move {
501            let result = timeout(self_clone.config.response_send_timeout, async move {
502                // Validate the request message. This includes:
503                // - Checking the signature and making sure it's valid
504                // - Checking the timestamp and making sure it's not too old
505                // - Calling the request's application-specific validation function
506                request_message
507                    .validate(self_clone.config.incoming_request_ttl)
508                    .await
509                    .with_context(|| "failed to validate request")?;
510
511                // Try to fetch the response data from the data source
512                let response = self_clone
513                    .data_source
514                    .derive_response_for(&request_message.request)
515                    .await
516                    .with_context(|| "failed to derive response for request")?;
517
518                // Create the response message and serialize it
519                let response = Bytes::from(
520                    Message::Response::<Req, K>(ResponseMessage {
521                        request_hash: blake3::hash(&request_message.request.to_bytes()?),
522                        response,
523                    })
524                    .to_bytes()
525                    .with_context(|| "failed to serialize response message")?,
526                );
527
528                // Send the response to the requester
529                self_clone
530                    .sender
531                    .send_message(&response, request_message.public_key)
532                    .await
533                    .with_context(|| "failed to send response to requester")?;
534
535                Ok::<(), anyhow::Error>(())
536            })
537            .await
538            .map_err(|_| anyhow::anyhow!("timed out while sending response"))
539            .and_then(|result| result);
540
541            if let Err(e) = result {
542                debug!("Failed to send response to requester: {e:#}");
543            }
544        }));
545
546        // Add the response task to the outgoing responses queue. This will automatically cancel an older task
547        // if there are more than [`config.max_outgoing_responses`] responses in flight.
548        outgoing_responses.push(response_task);
549    }
550
551    /// Handle a response sent to us
552    fn handle_response(
553        self: &Arc<Self>,
554        response: ResponseMessage<Req>,
555        incoming_responses: &mut IncomingResponses,
556    ) {
557        // Get the entry in the map, ignoring it if it doesn't exist
558        let Some(active_request) = self
559            .active_requests
560            .read()
561            .get(&response.request_hash)
562            .cloned()
563            .and_then(|r| r.upgrade())
564        else {
565            return;
566        };
567
568        // Spawn a task to validate the response and send it to the requester (us)
569        let response_validate_timeout = self.config.response_validate_timeout;
570        let response_task = AbortOnDropHandle::new(tokio::spawn(async move {
571            if timeout(response_validate_timeout, async move {
572                // Make sure the response is valid for the given request
573                let validation_result = match (active_request.response_validation_fn)(
574                    &active_request.request,
575                    response.response,
576                )
577                .await
578                {
579                    Ok(validation_result) => validation_result,
580                    Err(e) => {
581                        warn!("Received invalid response: {e:#}");
582                        return;
583                    },
584                };
585
586                // Send the response to the requester (the user of [`RequestResponse::request`])
587                let _ = active_request.sender.try_broadcast(validation_result);
588            })
589            .await
590            .is_err()
591            {
592                warn!("Timed out while validating response");
593            }
594        }));
595
596        // Add the response task to the incoming responses queue. This will automatically cancel an older task
597        // if there are more than [`config.max_incoming_responses`] responses being processed
598        incoming_responses.push(response_task);
599    }
600}

602/// An active request. This is what we use to track a request and its corresponding response
603/// in the protocol
604#[derive(Clone, Deref)]
605pub struct ActiveRequest<R: Request>(Arc<ActiveRequestInner<R>>);
606
607/// The inner implementation of an active request
608pub struct ActiveRequestInner<R: Request> {
609    /// The sender to use for the protocol
610    sender: async_broadcast::Sender<ThreadSafeAny>,
611    /// The receiver to use for the protocol
612    receiver: async_broadcast::Receiver<ThreadSafeAny>,
613
614    /// The request that we are waiting for a response to
615    request: R,
616
617    /// The function used to validate the response
618    response_validation_fn: ResponseValidationFn<R>,
619
620    /// A copy of the map of currently active requests
621    active_requests: ActiveRequestsMap<R>,
622    /// The hash of the request. We need this so we can remove ourselves from the map
623    request_hash: RequestHash,
624}
625
626impl<R: Request> Drop for ActiveRequestInner<R> {
627    fn drop(&mut self) {
628        self.active_requests.write().remove(&self.request_hash);
629    }
630}

743    // Create a test data source that pretends to have the data or not
744    #[derive(Clone)]
745    struct TestDataSource {
746        /// Whether we have the data or not
747        has_data: bool,
748        /// The time at which the data will be available if we have it
749        data_available_time: Instant,
750
751        /// Whether or not the data will be taken once served
752        take_data: bool,
753        /// Whether or not the data has been taken
754        taken: Arc<AtomicBool>,
755    }
756
757    #[async_trait]
758    impl DataSource<TestRequest> for TestDataSource {
759        async fn derive_response_for(&self, request: &TestRequest) -> Result<Vec<u8>> {
760            // Return a response if we hit the hit rate
761            if self.has_data && Instant::now() >= self.data_available_time {
762                if self.take_data && !self.taken.swap(true, std::sync::atomic::Ordering::Relaxed) {
763                    return Err(anyhow::anyhow!("data already taken"));
764                }
765                Ok(blake3::hash(&request.0).as_bytes().to_vec())
766            } else {
767                Err(anyhow::anyhow!("did not have the data"))
768            }
769        }
770    }

785    /// Create fully connected test networks with `num_participants` participants
786    fn create_participants(
787        num: usize,
788    ) -> Vec<(TestSender, mpsc::Receiver<Bytes>, (BLSPubKey, BLSPrivKey))> {
789        // The entire network
790        let mut network = HashMap::new();
791
792        // All receivers in the network
793        let mut receivers = Vec::new();
794
795        // All keypairs in the network
796        let mut keypairs = Vec::new();
797
798        // For each participant,
799        for i in 0..num {
800            // Create a unique `BLSPubKey`
801            let (public_key, private_key) =
802                BLSPubKey::generated_from_seed_indexed([2; 32], i.try_into().unwrap());
803
804            // Add the keypair to the list
805            keypairs.push((public_key, private_key));
806
807            // Create a channel for sending and receiving messages
808            let (sender, receiver) = mpsc::channel::<Bytes>(100);
809
810            // Add the participant to the network
811            network.insert(public_key, sender);
812
813            // Add the receiver to the list of receivers
814            receivers.push(receiver);
815        }
816
817        // Create a test sender from the network
818        let sender = TestSender {
819            network: Arc::new(network),
820        };
821
822        // Return all senders and receivers
823        receivers
824            .into_iter()
825            .zip(keypairs)
826            .map(|(r, k)| (sender.clone(), r, k))
827            .collect()
828    }

830    /// The configuration for an integration test
831    #[derive(Clone)]
832    struct IntegrationTestConfig {
833        /// The request response protocol configuration
834        request_response_config: RequestResponseConfig,
835        /// The number of participants in the network
836        num_participants: usize,
837        /// The number of participants that have the data
838        num_participants_with_data: usize,
839        /// The timeout for the requests
840        request_timeout: Duration,
841        /// The delay before the nodes have the data available
842        data_available_delay: Duration,
843    }
844
845    /// The result of an integration test
846    struct IntegrationTestResult {
847        /// The number of nodes that received a response
848        num_succeeded: usize,
849    }
850
851    /// Run an integration test with the given parameters
852    async fn run_integration_test(config: IntegrationTestConfig) -> IntegrationTestResult {
853        // Create a fully connected network with `num_participants` participants
854        let participants = create_participants(config.num_participants);
855
856        // Create a join set to wait for all the tasks to finish
857        let mut join_set = JoinSet::new();
858
859        // We need to keep these here so they don't get dropped
860        let handles = Arc::new(Mutex::new(Vec::new()));
861
862        // For each one, create a new [`RequestResponse`] protocol
863        for (i, (sender, receiver, (public_key, private_key))) in
864            participants.into_iter().enumerate()
865        {
866            let config_clone = config.request_response_config.clone();
867            let handles_clone = Arc::clone(&handles);
868            join_set.spawn(async move {
869                let protocol = RequestResponse::new(
870                    config_clone,
871                    sender.clone(),
872                    receiver,
873                    sender,
874                    TestDataSource {
875                        has_data: i < config.num_participants_with_data,
876                        data_available_time: Instant::now() + config.data_available_delay,
877                        take_data: false,
878                        taken: Arc::new(AtomicBool::new(false)),
879                    },
880                );
881
882                // Add the handle to the handles list so it doesn't get dropped and
883                // cancelled
884                #[allow(clippy::used_underscore_binding)]
885                handles_clone
886                    .lock()
887                    .unwrap()
888                    .push(Arc::clone(&protocol._receiving_task_handle));
889
890                // Create a random request
891                let request = TestRequest(vec![rand::thread_rng().gen(); 100]);
892
893                // Get the hash of the request
894                let request_hash = blake3::hash(&request.0).as_bytes().to_vec();
895
896                // Create a new request message
897                let request = RequestMessage::new_signed(&public_key, &private_key, &request)
898                    .expect("failed to create request message");
899
900                // Request the data from the protocol
901                let response = protocol
902                    .request(
903                        request,
904                        config.request_timeout,
905                        |_request, response| async move { Ok(response) },
906                    )
907                    .await?;
908
909                // Make sure the response is the hash of the request
910                assert_eq!(response, request_hash);
911
912                Ok::<(), anyhow::Error>(())
913            });
914        }
915
916        // Wait for all the tasks to finish
917        let mut num_succeeded = config.num_participants;
918        while let Some(result) = join_set.join_next().await {
919            if result.is_err() || result.unwrap().is_err() {
920                num_succeeded -= 1;
921            }
922        }
923
924        IntegrationTestResult { num_succeeded }
925    }

987        // Build a config
988        let config = default_protocol_config();
989
990        // Create two participants
991        let mut participants = Vec::new();
992
993        for (sender, receiver, (public_key, private_key)) in create_participants(2) {
994            // For each, create a new [`RequestResponse`] protocol
995            let protocol = RequestResponse::new(
996                config.clone(),
997                sender.clone(),
998                receiver,
999                sender,
1000                TestDataSource {
1001                    take_data: true,
1002                    has_data: true,
1003                    data_available_time: Instant::now() + Duration::from_secs(2),
1004                    taken: Arc::new(AtomicBool::new(false)),
1005                },
1006            );
1007
1008            // Add the participants to the list
1009            participants.push((protocol, public_key, private_key));
1010        }
1011
1012        // Take the first participant
1013        let one = Arc::new(participants.remove(0));
1014
1015        // Create the request that they should all be able to join on
1016        let request = TestRequest(vec![rand::thread_rng().gen(); 100]);
1017
1018        // Create a join set to wait for all the tasks to finish
1019        let mut join_set = JoinSet::new();
1020
1021        // Make 10 requests with the same hash
1022        for _ in 0..10 {
1023            // Clone the first participant
1024            let one_clone = Arc::clone(&one);
1025
1026            // Clone the request
1027            let request_clone = request.clone();
1028
1029            // Spawn a task to request the data
1030            join_set.spawn(async move {
1031                // Create a new, signed request message
1032                let request_message =
1033                    RequestMessage::new_signed(&one_clone.1, &one_clone.2, &request_clone)?;
1034
1035                // Start requesting it
1036                one_clone
1037                    .0

1043                    .await?;

1045                Ok::<(), anyhow::Error>(())
1046            });
1047        }
1048
1049        // Wait for all the tasks to finish, making sure they all succeed
1050        while let Some(result) = join_set.join_next().await {

1052                .expect("failed to join task")
1053                .expect("failed to request data");
