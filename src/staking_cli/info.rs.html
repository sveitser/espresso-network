12pub async fn stake_table_info(
13    l1_url: Url,
14    stake_table_address: Address,
15    l1_block_number: u64,
16) -> Result<Vec<Validator<BLSPubKey>>> {
17    let l1 = L1Client::new(vec![l1_url])?;
18    let validators =
19        StakeTableFetcher::fetch_all_validators(l1, stake_table_address, l1_block_number).await?;
20
21    Ok(validators
22        .into_iter()
23        .map(|(_address, validator)| validator)
24        .collect())
25}

27pub fn display_stake_table(stake_table: Vec<Validator<BLSPubKey>>, compact: bool) -> Result<()> {
28    let mut stake_table = stake_table.clone();
29    stake_table.sort_by(|a, b| a.stake.cmp(&b.stake));
30
31    for validator in stake_table.iter() {
32        let comm: Commission = validator.commission.try_into()?;
33        let bls_key = validator.stake_table_key.to_string();
34        let key_str = if compact {
35            let end = bls_key.chars().map(|c| c.len_utf8()).take(40).sum();
36            format!("{}..", &bls_key[..end])
37        } else {
38            bls_key.to_string()
39        };
40        tracing::info!(
41            "Validator {}: {key_str} comm={comm} stake={} ESP",
42            validator.account,
43            format_ether(validator.stake),
44        );
45
46        if validator.delegators.is_empty() {
47            tracing::info!(" - No delegators");
48            continue;
49        }
50
51        // sort delegators by address for easier reading
52        let mut delegators = validator.delegators.iter().collect::<Vec<_>>();
53        delegators.sort_by(|a, b| a.0.cmp(b.0));
54        for (delegator, stake) in delegators {
55            tracing::info!(
56                " - Delegator {delegator}: stake={} ESP",
57                format_ether(*stake)
58            );
59        }
60    }
61    Ok(())
62}
