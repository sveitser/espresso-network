64pub async fn stake_in_contract_for_test(
65    rpc_url: Url,
66    grant_recipient: PrivateKeySigner,
67    stake_table_address: Address,
68    token_address: Address,
69    validator_keys: Vec<(PrivateKeySigner, BLSKeyPair, StateKeyPair)>,
70    multiple_delegators: bool,
71) -> Result<()> {
72    tracing::info!("staking to stake table contract for demo");
73
74    tracing::info!("stake table address: {}", stake_table_address);
75
76    let token_signer = make_provider(&rpc_url, grant_recipient.clone());
77
78    tracing::info!("ESP token address: {token_address}");
79    let token = EspToken::new(token_address, token_signer.clone());
80    let token_balance = token.balanceOf(grant_recipient.address()).call().await?._0;
81    tracing::info!(
82        "token distributor account {} balance: {} ESP",
83        token_signer.default_signer_address(),
84        format_ether(token_balance)
85    );
86    if token_balance.is_zero() {
87        panic!("grant recipient has no ESP tokens, funding won't work");
88    }
89
90    let fund_amount_esp = parse_ether("1000")?;
91    let fund_amount_eth = parse_ether("10")?;
92
93    // Set up deterministic rng
94    let seed = [42u8; 32];
95    let mut rng = ChaCha20Rng::from_seed(seed);
96
97    for (val_index, (signer, bls_key_pair, state_key_pair)) in
98        validator_keys.into_iter().enumerate()
99    {
100        let validator_provider = make_provider(&rpc_url, signer);
101        let validator_address = validator_provider.default_signer_address();
102
103        tracing::info!("fund val {val_index} address: {validator_address}, {fund_amount_eth} ETH");
104        let tx = TransactionRequest::default()
105            .with_to(validator_address)
106            .with_value(fund_amount_eth);
107        let receipt = token_signer
108            .send_transaction(tx)
109            .await?
110            .get_receipt()
111            .await?;
112        assert!(receipt.status());
113
114        let bal = validator_provider.get_balance(validator_address).await?;
115
116        // 1% commission and more
117        let commission = Commission::try_from(100u64 + 10u64 * val_index as u64)?;
118
119        // delegate 100 to 500 ESP
120        let delegate_amount = parse_ether("100")? * U256::from(val_index % 5 + 1);
121        let delegate_amount_esp = format_ether(delegate_amount);
122
123        tracing::info!("validator {val_index} address: {validator_address}, balance: {bal}");
124
125        tracing::info!("transfer {fund_amount_esp} ESP to {validator_address}",);
126        let receipt = token
127            .transfer(validator_address, fund_amount_esp)
128            .send()
129            .await
130            .maybe_decode_revert::<EspTokenErrors>()?
131            .get_receipt()
132            .await?;
133        assert!(receipt.status());
134
135        tracing::info!("approve {fund_amount_esp} ESP for {stake_table_address}",);
136        let validator_token = EspToken::new(token_address, validator_provider.clone());
137        let receipt = validator_token
138            .approve(stake_table_address, fund_amount_esp)
139            .send()
140            .await
141            .maybe_decode_revert::<EspTokenErrors>()?
142            .get_receipt()
143            .await?;
144        assert!(receipt.status());
145
146        tracing::info!("deploy validator {val_index} with commission {commission}");
147        let receipt = register_validator(
148            &validator_provider,
149            stake_table_address,
150            commission,
151            validator_address,
152            bls_key_pair,
153            state_key_pair.ver_key(),
154        )
155        .await?;
156        assert!(receipt.status());
157
158        tracing::info!(
159            "delegate {delegate_amount_esp} ESP for validator {val_index} from {validator_address}"
160        );
161        let receipt = delegate(
162            &validator_provider,
163            stake_table_address,
164            validator_address,
165            delegate_amount,
166        )
167        .await?;
168        assert!(receipt.status());
169
170        if multiple_delegators {
171            tracing::info!("adding multiple delegators for validator  {val_index} ");
172
173            let num_delegators = rng.gen_range(2..=5);
174
175            add_multiple_delegators(
176                &rpc_url,
177                validator_address,
178                &token_signer,
179                &token,
180                stake_table_address,
181                token_address,
182                &mut rng,
183                num_delegators,
184            )
185            .await?;
186        }
187    }
188    tracing::info!("completed staking for demo");
189    Ok(())
190}

192#[allow(clippy::too_many_arguments)]
193async fn add_multiple_delegators<F: TxFiller<Ethereum>, P: Provider<Ethereum>>(
194    rpc_url: &Url,
195    validator_address: Address,
196    token_signer: &FillProvider<F, P, Ethereum>,
197    token: &EspToken::EspTokenInstance<(), FillProvider<F, P, Ethereum>>,
198    stake_table_address: Address,
199    token_address: Address,
200    rng: &mut ChaCha20Rng,
201    num_delegators: u64,
202) -> Result<()> {
203    let fund_amount_esp = parse_ether("1000")?;
204    let fund_amount_eth = parse_ether("10")?;
205
206    for delegator_index in 0..num_delegators {
207        let delegator_wallet: LocalSigner<SigningKey> = SigningKey::random(rng).into();
208        let delegator_address = delegator_wallet.address();
209        tracing::info!("delegator {delegator_index}: address {delegator_address}");
210
211        let tx = TransactionRequest::default()
212            .with_to(delegator_address)
213            .with_value(fund_amount_eth);
214        let receipt = token_signer
215            .send_transaction(tx)
216            .await?
217            .get_receipt()
218            .await?;
219        assert!(receipt.status());
220
221        tracing::info!("delegator {delegator_index}: funded with {fund_amount_eth} ETH");
222
223        let random_amount: u64 = rng.gen_range(100..=500);
224        let delegate_amount = parse_ether(&random_amount.to_string())?;
225        let delegate_amount_esp = format_ether(delegate_amount);
226
227        let receipt = token
228            .transfer(delegator_address, fund_amount_esp)
229            .send()
230            .await?
231            .get_receipt()
232            .await?;
233        assert!(receipt.status());
234
235        tracing::info!("delegator {delegator_index}: received {fund_amount_esp} ESP");
236
237        let delegator_provider = make_provider(rpc_url, delegator_wallet.clone());
238
239        let validator_token = EspToken::new(token_address, delegator_provider.clone());
240        let receipt = validator_token
241            .approve(stake_table_address, delegate_amount)
242            .send()
243            .await?
244            .get_receipt()
245            .await?;
246        assert!(receipt.status());
247
248        tracing::info!(
249            "delegator {delegator_index}: approved {delegate_amount_esp} ESP to stake table"
250        );
251
252        let receipt = delegate(
253            &delegator_provider,
254            stake_table_address,
255            validator_address,
256            delegate_amount,
257        )
258        .await?;
259        assert!(receipt.status());
260
261        tracing::info!("delegator {delegator_index}: delegation complete");
262    }
263
264    Ok(())
265}

267/// Register validators, and delegate to themselves for demo purposes.
268///
269/// The environment variables used only for this function but not for the normal staking CLI are
270/// loaded directly from the environment.
271///
272/// Account indexes 20+ of the dev mnemonic are used for the validator accounts.
273pub async fn stake_for_demo(config: &Config, num_validators: u16) -> Result<()> {
274    tracing::info!("staking to stake table contract for demo");
275
276    let mk_signer = |account_index| -> Result<PrivateKeySigner> {
277        Ok(MnemonicBuilder::<English>::default()
278            .phrase(config.signer.mnemonic.as_ref().unwrap())
279            .index(account_index)?
280            .build()?)
281    };
282
283    let grant_recipient = mk_signer(config.signer.account_index.unwrap())?;
284
285    tracing::info!(
286        "grant recipient account for token funding: {}",
287        grant_recipient.address()
288    );
289
290    let token_address = config.token_address;
291    tracing::info!("ESP token address: {}", token_address);
292    let stake_table_address = config.stake_table_address;
293    tracing::info!("stake table address: {}", stake_table_address);
294
295    let mut validator_keys = vec![];
296    for val_index in 0..num_validators {
297        let signer = mk_signer(20u32 + val_index as u32)?;
298        let consensus_private_key = parse_bls_priv_key(&dotenvy::var(format!(
299            "ESPRESSO_DEMO_SEQUENCER_STAKING_PRIVATE_KEY_{val_index}"
300        ))?)?
301        .into();
302        let state_private_key = parse_state_priv_key(&dotenvy::var(format!(
303            "ESPRESSO_DEMO_SEQUENCER_STATE_PRIVATE_KEY_{val_index}"
304        ))?)?;
305        validator_keys.push((
306            signer,
307            consensus_private_key,
308            StateKeyPair::from_sign_key(state_private_key),
309        ));
310    }
311
312    stake_in_contract_for_test(
313        config.rpc_url.clone(),
314        grant_recipient,
315        config.stake_table_address,
316        config.token_address,
317        validator_keys,
318        false,
319    )
320    .await?;
321
322    tracing::info!("completed staking for demo");
323    Ok(())
324}

326/// Commonly used contract deployment routine.
327// TODO move to proper place for shared code. See:
328// https://github.com/EspressoSystems/espresso-network/pull/3083#discussion_r2048832370
329#[allow(clippy::too_many_arguments)]
330pub async fn pos_deploy_routine(
331    l1_url: &Url,
332    signer: &LocalSigner<SigningKey>, // TODO maybe from_instance(AnvilInstance)
333    blocks_per_epoch: u64,
334    epoch_start_block: u64,
335    initial_stake_table: StakeTableVecBased,
336    private_keys: Vec<(PrivateKeySigner, BLSKeyPair, StateKeyPair)>,
337    _multisig: Option<Address>,
338    multiple_delegators: bool,
339) -> anyhow::Result<Address> {
340    let contracts = &mut Contracts::new();
341
342    let wallet = EthereumWallet::from(signer.clone());
343    let provider = ProviderBuilder::new()
344        .wallet(wallet.clone())
345        .on_http(l1_url.clone());
346    let admin = provider.get_accounts().await?[0];
347
348    let (genesis_state, genesis_stake) =
349        legacy_light_client_genesis_from_stake_table(initial_stake_table.clone())?;
350
351    // deploy EspToken, proxy
352    let token_proxy_addr = deployer::deploy_token_proxy(&provider, contracts, admin, admin).await?;
353
354    // deploy light client v1, proxy
355    let lc_proxy_addr = deployer::deploy_light_client_proxy(
356        &provider,
357        contracts,
358        true, // use mock
359        genesis_state.clone(),
360        genesis_stake.clone(),
361        admin,
362        None, // no permissioned prover
363    )
364    .await?;
365    // upgrade to LightClientV2
366    deployer::upgrade_light_client_v2(
367        &provider,
368        contracts,
369        true, // use mock
370        blocks_per_epoch,
371        epoch_start_block,
372    )
373    .await?;
374
375    // deploy permissionless stake table
376    let exit_escrow_period = U256::from(300); // 300 sec
377    let _stake_table_proxy_addr = deployer::deploy_stake_table_proxy(
378        &provider,
379        contracts,
380        token_proxy_addr,
381        lc_proxy_addr,
382        exit_escrow_period,
383        admin,
384    )
385    .await?;
386
387    let stake_table_address = contracts
388        .address(Contract::StakeTableProxy)
389        .expect("stake table deployed");
390
391    stake_in_contract_for_test(
392        l1_url.clone(),
393        signer.clone(),
394        stake_table_address,
395        contracts
396            .address(Contract::EspTokenProxy)
397            .expect("ESP token deployed"),
398        private_keys,
399        multiple_delegators,
400    )
401    .await?;
402
403    Ok(stake_table_address)
404}

428    fn staking_priv_keys() -> Vec<(PrivateKeySigner, BLSKeyPair, StateKeyPair)> {
429        let seed = [42u8; 32];
430        let num_nodes = STAKE_TABLE_CAPACITY_FOR_TEST;
431
432        let (_, priv_keys): (Vec<_>, Vec<_>) = (0..num_nodes)
433            .map(|i| <PubKey as SignatureKey>::generated_from_seed_indexed(seed, i))
434            .unzip();
435        let state_key_pairs = (0..num_nodes)
436            .map(|i| StateKeyPair::generate_from_seed_indexed(seed, i))
437            .collect::<Vec<_>>();
438
439        let mut rng = ChaCha20Rng::from_seed([42u8; 32]); // Create a deterministic RNG
440        let eth_key_pairs = (0..num_nodes).map(|_| SigningKey::random(&mut rng).into());
441        eth_key_pairs
442            .zip(priv_keys.iter())
443            .zip(state_key_pairs.iter())
444            .map(|((eth, bls), state)| (eth, bls.clone().into(), state.clone()))
445            .collect()
446    }
