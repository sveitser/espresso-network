109pub struct NetworkParams {
110    /// The address where a CDN marshal is located
111    pub cdn_endpoint: String,
112    pub orchestrator_url: Url,
113    pub state_relay_server_url: Url,
114    pub private_staking_key: BLSPrivKey,
115    pub private_state_key: StateSignKey,
116    pub state_peers: Vec<Url>,
117    pub config_peers: Option<Vec<Url>>,
118    pub catchup_backoff: BackoffParams,
119    /// The address to advertise as our public API's URL
120    pub public_api_url: Option<Url>,
121
122    /// The address to send to other Libp2p nodes to contact us
123    pub libp2p_advertise_address: String,
124    /// The address to bind to for Libp2p
125    pub libp2p_bind_address: String,
126    /// The (optional) bootstrap node addresses for Libp2p. If supplied, these will
127    /// override the bootstrap nodes specified in the config file.
128    pub libp2p_bootstrap_nodes: Option<Vec<Multiaddr>>,
129
130    /// The heartbeat interval
131    pub libp2p_heartbeat_interval: Duration,
132
133    /// The number of past heartbeats to gossip about
134    pub libp2p_history_gossip: usize,
135    /// The number of past heartbeats to remember the full messages for
136    pub libp2p_history_length: usize,
137
138    /// The target number of peers in the mesh
139    pub libp2p_mesh_n: usize,
140    /// The maximum number of peers in the mesh
141    pub libp2p_mesh_n_high: usize,
142    /// The minimum number of peers in the mesh
143    pub libp2p_mesh_n_low: usize,
144    /// The minimum number of mesh peers that must be outbound
145    pub libp2p_mesh_outbound_min: usize,
146
147    /// The maximum gossip message size
148    pub libp2p_max_gossip_transmit_size: usize,
149
150    /// The maximum direct message size
151    pub libp2p_max_direct_transmit_size: u64,
152
153    /// The maximum number of IHAVE messages to accept from a Libp2p peer within a heartbeat
154    pub libp2p_max_ihave_length: usize,
155
156    /// The maximum number of IHAVE messages to accept from a Libp2p peer within a heartbeat
157    pub libp2p_max_ihave_messages: usize,
158
159    /// The time period that message hashes are stored in the cache
160    pub libp2p_published_message_ids_cache_time: Duration,
161
162    /// The time to wait for a Libp2p message requested through IWANT following an IHAVE advertisement
163    pub libp2p_iwant_followup_time: Duration,
164
165    /// The maximum number of Libp2p messages we will process in a given RPC
166    pub libp2p_max_messages_per_rpc: Option<usize>,
167
168    /// How many times we will allow a peer to request the same message id through IWANT gossip before we start ignoring them
169    pub libp2p_gossip_retransmission: u32,
170
171    /// If enabled newly created messages will always be sent to all peers that are subscribed to the topic and have a good enough score
172    pub libp2p_flood_publish: bool,
173
174    /// The time period that Libp2p message hashes are stored in the cache
175    pub libp2p_duplicate_cache_time: Duration,
176
177    /// Time to live for Libp2p fanout peers
178    pub libp2p_fanout_ttl: Duration,
179
180    /// Initial delay in each Libp2p heartbeat
181    pub libp2p_heartbeat_initial_delay: Duration,
182
183    /// How many Libp2p peers we will emit gossip to at each heartbeat
184    pub libp2p_gossip_factor: f64,
185
186    /// Minimum number of Libp2p peers to emit gossip to during a heartbeat
187    pub libp2p_gossip_lazy: usize,
188}

196pub async fn init_node<P: SequencerPersistence + MembershipPersistence, V: Versions>(
197    genesis: Genesis,
198    network_params: NetworkParams,
199    metrics: &dyn Metrics,
200    persistence: P,
201    l1_params: L1Params,
202    storage: Option<Arc<SqlStorage>>,
203    seq_versions: V,
204    event_consumer: impl EventConsumer + 'static,
205    is_da: bool,
206    identity: Identity,
207    proposal_fetcher_config: ProposalFetcherConfig,
208) -> anyhow::Result<SequencerContext<network::Production, P, V>>
209where
210    Arc<P>: Storage<SeqTypes>,
211{
212    // Expose git information via status API.
213    metrics
214        .text_family(
215            "version".into(),
216            vec!["rev".into(), "desc".into(), "timestamp".into()],
217        )
218        .create(vec![
219            env!("VERGEN_GIT_SHA").into(),
220            env!("VERGEN_GIT_DESCRIBE").into(),
221            env!("VERGEN_GIT_COMMIT_TIMESTAMP").into(),
222        ]);
223
224    // Expose Node Entity Information via the status/metrics API
225    metrics
226        .text_family(
227            "node_identity_general".into(),
228            vec![
229                "name".into(),
230                "company_name".into(),
231                "company_website".into(),
232                "operating_system".into(),
233                "node_type".into(),
234                "network_type".into(),
235            ],
236        )
237        .create(vec![
238            identity.node_name.unwrap_or("".into()),
239            identity.company_name.unwrap_or("".into()),
240            identity
241                .company_website
242                .map(|u| u.into())
243                .unwrap_or("".into()),
244            identity.operating_system.unwrap_or("".into()),
245            identity.node_type.unwrap_or("".into()),
246            identity.network_type.unwrap_or("".into()),
247        ]);
248
249    // Expose Node Identity Location via the status/metrics API
250    metrics
251        .text_family(
252            "node_identity_location".into(),
253            vec!["country".into(), "latitude".into(), "longitude".into()],
254        )
255        .create(vec![
256            identity.country_code.unwrap_or("".into()),
257            identity
258                .latitude
259                .map(|l| l.to_string())
260                .unwrap_or("".into()),
261            identity
262                .longitude
263                .map(|l| l.to_string())
264                .unwrap_or("".into()),
265        ]);
266
267    // Stick our public key in `metrics` so it is easily accessible via the status API.
268    let pub_key = BLSPubKey::from_private(&network_params.private_staking_key);
269    metrics
270        .text_family("node".into(), vec!["key".into()])
271        .create(vec![pub_key.to_string()]);
272
273    // Parse the Libp2p bind and advertise addresses to multiaddresses
274    let libp2p_bind_address = derive_libp2p_multiaddr(&network_params.libp2p_bind_address)
275        .with_context(|| {
276            format!(
277                "Failed to derive Libp2p bind address of {}",
278                &network_params.libp2p_bind_address
279            )
280        })?;
281    let libp2p_advertise_address =
282        derive_libp2p_multiaddr(&network_params.libp2p_advertise_address).with_context(|| {
283            format!(
284                "Failed to derive Libp2p advertise address of {}",
285                &network_params.libp2p_advertise_address
286            )
287        })?;
288
289    info!("Libp2p bind address: {}", libp2p_bind_address);
290    info!("Libp2p advertise address: {}", libp2p_advertise_address);
291
292    // Orchestrator client
293    let orchestrator_client = OrchestratorClient::new(network_params.orchestrator_url);
294    let state_key_pair = StateKeyPair::from_sign_key(network_params.private_state_key);
295    let validator_config = ValidatorConfig {
296        public_key: pub_key,
297        private_key: network_params.private_staking_key,
298        stake_value: U256::ONE,
299        state_public_key: state_key_pair.ver_key(),
300        state_private_key: state_key_pair.sign_key(),
301        is_da,
302    };
303
304    // Derive our Libp2p public key from our private key
305    let libp2p_public_key = derive_libp2p_peer_id::<<SeqTypes as NodeType>::SignatureKey>(
306        &validator_config.private_key,
307    )
308    .with_context(|| "Failed to derive Libp2p peer ID")?;
309
310    // Print the libp2p public key
311    info!("Starting Libp2p with PeerID: {}", libp2p_public_key);
312
313    let (mut network_config, wait_for_orchestrator) = match (
314        persistence.load_config().await?,
315        network_params.config_peers,
316    ) {
317        (Some(config), _) => {
318            tracing::info!("loaded network config from storage, rejoining existing network");
319            (config, false)
320        },
321        // If we were told to fetch the config from an already-started peer, do so.
322        (None, Some(peers)) => {
323            tracing::info!(?peers, "loading network config from peers");
324            let peers = StatePeers::<SequencerApiVersion>::from_urls(
325                peers,
326                network_params.catchup_backoff,
327                &NoMetrics,
328            );
329            let config = peers.fetch_config(validator_config.clone()).await?;
330
331            tracing::info!(
332                node_id = config.node_index,
333                stake_table = ?config.config.known_nodes_with_stake,
334                "loaded config",
335            );
336            persistence.save_config(&config).await?;
337            (config, false)
338        },
339        // Otherwise, this is a fresh network; load from the orchestrator.
340        (None, None) => {
341            tracing::info!("loading network config from orchestrator");
342            tracing::error!(
343                "waiting for other nodes to connect, DO NOT RESTART until fully connected"
344            );
345            let config = get_complete_config(
346                &orchestrator_client,
347                validator_config.clone(),
348                // Register in our Libp2p advertise address and public key so other nodes
349                // can contact us on startup
350                Some(libp2p_advertise_address),
351                Some(libp2p_public_key),
352            )
353            .await?
354            .0;
355
356            tracing::info!(
357                node_id = config.node_index,
358                stake_table = ?config.config.known_nodes_with_stake,
359                "loaded config",
360            );
361            persistence.save_config(&config).await?;
362            tracing::error!("all nodes connected");
363            (config, true)
364        },
365    };
366
367    if let Some(upgrade) = genesis.upgrades.get(&V::Upgrade::VERSION) {
368        upgrade.set_hotshot_config_parameters(&mut network_config.config);
369    }
370
371    let epoch_height = genesis.epoch_height.unwrap_or_default();
372    let epoch_start_block = genesis.epoch_start_block.unwrap_or_default();
373
374    tracing::info!("setting epoch_height={epoch_height:?}");
375    tracing::info!("setting epoch_start_block={epoch_start_block:?}");
376    network_config.config.epoch_height = epoch_height;
377    network_config.config.epoch_start_block = epoch_start_block;
378
379    // If the `Libp2p` bootstrap nodes were supplied via the command line, override those
380    // present in the config file.
381    if let Some(bootstrap_nodes) = network_params.libp2p_bootstrap_nodes {
382        if let Some(libp2p_config) = network_config.libp2p_config.as_mut() {
383            // If the libp2p configuration is present, we can override the bootstrap nodes.
384
385            // Split off the peer ID from the addresses
386            libp2p_config.bootstrap_nodes = bootstrap_nodes
387                .into_iter()
388                .map(split_off_peer_id)
389                .collect::<Result<Vec<_>, _>>()
390                .with_context(|| "Failed to parse peer ID from bootstrap node")?;
391        } else {
392            // If not, don't try launching with them. Eventually we may want to
393            // provide a default configuration here instead.
394            tracing::warn!("No libp2p configuration found, ignoring supplied bootstrap nodes");
395        }
396    }
397
398    let node_index = network_config.node_index;
399
400    // If we are a DA node, we need to subscribe to the DA topic
401    let topics = {
402        let mut topics = vec![CdnTopic::Global];
403        if is_da {
404            topics.push(CdnTopic::Da);
405        }
406        topics
407    };
408
409    // Initialize the push CDN network (and perform the initial connection)
410    let cdn_network = PushCdnNetwork::new(
411        network_params.cdn_endpoint,
412        topics,
413        KeyPair {
414            public_key: WrappedSignatureKey(validator_config.public_key),
415            private_key: validator_config.private_key.clone(),
416        },
417        CdnMetricsValue::new(metrics),
418    )
419    .with_context(|| format!("Failed to create CDN network {node_index}"))?;
420
421    // Configure gossipsub based on the command line options
422    let gossip_config = GossipConfig {
423        heartbeat_interval: network_params.libp2p_heartbeat_interval,
424        history_gossip: network_params.libp2p_history_gossip,
425        history_length: network_params.libp2p_history_length,
426        mesh_n: network_params.libp2p_mesh_n,
427        mesh_n_high: network_params.libp2p_mesh_n_high,
428        mesh_n_low: network_params.libp2p_mesh_n_low,
429        mesh_outbound_min: network_params.libp2p_mesh_outbound_min,
430        max_ihave_messages: network_params.libp2p_max_ihave_messages,
431        max_transmit_size: network_params.libp2p_max_gossip_transmit_size,
432        max_ihave_length: network_params.libp2p_max_ihave_length,
433        published_message_ids_cache_time: network_params.libp2p_published_message_ids_cache_time,
434        iwant_followup_time: network_params.libp2p_iwant_followup_time,
435        max_messages_per_rpc: network_params.libp2p_max_messages_per_rpc,
436        gossip_retransmission: network_params.libp2p_gossip_retransmission,
437        flood_publish: network_params.libp2p_flood_publish,
438        duplicate_cache_time: network_params.libp2p_duplicate_cache_time,
439        fanout_ttl: network_params.libp2p_fanout_ttl,
440        heartbeat_initial_delay: network_params.libp2p_heartbeat_initial_delay,
441        gossip_factor: network_params.libp2p_gossip_factor,
442        gossip_lazy: network_params.libp2p_gossip_lazy,
443    };
444
445    // Configure request/response based on the command line options
446    let request_response_config = RequestResponseConfig {
447        request_size_maximum: network_params.libp2p_max_direct_transmit_size,
448        response_size_maximum: network_params.libp2p_max_direct_transmit_size,
449    };
450
451    let l1_client = l1_params
452        .options
453        .with_metrics(metrics)
454        .connect(l1_params.urls)
455        .with_context(|| "failed to create L1 client")?;
456    genesis.validate_fee_contract(&l1_client).await?;
457
458    l1_client.spawn_tasks().await;
459    let l1_genesis = match genesis.l1_finalized {
460        L1Finalized::Block(b) => b,
461        L1Finalized::Number { number } => l1_client.wait_for_finalized_block(number).await,
462        L1Finalized::Timestamp { timestamp } => {
463            l1_client
464                .wait_for_finalized_block_with_timestamp(U256::from(timestamp.unix_timestamp()))
465                .await
466        },
467    };
468
469    let mut genesis_state = ValidatedState {
470        chain_config: genesis.chain_config.into(),
471        ..Default::default()
472    };
473    for (address, amount) in genesis.accounts {
474        tracing::info!(%address, %amount, "Prefunding account for demo");
475        genesis_state.prefund_account(address, amount);
476    }
477
478    // Create the list of parallel catchup providers
479    let state_catchup_providers = ParallelStateCatchup::new(&[]);
480
481    // Add the state peers to the list
482    let state_peers = StatePeers::<SequencerApiVersion>::from_urls(
483        network_params.state_peers,
484        network_params.catchup_backoff,
485        metrics,
486    );
487    state_catchup_providers.add_provider(Arc::new(state_peers));
488
489    // Add the local (persistence) catchup provider to the list (if we can)
490    match persistence
491        .clone()
492        .into_catchup_provider(network_params.catchup_backoff)
493    {
494        Ok(catchup) => {
495            state_catchup_providers.add_provider(Arc::new(catchup));
496        },
497        Err(e) => {
498            tracing::warn!(
499                "Failed to create local catchup provider: {e:#}. Only using remote catchup."
500            );
501        },
502    };
503
504    let fetcher = StakeTableFetcher::new(
505        Arc::new(state_catchup_providers.clone()),
506        Arc::new(Mutex::new(persistence.clone())),
507        l1_client.clone(),
508        genesis.chain_config,
509    );
510    fetcher.spawn_update_loop().await;
511    // Create the HotShot membership
512    let mut membership = EpochCommittees::new_stake(
513        network_config.config.known_nodes_with_stake.clone(),
514        network_config.config.known_da_nodes.clone(),
515        fetcher,
516    );
517    membership.reload_stake(RECENT_STAKE_TABLES_LIMIT).await;
518
519    let membership: Arc<RwLock<EpochCommittees>> = Arc::new(RwLock::new(membership));
520    let persistence = Arc::new(persistence);
521    let coordinator = EpochMembershipCoordinator::new(
522        membership,
523        Some(storage_add_drb_result(persistence.clone())),
524        network_config.config.epoch_height,
525    );
526
527    let instance_state = NodeState {
528        chain_config: genesis.chain_config,
529        l1_client,
530        genesis_header: genesis.header,
531        genesis_state,
532        l1_genesis: Some(l1_genesis),
533        node_id: node_index,
534        upgrades: genesis.upgrades,
535        current_version: V::Base::VERSION,
536        epoch_height: Some(epoch_height),
537        state_catchup: Arc::new(state_catchup_providers.clone()),
538        coordinator: coordinator.clone(),
539    };
540
541    // Initialize the Libp2p network
542    let network = {
543        let p2p_network = Libp2pNetwork::from_config(
544            network_config.clone(),
545            DhtNoPersistence,
546            coordinator.membership().clone(),
547            gossip_config,
548            request_response_config,
549            libp2p_bind_address,
550            &validator_config.public_key,
551            // We need the private key so we can derive our Libp2p keypair
552            // (using https://docs.rs/blake3/latest/blake3/fn.derive_key.html)
553            &validator_config.private_key,
554            hotshot::traits::implementations::Libp2pMetricsValue::new(metrics),
555        )
556        .await
557        .with_context(|| {
558            format!(
559                "Failed to create libp2p network on node {node_index}; binding to {:?}",
560                network_params.libp2p_bind_address
561            )
562        })?;
563
564        tracing::warn!("Waiting for at least one connection to be initialized");
565        select! {
566            _ = cdn_network.wait_for_ready() => {
567                tracing::warn!("CDN connection initialized");
568            },
569            _ = p2p_network.wait_for_ready() => {
570                tracing::warn!("P2P connection initialized");
571            },
572        };
573
574        // Combine the CDN and P2P networks
575        Arc::from(CombinedNetworks::new(
576            cdn_network,
577            p2p_network,
578            Some(Duration::from_secs(1)),
579        ))
580    };
581
582    let mut ctx = SequencerContext::init(
583        network_config,
584        validator_config,
585        coordinator,
586        instance_state,
587        storage,
588        state_catchup_providers,
589        persistence,
590        network,
591        Some(network_params.state_relay_server_url),
592        metrics,
593        genesis.stake_table.capacity,
594        event_consumer,
595        seq_versions,
596        proposal_fetcher_config,
597    )
598    .await?;
599    if wait_for_orchestrator {
600        ctx = ctx.wait_for_orchestrator(orchestrator_client);
601    }
602    Ok(ctx)
603}

1111            &self,
1112            i: usize,

1114            mut persistence_opt: P,

1116            storage: Option<Arc<SqlStorage>>,
1117            metrics: &dyn Metrics,
1118            stake_table_capacity: usize,
1119            event_consumer: impl EventConsumer + 'static,
1120            bind_version: V,
1121            upgrades: BTreeMap<Version, Upgrade>,
1122        ) -> SequencerContext<network::Memory, P::Persistence, V> {
1123            let config = self.config.clone();
1124            let my_peer_config = &config.known_nodes_with_stake[i];
1125            let is_da = config.known_da_nodes.contains(my_peer_config);
1126

1129                public_key: my_peer_config.stake_table_entry.stake_key,
1130                private_key: self.priv_keys[i].clone(),
1131                stake_value: my_peer_config.stake_table_entry.stake_amount,
1132                state_public_key: self.state_key_pairs[i].ver_key(),
1133                state_private_key: self.state_key_pairs[i].sign_key(),
1134                is_da,
1135            };
1136
1137            let topics = if is_da {
1138                vec![Topic::Global, Topic::Da]
1139            } else {
1140                vec![Topic::Global]
1141            };

1144                &my_peer_config.stake_table_entry.stake_key,
1145                &self.master_map,

1148            ));
1149
1150            // Make sure the builder account is funded.
1151            let builder_account = Self::builder_key().fee_account();

1156
1157            let chain_config = state.chain_config.resolve().unwrap_or_default();

1164                catchup_providers.add_provider(Arc::new(state_peers));

1169                .clone()

1188            };

1190                .connect(vec![self.l1_url.clone()])
1191                .expect("failed to create L1 client");

1193

1195                Arc::new(catchup_providers.clone()),
1196                Arc::new(Mutex::new(persistence.clone())),
1197                l1_client.clone(),
1198                chain_config,
1199            );
1200            fetcher.spawn_update_loop().await;
1201
1202            let mut membership = EpochCommittees::new_stake(
1203                config.known_nodes_with_stake.clone(),
1204                config.known_da_nodes.clone(),
1205                fetcher,
1206            );
1207            membership.reload_stake(50).await;
1208

1210            let persistence = Arc::new(persistence);

1214                Some(storage_add_drb_result(persistence.clone())),
1215                config.epoch_height,
1216            );
1217
1218            let node_state = NodeState::new(
1219                i as u64,
1220                chain_config,
1221                l1_client,
1222                Arc::new(catchup_providers.clone()),
1223                V::Base::VERSION,
1224                coordinator.clone(),
1225            )
1226            .with_current_version(V::Base::version())
1227            .with_genesis(state)
1228            .with_epoch_height(config.epoch_height)
1229            .with_upgrades(upgrades);

1232                i,
1233                key = %my_peer_config.stake_table_entry.stake_key,
1234                state_key = %my_peer_config.state_ver_key,
1235                "starting node",
1236            );

1240                    config,
1241                    // For testing, we use a fake network, so the rest of the network config beyond
1242                    // the base consensus config does not matter.

1244                },

1268    // Wait for decide event, make sure it matches submitted transaction. Return the block number
1269    // containing the transaction.
1270    pub async fn wait_for_decide_on_handle(
1271        events: &mut (impl Stream<Item = Event> + Unpin),
1272        submitted_txn: &Transaction,
1273    ) -> u64 {
1274        let commitment = submitted_txn.commit();
1275
1276        // Keep getting events until we see a Decide event
1277        loop {
1278            let event = events.next().await.unwrap();
1279            tracing::info!("Received event from handle: {event:?}");
1280
1281            if let Decide { leaf_chain, .. } = event.event {
1282                if let Some(height) = leaf_chain.iter().find_map(|LeafInfo { leaf, .. }| {
1283                    if leaf
1284                        .block_payload()
1285                        .as_ref()?
1286                        .transaction_commitments(leaf.block_header().metadata())
1287                        .contains(&commitment)
1288                    {
1289                        Some(leaf.block_header().block_number())
1290                    } else {
1291                        None
1292                    }
1293                }) {
1294                    return height;
1295                }
1296            } else {
1297                // Keep waiting
1298            }
1299        }
1300    }
