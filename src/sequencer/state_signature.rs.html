34/// Capacity for the in memory signature storage.
35const SIGNATURE_STORAGE_CAPACITY: usize = 100;
36
37#[derive(Debug)]
38pub struct StateSigner<ApiVer: StaticVersionType> {
39    /// Key for signing a new light client state
40    sign_key: StateSignKey,
41
42    /// Key for verifying a light client state
43    ver_key: StateVerKey,
44
45    /// The most recent light client state signatures
46    signatures: RwLock<StateSignatureMemStorage>,
47
48    /// Commitment for current fixed stake table
49    voting_stake_table: StakeTableState,
50
51    /// epoch for the current stake table state
52    voting_stake_table_epoch: Option<<SeqTypes as NodeType>::Epoch>,
53
54    /// Capacity of the stake table
55    stake_table_capacity: u64,
56
57    /// The state relay server url
58    relay_server_client: Option<Client<ServerError, ApiVer>>,
59}

61impl<ApiVer: StaticVersionType> StateSigner<ApiVer> {
62    pub fn new(
63        sign_key: StateSignKey,
64        ver_key: StateVerKey,
65        voting_stake_table: StakeTableState,
66        voting_stake_table_epoch: Option<<SeqTypes as NodeType>::Epoch>,
67        stake_table_capacity: u64,
68    ) -> Self {
69        Self {
70            sign_key,
71            ver_key,
72            voting_stake_table,
73            voting_stake_table_epoch,
74            stake_table_capacity,
75            signatures: Default::default(),
76            relay_server_client: Default::default(),
77        }
78    }
79
80    /// Connect to the given state relay server to send signed HotShot states to.
81    pub fn with_relay_server(mut self, url: Url) -> Self {
82        self.relay_server_client = Some(Client::new(url));
83        self
84    }
85
86    pub(super) async fn handle_event<N, P, V>(
87        &mut self,
88        event: &Event<SeqTypes>,
89        consensus_state: Arc<RwLock<Consensus<N, P, V>>>,
90    ) where
91        N: ConnectedNetwork<PubKey>,
92        P: SequencerPersistence,
93        V: Versions,
94    {
95        let EventType::Decide { leaf_chain, .. } = &event.event else {
96            return;
97        };
98        let Some(LeafInfo { leaf, .. }) = leaf_chain.first() else {
99            return;
100        };
101        match leaf
102            .block_header()
103            .get_light_client_state(leaf.view_number())
104        {
105            Ok(state) => {
106                tracing::debug!("New leaves decided. Latest block height: {}", leaf.height(),);
107
108                let consensus = consensus_state.read().await;
109                let cur_block_height = state.block_height;
110                let blocks_per_epoch = consensus.epoch_height;
111
112                // The last few state updates are handled in the consensus, we do not sign them.
113                if leaf.with_epoch & is_ge_epoch_root(cur_block_height, blocks_per_epoch) {
114                    return;
115                }
116
117                let option_state_epoch = option_epoch_from_block_number::<SeqTypes>(
118                    leaf.with_epoch,
119                    cur_block_height,
120                    blocks_per_epoch,
121                );
122
123                if self.voting_stake_table_epoch != option_state_epoch {
124                    let Ok(membership) = consensus
125                        .membership_coordinator
126                        .stake_table_for_epoch(option_state_epoch)
127                        .await
128                    else {
129                        tracing::error!(
130                            "Fail to get membership for epoch: {:?}",
131                            option_state_epoch
132                        );
133                        return;
134                    };
135                    self.voting_stake_table_epoch = option_state_epoch;
136                    self.voting_stake_table = compute_stake_table_commitment(
137                        &membership.stake_table().await,
138                        self.stake_table_capacity as usize,
139                    );
140                }
141
142                let signature = self.sign_new_state(&state, self.voting_stake_table).await;
143
144                if let Some(client) = &self.relay_server_client {
145                    let request_body = StateSignatureRequestBody {
146                        key: self.ver_key.clone(),
147                        state,
148                        next_stake: self.voting_stake_table,
149                        signature,
150                    };
151                    if let Err(error) = client
152                        .post::<()>("api/state")
153                        .body_binary(&request_body)
154                        .unwrap()
155                        .send()
156                        .await
157                    {
158                        tracing::warn!("Error posting signature to the relay server: {:?}", error);
159                    }
160                }
161            },
162            Err(err) => {
163                tracing::error!("Error generating light client state: {:?}", err)
164            },
165        }
166    }
167
168    /// Return a signature of a light client state at given height.
169    pub async fn get_state_signature(&self, height: u64) -> Option<StateSignatureRequestBody> {
170        let pool_guard = self.signatures.read().await;
171        pool_guard.get_signature(height)
172    }
173
174    /// Sign the light client state at given height and store it.
175    async fn sign_new_state(
176        &self,
177        state: &LightClientState,
178        next_stake_table: StakeTableState,
179    ) -> StateSignature {
180        let signature = <SchnorrPubKey as StateSignatureKey>::sign_state(
181            &self.sign_key,
182            state,
183            &next_stake_table,
184        )
185        .unwrap();
186        let mut pool_guard = self.signatures.write().await;
187        pool_guard.push(
188            state.block_height,
189            StateSignatureRequestBody {
190                key: self.ver_key.clone(),
191                state: *state,
192                next_stake: next_stake_table,
193                signature: signature.clone(),
194            },
195        );
196        tracing::debug!(
197            "New signature added for block height {}",
198            state.block_height
199        );
200        signature
201    }
202}

204/// A rolling in-memory storage for the most recent light client state signatures.
205#[derive(Debug, Default)]
206pub struct StateSignatureMemStorage {
207    pool: HashMap<u64, StateSignatureRequestBody>,
208    deque: VecDeque<u64>,
209}
210
211impl StateSignatureMemStorage {
212    pub fn push(&mut self, height: u64, signature: StateSignatureRequestBody) {
213        self.pool.insert(height, signature);
214        self.deque.push_back(height);
215        if self.pool.len() > SIGNATURE_STORAGE_CAPACITY {
216            self.pool.remove(&self.deque.pop_front().unwrap());
217        }
218    }
219
220    pub fn get_signature(&self, height: u64) -> Option<StateSignatureRequestBody> {
221        self.pool.get(&height).cloned()
222    }
