18/// A concrete type wrapper around `RequestResponse`. We need this so that we can implement
19/// local traits like `StateCatchup`. It also helps with readability.
20#[derive(Clone, Deref)]
21pub struct RequestResponseProtocol<
22    I: NodeImplementation<SeqTypes>,
23    V: Versions,
24    N: ConnectedNetwork<PubKey>,
25    P: SequencerPersistence,
26> {
27    #[deref]
28    #[allow(clippy::type_complexity)]
29    /// The actual inner request response protocol
30    inner: RequestResponse<
31        Sender,
32        Receiver<Bytes>,
33        Request,
34        RecipientSource<I, V>,
35        DataSource<I, V, N, P>,
36        PubKey,
37    >,
38
39    /// The configuration we used for the above inner protocol. This is nice to have for
40    /// estimating when we should make another request
41    config: RequestResponseConfig,
42
43    /// The public key of this node
44    public_key: PubKey,
45    /// The private key of this node
46    private_key: BLSPrivKey,
47}
48
49impl<
50        I: NodeImplementation<SeqTypes>,
51        V: Versions,
52        N: ConnectedNetwork<PubKey>,
53        P: SequencerPersistence,
54    > RequestResponseProtocol<I, V, N, P>
55{
56    /// Create a new RequestResponseProtocol from the inner
57    pub fn new(
58        // The configuration for the protocol
59        config: RequestResponseConfig,
60        // The network sender that [`RequestResponseProtocol`] will use to send messages
61        sender: Sender,
62        // The network receiver that [`RequestResponseProtocol`] will use to receive messages
63        receiver: Receiver<Bytes>,
64        // The recipient source that [`RequestResponseProtocol`] will use to get the recipients
65        // that a specific message should expect responses from
66        recipient_source: RecipientSource<I, V>,
67        // The [response] data source that [`RequestResponseProtocol`] will use to derive the
68        // response data for a specific request
69        data_source: DataSource<I, V, N, P>,
70        // The public key of this node
71        public_key: PubKey,
72        // The private key of this node
73        private_key: BLSPrivKey,
74    ) -> Self {
75        Self {
76            inner: RequestResponse::new(
77                config.clone(),
78                sender,
79                receiver,
80                recipient_source,
81                data_source,
82            ),
83            config,
84            public_key,
85            private_key,
86        }
87    }
88}
