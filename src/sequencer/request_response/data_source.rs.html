61/// Implement the trait that allows the [`RequestResponseProtocol`] to calculate/derive a response for a specific request
62#[async_trait]
63impl<
64        I: NodeImplementation<SeqTypes>,
65        V: Versions,
66        N: ConnectedNetwork<PubKey>,
67        P: SequencerPersistence,
68    > DataSourceTrait<Request> for DataSource<I, V, N, P>
69{
70    async fn derive_response_for(&self, request: &Request) -> Result<Response> {
71        match request {
72            Request::Accounts(height, view, accounts) => {
73                // Try to get accounts from memory first, then fall back to storage
74                if let Some(state) = self.consensus.state(ViewNumber::new(*view)).await {
75                    if let Ok(accounts) =
76                        retain_accounts(&state.fee_merkle_tree, accounts.iter().copied())
77                    {
78                        return Ok(Response::Accounts(accounts));
79                    }
80                }
81
82                // Fall back to storage
83                let (merkle_tree, leaf) = self
84                    .storage
85                    .as_ref()
86                    .with_context(|| "storage was not initialized")?
87                    .get_accounts(&self.node_state, *height, ViewNumber::new(*view), accounts)
88                    .await
89                    .with_context(|| "failed to get accounts from sql storage")?;
90
91                // If we successfully fetched accounts from storage, try to add them back into the in-memory
92                // state.
93                if let Err(err) = add_fee_accounts_to_state::<N, V, P>(
94                    &self.consensus.consensus(),
95                    &ViewNumber::new(*view),
96                    accounts,
97                    &merkle_tree,
98                    leaf,
99                )
100                .await
101                {
102                    tracing::warn!(?view, "Cannot update fetched account state: {err:#}");
103                }
104
105                Ok(Response::Accounts(merkle_tree))
106            },
107
108            Request::Leaf(height) => {
109                // Try to get the leaves from memory first, then fall back to storage
110                let mut leaves = self.consensus.consensus().read().await.undecided_leaves();
111                leaves.sort_by_key(|l| l.view_number());
112
113                if let Some((position, mut last_leaf)) =
114                    leaves.iter().find_position(|l| l.height() == *height)
115                {
116                    let mut leaf_chain = vec![last_leaf.clone()];
117                    for leaf in leaves.iter().skip(position + 1) {
118                        if leaf.justify_qc().view_number() == last_leaf.view_number() {
119                            leaf_chain.push(leaf.clone());
120                        } else {
121                            continue;
122                        }
123                        if leaf.view_number() == last_leaf.view_number() + 1 {
124                            // one away from decide
125                            last_leaf = leaf;
126                            break;
127                        }
128                        last_leaf = leaf;
129                    }
130
131                    // Make sure we got one more leaf to confirm the decide
132                    for leaf in leaves
133                        .iter()
134                        .skip_while(|l| l.view_number() <= last_leaf.view_number())
135                    {
136                        if leaf.justify_qc().view_number() == last_leaf.view_number() {
137                            leaf_chain.push(leaf.clone());
138                            return Ok(Response::Leaf(leaf_chain));
139                        }
140                    }
141                }
142
143                // Fall back to storage
144                let leaf_chain = self
145                    .storage
146                    .as_ref()
147                    .with_context(|| "storage was not initialized")?
148                    .get_leaf_chain(*height)
149                    .await
150                    .with_context(|| "failed to get leaf from sql storage")?;
151
152                Ok(Response::Leaf(leaf_chain))
153            },
154            Request::ChainConfig(commitment) => {
155                // Try to get the chain config from memory first, then fall back to storage
156                let chain_config_from_memory = self.consensus.decided_state().await.chain_config;
157                if chain_config_from_memory.commit() == *commitment {
158                    if let Some(chain_config) = chain_config_from_memory.resolve() {
159                        return Ok(Response::ChainConfig(chain_config));
160                    }
161                }
162
163                // Fall back to storage
164                Ok(Response::ChainConfig(
165                    self.storage
166                        .as_ref()
167                        .with_context(|| "storage was not initialized")?
168                        .get_chain_config(*commitment)
169                        .await
170                        .with_context(|| "failed to get chain config from sql storage")?,
171                ))
172            },
173            Request::BlocksFrontier(height, view) => {
174                // First try to respond from memory
175                let blocks_frontier_from_memory: Option<Result<BlocksFrontier>> = self
176                    .consensus
177                    .state(ViewNumber::new(*view))
178                    .await
179                    .map(|state| {
180                        let tree = &state.block_merkle_tree;
181                        let frontier = tree.lookup(tree.num_leaves() - 1).expect_ok()?.1;
182                        Ok(frontier)
183                    });
184
185                if let Some(Ok(blocks_frontier_from_memory)) = blocks_frontier_from_memory {
186                    return Ok(Response::BlocksFrontier(blocks_frontier_from_memory));
187                } else {
188                    // If we can't get the blocks frontier from memory, fall through to storage
189                    let blocks_frontier_from_storage = self
190                        .storage
191                        .as_ref()
192                        .with_context(|| "storage was not initialized")?
193                        .get_frontier(&self.node_state, *height, ViewNumber::new(*view))
194                        .await
195                        .with_context(|| "failed to get blocks frontier from sql storage")?;
196
197                    Ok(Response::BlocksFrontier(blocks_frontier_from_storage))
198                }
199            },
200            Request::RewardAccounts(height, view, accounts) => {
201                // Try to get the reward accounts from memory first, then fall back to storage
202                if let Some(state) = self.consensus.state(ViewNumber::new(*view)).await {
203                    if let Ok(reward_accounts) =
204                        retain_reward_accounts(&state.reward_merkle_tree, accounts.iter().copied())
205                    {
206                        return Ok(Response::RewardAccounts(reward_accounts));
207                    }
208                }
209
210                // Fall back to storage
211                let (merkle_tree, leaf) = self
212                    .storage
213                    .as_ref()
214                    .with_context(|| "storage was not initialized")?
215                    .get_reward_accounts(
216                        &self.node_state,
217                        *height,
218                        ViewNumber::new(*view),
219                        accounts,
220                    )
221                    .await
222                    .with_context(|| "failed to get accounts from sql storage")?;
223
224                // If we successfully fetched accounts from storage, try to add them back into the in-memory
225                // state.
226                if let Err(err) = add_reward_accounts_to_state::<N, V, P>(
227                    &self.consensus.consensus(),
228                    &ViewNumber::new(*view),
229                    accounts,
230                    &merkle_tree,
231                    leaf,
232                )
233                .await
234                {
235                    tracing::warn!(?view, "Cannot update fetched account state: {err:#}");
236                }
237
238                Ok(Response::RewardAccounts(merkle_tree))
239            },
240        }
241    }
242}

244/// Get a partial snapshot of the given reward state, which contains only the specified accounts.
245///
246/// Fails if one of the requested accounts is not represented in the original `state`.
247pub fn retain_reward_accounts(
248    state: &RewardMerkleTree,
249    accounts: impl IntoIterator<Item = RewardAccount>,
250) -> anyhow::Result<RewardMerkleTree> {
251    let mut snapshot = RewardMerkleTree::from_commitment(state.commitment());
252    for account in accounts {
253        match state.universal_lookup(account) {
254            LookupResult::Ok(elem, proof) => {
255                // This remember cannot fail, since we just constructed a valid proof, and are
256                // remembering into a tree with the same commitment.
257                snapshot.remember(account, *elem, proof).unwrap();
258            },
259            LookupResult::NotFound(proof) => {
260                // Likewise this cannot fail.
261                snapshot.non_membership_remember(account, proof).unwrap()
262            },
263            LookupResult::NotInMemory => {
264                bail!("missing account {account}");
265            },
266        }
267    }
268
269    Ok(snapshot)
