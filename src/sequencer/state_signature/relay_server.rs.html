34/// State that checks the light client state update and the signature collection
35pub struct StateRelayServerState {
36    /// Sequencer endpoint to query for stake table info
37    sequencer_url: Url,
38
39    /// Epoch length (fetched from HotShot config)
40    blocks_per_epoch: Option<u64>,
41    /// the first block where epoch 1 commence
42    epoch_start_block: Option<u64>,
43    /// Minimum weight to form an available state signature bundle, map: epoch_num -> threshold
44    thresholds: HashMap<u64, U256>,
45    /// Stake table: map: epoch_num -> map(vk, weight)
46    known_nodes: HashMap<u64, HashMap<StateVerKey, U256>>,
47    /// Signatures bundles for each block height
48    /// NOTE: nested hash-map because state signer could "vote/sign" different light client state for the same height
49    bundles: HashMap<u64, HashMap<LightClientState, StateSignaturesBundle>>,
50
51    /// The latest state signatures bundle whose total weight exceeds the threshold
52    latest_available_bundle: Option<StateSignaturesBundle>,
53    /// The block height of the latest available state signature bundle
54    latest_block_height: Option<u64>,
55
56    /// A ordered queue of block heights, used for garbage collection.
57    queue: BTreeSet<u64>,
58
59    /// shutdown signal
60    shutdown: Option<oneshot::Receiver<()>>,
61}

63impl StateRelayServerState {
64    /// Init the server state
65    pub fn new(sequencer_url: Url) -> Self {
66        Self {
67            sequencer_url,
68            blocks_per_epoch: None,
69            epoch_start_block: None,
70            thresholds: HashMap::new(),
71            known_nodes: HashMap::new(),
72            bundles: HashMap::new(),
73            latest_available_bundle: None,
74            latest_block_height: None,
75            queue: BTreeSet::new(),
76            shutdown: None,
77        }
78    }
79
80    /// after relay server started, when the first signature arrive, we query sequencer for the genesis and update local state.
81    /// The main reason we don't initialize at constructor (i.e. `Self::new()`) is due to cyclic dependency:
82    /// seq0 depends on relay server to be running to post light client signatures to;
83    /// relay server depends on seq0 to be running to query stake tables.
84    /// Thus, our strategy is to starts relay server with `None` and empty states and fill it only when needed.
85    ///
86    /// Another subtlety is our epoch doesn't starts from 1, because PoS will be activated at some block height,
87    /// thus `first_epoch` is not necessarily 1, but the `epoch_from_block_number(epoch_start_block, blocks_per_epoch)`.
88    async fn init_genesis(&mut self) -> anyhow::Result<()> {
89        // fetch genesis info from sequencer
90        if self.blocks_per_epoch.is_none() || self.epoch_start_block.is_none() {
91            let (blocks_per_epoch, epoch_start_block) =
92                fetch_epoch_config_from_sequencer(&self.sequencer_url).await?;
93            // set local state
94            self.blocks_per_epoch.get_or_insert(blocks_per_epoch);
95            self.epoch_start_block.get_or_insert(epoch_start_block);
96        }
97        let (blocks_per_epoch, epoch_start_block) = (
98            // both safe unwrap
99            self.blocks_per_epoch.unwrap(),
100            self.epoch_start_block.unwrap(),
101        );
102
103        let first_epoch = epoch_from_block_number(epoch_start_block, blocks_per_epoch);
104        tracing::info!(%blocks_per_epoch, %epoch_start_block, "Initializing genesis stake table with ");
105
106        let genesis_stake_table =
107            fetch_stake_table_from_sequencer(&self.sequencer_url, None).await?;
108        let genesis_total_stake = genesis_stake_table.total_stakes();
109
110        // init local state
111        self.thresholds
112            .insert(first_epoch, one_honest_threshold(genesis_total_stake));
113
114        let mut genesis_known_nodes = HashMap::<StateVerKey, U256>::new();
115        for entry in genesis_stake_table.0 {
116            genesis_known_nodes
117                .insert(entry.state_ver_key.clone(), entry.stake_table_entry.stake());
118        }
119
120        self.known_nodes.insert(first_epoch, genesis_known_nodes);
121
122        tracing::info!(%first_epoch, "Stake table synced ");
123        Ok(())
124    }
125
126    /// sync the stake table at `height` for the relayer server, fetching from the sequencer.
127    /// If the requested `height` is older than `latest_block_height`, then does nothing.
128    ///
129    /// NOTE: should not be publicly invocable, always in-sync with `self.queue` for easier garbage collection.
130    async fn sync_stake_table(&mut self, height: u64) -> anyhow::Result<()> {
131        let blocks_per_epoch = self.blocks_per_epoch.expect("forget to init genesis");
132        let epoch_start_block = self.epoch_start_block.expect("forget to init genesis");
133        let epoch = epoch_from_block_number(height, blocks_per_epoch);
134        let latest_epoch = epoch_from_block_number(
135            self.latest_block_height.unwrap_or_default(),
136            blocks_per_epoch,
137        );
138        if epoch <= latest_epoch {
139            tracing::debug!(
140                "Skipped stake table sync: requested epoch: {}, latest: {}",
141                epoch,
142                latest_epoch
143            );
144            return Ok(());
145        }
146        if self.known_nodes.contains_key(&epoch) {
147            tracing::debug!(%epoch, "Skipped stake table sync: already synced ");
148            return Ok(());
149        }
150
151        tracing::info!(%epoch,"Syncing stake table ");
152
153        let peer_configs = {
154            let client = surf_disco::Client::<ServerError, StaticVersion<0, 1>>::new(
155                self.sequencer_url.clone(),
156            );
157            if height >= epoch_start_block {
158                loop {
159                    match client
160                        .get::<Vec<PeerConfig<SeqTypes>>>(&format!("node/stake-table/{epoch}"))
161                        .send()
162                        .await
163                    {
164                        Ok(config) => break config,
165                        Err(e) => {
166                            tracing::error!("Failed to fetch stake table: {e}");
167                            sleep(Duration::from_secs(5)).await;
168                        },
169                    }
170                }
171            } else {
172                loop {
173                    match client
174                        .get::<PublicNetworkConfig>("config/hotshot")
175                        .send()
176                        .await
177                    {
178                        Ok(config) => break config.hotshot_config().known_nodes_with_stake(),
179                        Err(e) => {
180                            tracing::error!("Failed to fetch stake table: {e}");
181                            sleep(Duration::from_secs(5)).await;
182                        },
183                    }
184                }
185            }
186        };
187
188        // now update the local state for that epoch
189        let mut total_weights = U256::ZERO;
190        let mut new_nodes = HashMap::<StateVerKey, U256>::new();
191        for peer in peer_configs.iter() {
192            let weight = peer.stake_table_entry.stake_amount;
193            new_nodes.insert(peer.state_ver_key.clone(), weight);
194            total_weights += weight;
195        }
196        self.known_nodes.insert(epoch, new_nodes);
197        self.thresholds
198            .insert(epoch, one_honest_threshold(total_weights));
199
200        tracing::info!(%epoch, "Stake table synced ");
201        Ok(())
202    }
203
204    /// Centralizing all garbage-collection logic, won't panic, won't error, simply do nothing if nothing to prune.
205    /// `until_height` is inclusive, meaning that would also be pruned.
206    pub fn prune(&mut self, until_height: u64) {
207        let blocks_per_epoch = self.blocks_per_epoch.expect("forget to init genesis");
208        let oldest_epoch = if let Some(&height) = self.queue.first() {
209            epoch_from_block_number(height, blocks_per_epoch)
210        } else {
211            1
212        };
213
214        while let Some(&height) = self.queue.first() {
215            if height > until_height {
216                return;
217            }
218            self.bundles.remove(&height);
219            self.queue.pop_first();
220            tracing::debug!(%height, "garbage collected for ");
221        }
222
223        let newest_epoch = epoch_from_block_number(until_height + 1, blocks_per_epoch);
224        if newest_epoch > oldest_epoch {
225            for epoch in oldest_epoch..newest_epoch {
226                self.thresholds.remove(&epoch);
227                self.known_nodes.remove(&epoch);
228                tracing::debug!(%epoch, "garbage collected for ");
229            }
230        }
231    }
232
233    pub fn with_shutdown_signal(
234        mut self,
235        shutdown_listener: Option<oneshot::Receiver<()>>,
236    ) -> Self {
237        if self.shutdown.is_some() {
238            panic!("A shutdown signal is already registered and can not be registered twice");
239        }
240        self.shutdown = shutdown_listener;
241        self
242    }
243
244    pub fn with_blocks_per_epoch(mut self, blocks_per_epoch: u64) -> Self {
245        self.blocks_per_epoch = Some(blocks_per_epoch);
246        self
247    }
248    pub fn with_epoch_start_block(mut self, epoch_start_block: u64) -> Self {
249        self.epoch_start_block = Some(epoch_start_block);
250        self
251    }
252    pub fn with_thresholds(mut self, thresholds: HashMap<u64, U256>) -> Self {
253        self.thresholds = thresholds;
254        self
255    }
256    pub fn with_known_nodes(
257        mut self,
258        known_nodes: HashMap<u64, HashMap<StateVerKey, U256>>,
259    ) -> Self {
260        self.known_nodes = known_nodes;
261        self
262    }
263}

265#[async_trait::async_trait]
266pub trait StateRelayServerDataSource {
267    /// Get the latest available signatures bundle.
268    /// # Errors
269    /// Errors if there's no available signatures bundle.
270    fn get_latest_signature_bundle(&self) -> Result<StateSignaturesBundle, ServerError>;
271
272    /// Post a signature to the relay server
273    /// # Errors
274    /// Errors if the signature is invalid, already posted, or no longer needed.
275    async fn post_signature(&mut self, req: StateSignatureRequestBody) -> Result<(), ServerError>;
276}
277
278#[async_trait::async_trait]
279impl StateRelayServerDataSource for StateRelayServerState {
280    fn get_latest_signature_bundle(&self) -> Result<StateSignaturesBundle, ServerError> {
281        match &self.latest_available_bundle {
282            Some(bundle) => Ok(bundle.clone()),
283            None => Err(ServerError::catch_all(
284                StatusCode::NOT_FOUND,
285                "The light client state signatures are not ready.".to_owned(),
286            )),
287        }
288    }
289
290    async fn post_signature(&mut self, req: StateSignatureRequestBody) -> Result<(), ServerError> {
291        let block_height = req.state.block_height;
292        if block_height <= self.latest_block_height.unwrap_or(0) {
293            // This signature is no longer needed
294            return Ok(());
295        }
296
297        let blocks_per_epoch = match self.blocks_per_epoch {
298            Some(v) => v,
299            None => {
300                self.init_genesis().await.map_err(|e| {
301                    ServerError::catch_all(StatusCode::INTERNAL_SERVER_ERROR, format!("{e}"))
302                })?;
303                self.blocks_per_epoch
304                    .expect("internal err, init_genesis() wrong")
305            },
306        };
307        let epoch = epoch_from_block_number(block_height, blocks_per_epoch);
308
309        // if epoch is activated and block_height is greater than epoch root, ignore and drop
310        if block_height > self.epoch_start_block.expect("init_genesis wrong")
311            && is_gt_epoch_root(block_height, blocks_per_epoch)
312        {
313            return Ok(());
314        }
315
316        if !self.known_nodes.contains_key(&epoch) {
317            self.sync_stake_table(block_height).await.map_err(|e| {
318                ServerError::catch_all(StatusCode::INTERNAL_SERVER_ERROR, format!("{e}"))
319            })?;
320        }
321
322        // retrieve the signer/sender's weight from the correct stake table for that epoch
323        let Some(nodes) = self.known_nodes.get(&epoch) else {
324            return Err(ServerError::catch_all(
325                StatusCode::INTERNAL_SERVER_ERROR,
326                "Stake table not found".to_owned(),
327            ));
328        };
329        let Some(threshold) = self.thresholds.get(&epoch) else {
330            return Err(ServerError::catch_all(
331                StatusCode::INTERNAL_SERVER_ERROR,
332                "Threshold not found".to_owned(),
333            ));
334        };
335        let Some(weight) = nodes.get(&req.key) else {
336            return Err(ServerError::catch_all(
337                StatusCode::UNAUTHORIZED,
338                "Signature posted by nodes not on the stake table".to_owned(),
339            ));
340        };
341
342        // sanity check the signature validity first before adding in
343        if !req
344            .key
345            .verify_state_sig(&req.signature, &req.state, &req.next_stake)
346        {
347            tracing::info!("Received invalid request: {:?}", req);
348            return Err(ServerError::catch_all(
349                StatusCode::BAD_REQUEST,
350                "The posted signature is not valid.".to_owned(),
351            ));
352        }
353
354        let bundles_at_height = self.bundles.entry(block_height).or_default();
355        self.queue.insert(block_height);
356
357        let bundle = bundles_at_height
358            .entry(req.state)
359            .or_insert(StateSignaturesBundle {
360                state: req.state,
361                next_stake: req.next_stake,
362                signatures: Default::default(),
363                accumulated_weight: U256::from(0),
364            });
365        tracing::debug!(
366            "Accepting new signature for block height {} from {}.",
367            block_height,
368            req.key
369        );
370        match bundle.signatures.entry(req.key) {
371            Entry::Occupied(_) => {
372                // A signature is already posted for this key with this state
373                return Err(ServerError::catch_all(
374                    StatusCode::BAD_REQUEST,
375                    "A signature of this light client state is already posted at this block height for this key.".to_owned(),
376                ));
377            },
378            Entry::Vacant(entry) => {
379                entry.insert(req.signature);
380                bundle.accumulated_weight += *weight;
381            },
382        }
383
384        if bundle.accumulated_weight >= *threshold {
385            tracing::info!(
386                "State signature bundle at block height {} is ready to serve.",
387                block_height
388            );
389            self.latest_block_height = Some(block_height);
390            self.latest_available_bundle = Some(bundle.clone());
391
392            // garbage collect
393            self.prune(block_height);
394        }
395
396        Ok(())
397    }
398}

400/// configurability options for the web server
401#[derive(Args, Default)]
402pub struct Options {
403    #[arg(
404        long = "state-relay-server-api-path",
405        env = "STATE_RELAY_SERVER_API_PATH"
406    )]
407    /// path to API
408    pub api_path: Option<PathBuf>,
409}
410
411/// Set up APIs for relay server
412fn define_api<State, ApiVer: StaticVersionType + 'static>(
413    options: &Options,
414    _: ApiVer,
415) -> Result<Api<State, ServerError, ApiVer>, ApiError>
416where
417    State: 'static + Send + Sync + ReadState + WriteState,
418    <State as ReadState>::State: Send + Sync + StateRelayServerDataSource,
419{
420    let mut api = match &options.api_path {
421        Some(path) => Api::<State, ServerError, ApiVer>::from_file(path)?,
422        None => {
423            let toml: toml::Value = toml::from_str(include_str!(
424                "../../api/state_relay_server.toml"
425            ))
426            .map_err(|err| ApiError::CannotReadToml {
427                reason: err.to_string(),
428            })?;
429            Api::<State, ServerError, ApiVer>::new(toml)?
430        },
431    };
432
433    api.get("getlateststate", |_req, state| {
434        async move { state.get_latest_signature_bundle() }.boxed()
435    })?
436    .post("poststatesignature", move |req, state| {
437        async move {
438            let body = req
439                .body_auto::<StateSignatureRequestBody, ApiVer>(ApiVer::instance())
440                .map_err(ServerError::from_request_error)?;
441            state.post_signature(body).await?;
442            Ok(())
443        }
444        .boxed()
445    })?;
446
447    Ok(api)
448}

450pub async fn run_relay_server<ApiVer: StaticVersionType + 'static>(
451    shutdown_listener: Option<oneshot::Receiver<()>>,
452    sequencer_url: Url,
453    url: Url,
454    bind_version: ApiVer,
455) -> anyhow::Result<()> {
456    let options = Options::default();
457    let api = define_api(&options, bind_version).unwrap();
458
459    let state = RwLock::new(
460        StateRelayServerState::new(sequencer_url).with_shutdown_signal(shutdown_listener),
461    );
462    let mut app = App::<RwLock<StateRelayServerState>, ServerError>::with_state(state);
463
464    app.register_module("api", api).unwrap();
465
466    let app_future = app.serve(url.clone(), bind_version);
467    app_future.await?;
468
469    tracing::info!(%url, "Relay server starts serving at ");
470
471    Ok(())
472}
