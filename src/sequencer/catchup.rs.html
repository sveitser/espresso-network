85/// A score of a catchup peer, based on our interactions with that peer.
86///
87/// The score accounts for malicious peers -- i.e. peers that gave us an invalid response to a
88/// verifiable request -- and faulty/unreliable peers -- those that fail to respond to requests at
89/// all. The score has a comparison function where higher is better, or in other words `p1 > p2`
90/// means we believe we are more likely to successfully catch up using `p1` than `p2`. This makes it
91/// convenient and efficient to collect peers in a priority queue which we can easily convert to a
92/// list sorted by reliability.
93#[derive(Clone, Copy, Debug, Default)]
94struct PeerScore {
95    requests: usize,
96    failures: usize,
97}
98
99impl Ord for PeerScore {
100    fn cmp(&self, other: &Self) -> Ordering {
101        // Compare failure rates: `self` is better than `other` if
102        //      self.failures / self.requests < other.failures / other.requests
103        // or equivalently
104        //      other.failures * self.requests > self.failures * other.requests
105        (other.failures * self.requests).cmp(&(self.failures * other.requests))
106    }
107}
108
109impl PartialOrd for PeerScore {
110    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
111        Some(self.cmp(other))
112    }
113}
114
115impl PartialEq for PeerScore {
116    fn eq(&self, other: &Self) -> bool {
117        self.cmp(other).is_eq()
118    }
119}

124pub struct StatePeers<ApiVer: StaticVersionType> {
125    // Peer IDs, ordered by reliability score. Each ID is an index into `clients`.
126    scores: Arc<RwLock<PriorityQueue<usize, PeerScore>>>,
127    clients: Vec<Client<ServerError, ApiVer>>,
128    backoff: BackoffParams,

131impl<ApiVer: StaticVersionType> StatePeers<ApiVer> {
132    async fn fetch<Fut>(
133        &self,
134        retry: usize,
135        f: impl Fn(Client<ServerError, ApiVer>) -> Fut,
136    ) -> anyhow::Result<Fut::Ok>
137    where
138        Fut: TryFuture<Error: Display>,
139    {
140        // Since we have generally have multiple peers we can catch up from, we want a fairly
141        // aggressive timeout for requests: if a peer is not responding quickly, we're better off
142        // just trying the next one rather than waiting, and this prevents a malicious peer from
143        // delaying catchup for a long time.
144        //
145        // However, if we set the timeout _too_ aggressively, we might fail to catch up even from an
146        // honest peer, and thus never make progress. Thus, we start with a timeout of 500ms, which
147        // is aggressive but still very reasonable for an HTTP request. If that fails with all of
148        // our peers, we increase the timeout by 1 second for each successive retry, until we
149        // eventually succeed.
150        let timeout_dur = Duration::from_millis(500) * (retry as u32 + 1);
151
152        // Keep track of which peers we make requests to and which succeed (`true`) or fail (`false`),
153        // so we can update reliability scores at the end.
154        let mut requests = HashMap::new();
155        let mut res = Err(anyhow!("failed fetching from every peer"));
156
157        // Try each peer in order of reliability score, until we succeed. We clone out of
158        // `self.scores` because it is small (contains only numeric IDs and scores), so this clone
159        // is a lot cheaper than holding the read lock the entire time we are making requests (which
160        // could be a while).
161        let mut scores = { (*self.scores.read().await).clone() };
162        while let Some((id, score)) = scores.pop() {
163            let client = &self.clients[id];
164            tracing::info!("fetching from {}", client.url);
165            match timeout(timeout_dur, f(client.clone()).into_future()).await {
166                Ok(Ok(t)) => {
167                    requests.insert(id, true);
168                    res = Ok(t);
169                    break;
170                },
171                Ok(Err(err)) => {
172                    tracing::warn!(id, ?score, peer = %client.url, "error from peer: {err:#}");
173                    requests.insert(id, false);
174                },
175                Err(_) => {
176                    tracing::warn!(id, ?score, peer = %client.url, ?timeout_dur, "request timed out");
177                    requests.insert(id, false);
178                },
179            }
180        }
181
182        // Update client scores.
183        let mut scores = self.scores.write().await;
184        for (id, success) in requests {
185            scores.change_priority_by(&id, |score| {
186                score.requests += 1;
187                self.clients[id].requests.add(1);
188                if !success {
189                    score.failures += 1;
190                    self.clients[id].failures.add(1);
191                }
192            });
193        }
194
195        res
196    }
197
198    pub fn from_urls(
199        urls: Vec<Url>,
200        backoff: BackoffParams,
201        metrics: &(impl Metrics + ?Sized),
202    ) -> Self {
203        if urls.is_empty() {
204            panic!("Cannot create StatePeers with no peers");
205        }
206
207        let metrics = metrics.subgroup("catchup".into());
208        let requests = metrics.counter_family("requests".into(), vec!["peer".into()]);
209        let failures = metrics.counter_family("request_failures".into(), vec!["peer".into()]);
210
211        let scores = urls
212            .iter()
213            .enumerate()
214            .map(|(i, _)| (i, PeerScore::default()))
215            .collect();
216        let clients = urls
217            .into_iter()
218            .map(|url| Client::new(url, &*requests, &*failures))
219            .collect();
220
221        Self {
222            clients,
223            scores: Arc::new(RwLock::new(scores)),
224            backoff,
225        }
226    }

246            })

404    /// decided view.
405    ///

421            bail!("merklized state catchup is not supported for this data source");

671    /// [`ChainConfig`]s, which it can feed to the STF during evaluation.
672    pub fn add_chain_config(&mut self, cf: ChainConfig) {
673        self.chain_configs.insert(cf.commit(), cf);

741/// A catchup implementation that parallelizes requests to many providers.
742/// It returns the result of the first non-erroring provider to complete.
743#[derive(Clone)]
744pub struct ParallelStateCatchup {
745    providers: Arc<Mutex<Vec<Arc<dyn StateCatchup>>>>,
746    remote_request_delay: Duration,
747}
748
749impl ParallelStateCatchup {
750    /// Create a new [`ParallelStateCatchup`] with two providers.
751    ///
752    /// `remote_request_delay` is the amount of time to wait for a local response (e.g. database)
753    /// before spawning remote requests for the same data.
754    pub fn new(providers: &[Arc<dyn StateCatchup>], remote_request_delay: Duration) -> Self {
755        Self {
756            providers: Arc::new(Mutex::new(providers.to_vec())),
757            remote_request_delay,
758        }
759    }
760
761    /// Add a provider to the list of providers
762    pub fn add_provider(&self, provider: Arc<dyn StateCatchup>) {
763        self.providers.lock().push(provider);
764    }
765
766    /// Perform an async operation on all providers, returning the first result to succeed
767    pub async fn on_all_providers<C, F, RT>(&self, closure: C) -> anyhow::Result<RT>
768    where
769        C: Fn(Arc<dyn StateCatchup>) -> F + Clone + Send + Sync + 'static,
770        F: Future<Output = anyhow::Result<RT>> + Send + 'static,
771        RT: Send + Sync + 'static,
772    {
773        // Make sure we have at least one provider
774        let providers = self.providers.lock().clone();
775
776        if providers.is_empty() {
777            return Err(anyhow::anyhow!("no providers were initialized"));
778        }
779
780        // Get the remote request delay so we can throw it into the future
781        let remote_request_delay = self.remote_request_delay;
782
783        // Spawn futures for each provider
784        let mut futures = FuturesUnordered::new();
785        for provider in providers {
786            if provider.is_local() {
787                // If this is a local provider, spawn the future directly
788                let closure = closure.clone();
789                futures.push(AbortOnDropHandle::new(tokio::spawn(closure(provider))));
790            } else {
791                // If this is a remote provider, spawn a future that sleeps for the remote request delay
792                // before calling the closure. These will automatically be cancelled when this function returns
793                let closure = closure.clone();
794                futures.push(AbortOnDropHandle::new(tokio::spawn(async move {
795                    tokio::time::sleep(remote_request_delay).await;
796                    closure(provider).await
797                })));
798            }
799        }
800
801        // Return the first successful result
802        while let Some(result) = futures.next().await {
803            // Unwrap the inner (join) result
804            let result = match result {
805                Ok(res) => res,
806                Err(err) => {
807                    warn!("Failed to join on provider: {err:#}. Trying next provider...");
808                    continue;
809                },
810            };
811
812            // If a provider fails, print why
813            let result = match result {
814                Ok(res) => res,
815                Err(err) => {
816                    warn!("Failed to fetch data: {err:#}. Trying next provider...");
817                    continue;
818                },
819            };
820
821            return Ok(result);
822        }
823
824        Err(anyhow::anyhow!("no providers returned a successful result"))
825    }
826}

1017/// Add accounts to the in-memory consensus state.
1018/// We use this during catchup after receiving verified accounts.
1019#[allow(clippy::type_complexity)]
1020pub async fn add_fee_accounts_to_state<
1021    N: ConnectedNetwork<PubKey>,
1022    V: Versions,
1023    P: SequencerPersistence,
1024>(
1025    consensus: &Arc<RwLock<Consensus<SeqTypes>>>,
1026    view: &<SeqTypes as NodeType>::View,
1027    accounts: &[FeeAccount],
1028    tree: &FeeMerkleTree,
1029    leaf: Leaf2,
1030) -> anyhow::Result<()> {
1031    // Get the consensus handle
1032    let mut consensus = consensus.write().await;
1033
1034    let (state, delta) = match consensus.validated_state_map().get(view) {
1035        Some(View {
1036            view_inner: ViewInner::Leaf { state, delta, .. },
1037        }) => {
1038            let mut state = (**state).clone();
1039
1040            // Add the fetched accounts to the state.
1041            for account in accounts {
1042                if let Some((proof, _)) = FeeAccountProof::prove(tree, (*account).into()) {
1043                    if let Err(err) = proof.remember(&mut state.fee_merkle_tree) {
1044                        tracing::warn!(
1045                            ?view,
1046                            %account,
1047                            "cannot update fetched account state: {err:#}"
1048                        );
1049                    }
1050                } else {
1051                    tracing::warn!(?view, %account, "cannot update fetched account state because account is not in the merkle tree");
1052                };
1053            }
1054
1055            (Arc::new(state), delta.clone())
1056        },
1057        _ => {
1058            // If we don't already have a leaf for this view, or if we don't have the view
1059            // at all, we can create a new view based on the recovered leaf and add it to
1060            // our state map. In this case, we must also add the leaf to the saved leaves
1061            // map to ensure consistency.
1062            let mut state = ValidatedState::from_header(leaf.block_header());
1063            state.fee_merkle_tree = tree.clone();
1064            (Arc::new(state), None)
1065        },
1066    };
1067
1068    consensus
1069        .update_leaf(leaf, Arc::clone(&state), delta)
1070        .with_context(|| "failed to update leaf")?;
1071
1072    Ok(())
1073}
