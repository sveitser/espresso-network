16/// Deploy contracts needed to run the sequencer.
17///
18/// This script deploys contracts needed to run the sequencer to an L1. It outputs a .env file
19/// containing the addresses of the deployed contracts.
20///
21/// This script can also be used to do incremental deployments. The only contract addresses
22/// needed to configure the sequencer network are ESPRESSO_SEQUENCER_FEE_CONTRACT_PROXY_ADDRESS,
23/// ESPRESSO_SEQUENCER_LIGHT_CLIENT_PROXY_ADDRESS and ESPRESSO_SEQUENCER_STAKE_TABLE_ADDRESS.
24/// These contracts, however, have dependencies, and a full deployment involves several
25/// contracts. Some of these contracts, especially libraries may already have been deployed, or
26/// perhaps one of the top-level contracts has been deployed and we only need to deploy the other
27/// one.
28///
29/// It is possible to pass in the addresses of already deployed contracts, in which case those
30/// addresses will be used in place of deploying a new contract wherever that contract is required
31/// in the deployment process. The generated .env file will include all the addresses passed in as
32/// well as those newly deployed.
33#[derive(Clone, Debug, Parser)]
34struct Options {
35    /// A JSON-RPC endpoint for the L1 to deploy to.
36    #[clap(
37        short,
38        long,
39        env = "ESPRESSO_SEQUENCER_L1_PROVIDER",
40        default_value = "http://localhost:8545"
41    )]
42    rpc_url: Url,
43
44    /// Request rate when polling L1.
45    #[clap(
46        long,
47        env = "ESPRESSO_SEQUENCER_L1_POLLING_INTERVAL",
48        default_value = "7s",
49        value_parser = parse_duration,
50    )]
51    pub l1_polling_interval: Duration,
52
53    /// URL of a sequencer node that is currently providing the HotShot config.
54    /// This is used to initialize the stake table.
55    #[clap(
56        long,
57        env = "ESPRESSO_SEQUENCER_URL",
58        default_value = "http://localhost:24000"
59    )]
60    pub sequencer_url: Url,
61
62    /// Mnemonic for an L1 wallet.
63    ///
64    /// This wallet is used to deploy the contracts, so the account indicated by ACCOUNT_INDEX must
65    /// be funded with with ETH.
66    #[clap(
67        long,
68        name = "MNEMONIC",
69        env = "ESPRESSO_SEQUENCER_ETH_MNEMONIC",
70        default_value = "test test test test test test test test test test test junk"
71    )]
72    mnemonic: String,
73
74    /// Address for the multisig wallet that will be the admin
75    ///
76    /// If provided, this the multisig wallet that will be able to upgrade contracts and execute
77    /// admin only functions on contracts. If not provided, admin power for all contracts will be
78    /// held by the account used to deploy the contracts (determined from MNEMONIC, ACCOUNT_INDEX).
79    #[clap(
80        long,
81        name = "MULTISIG_ADDRESS",
82        env = "ESPRESSO_SEQUENCER_ETH_MULTISIG_ADDRESS"
83    )]
84    multisig_address: Option<Address>,
85
86    /// Account index in the L1 wallet generated by MNEMONIC to use when deploying the contracts.
87    #[clap(
88        long,
89        name = "ACCOUNT_INDEX",
90        env = "ESPRESSO_DEPLOYER_ACCOUNT_INDEX",
91        default_value = "0"
92    )]
93    account_index: u32,
94
95    /// Option to deploy fee contracts
96    #[clap(long, default_value = "false")]
97    deploy_fee: bool,
98    /// Option to deploy LightClient V1 and proxy
99    #[clap(long, default_value = "false")]
100    deploy_light_client_v1: bool,
101    /// Option to upgrade to LightClient V2
102    #[clap(long, default_value = "false")]
103    upgrade_light_client_v2: bool,
104    #[clap(long, default_value = "false")]
105    deploy_esp_token: bool,
106    #[clap(long, default_value = "false")]
107    deploy_stake_table: bool,
108
109    /// Write deployment results to OUT as a .env file.
110    ///
111    /// If not provided, the results will be written to stdout.
112    #[clap(short, long, name = "OUT", env = "ESPRESSO_DEPLOYER_OUT_PATH")]
113    out: Option<PathBuf>,
114
115    #[clap(flatten)]
116    contracts: DeployedContracts,
117
118    /// If toggled, launch a mock LightClient contract with a smaller verification key for testing.
119    /// Applies to both V1 and V2 of LightClient.
120    #[clap(short, long)]
121    pub use_mock: bool,
122
123    /// Stake table capacity for the prover circuit
124    #[clap(short, long, env = "ESPRESSO_SEQUENCER_STAKE_TABLE_CAPACITY", default_value_t = STAKE_TABLE_CAPACITY)]
125    pub stake_table_capacity: usize,
126    ///
127    /// If the light client contract is being deployed and this is set, the prover will be
128    /// permissioned so that only this address can update the light client state. Otherwise, proving
129    /// will be permissionless.
130    ///
131    /// If the light client contract is not being deployed, this option is ignored.
132    #[clap(long, env = "ESPRESSO_SEQUENCER_PERMISSIONED_PROVER")]
133    permissioned_prover: Option<Address>,
134
135    /// Exit escrow period for the stake table contract.
136    ///
137    /// This is the period for which stake table contract will retain funds after withdrawals have
138    /// been requested. It should be set to a value that is at least 3 hotshot epochs plus ample
139    /// time to allow for submission of slashing evidence. Initially it will probably be around one
140    /// week.
141    #[clap(long, env = "ESPRESSO_SEQUENCER_STAKE_TABLE_EXIT_ESCROW_PERIOD", value_parser = parse_duration)]
142    exit_escrow_period: Option<Duration>,
143
144    /// The address that the tokens will be minted to.
145    ///
146    /// If unset the tokens will be minted to the deployer account.
147    #[clap(long, env = "ESP_TOKEN_INITIAL_GRANT_RECIPIENT_ADDRESS")]
148    initial_token_grant_recipient: Option<Address>,
149
150    #[clap(flatten)]
151    logging: logging::Config,
152}

154#[tokio::main]
155async fn main() -> anyhow::Result<()> {
156    let opt = Options::parse();
157    opt.logging.init();
158
159    let mut contracts = Contracts::from(opt.contracts);
160
161    let provider = build_provider(opt.mnemonic, opt.account_index, opt.rpc_url);
162
163    // First use builder to build constructor input arguments
164    let mut args_builder = DeployerArgsBuilder::default();
165    args_builder
166        .deployer(provider)
167        .mock_light_client(opt.use_mock);
168    if let Some(multisig) = opt.multisig_address {
169        args_builder.multisig(multisig);
170    }
171    if let Some(token_recipient) = opt.initial_token_grant_recipient {
172        args_builder.token_recipient(token_recipient);
173    }
174
175    if opt.deploy_light_client_v1 {
176        let (genesis_state, genesis_stake) =
177            light_client_genesis(&opt.sequencer_url, opt.stake_table_capacity).await?;
178        args_builder
179            .genesis_lc_state(genesis_state)
180            .genesis_st_state(genesis_stake);
181        if let Some(prover) = opt.permissioned_prover {
182            args_builder.permissioned_prover(prover);
183        }
184    }
185    if opt.upgrade_light_client_v2 {
186        // fetch epoch length from HotShot config
187        // Request the configuration until it is successful
188        let (blocks_per_epoch, epoch_start_block) = loop {
189            match surf_disco::Client::<ServerError, StaticVersion<0, 1>>::new(
190                opt.sequencer_url.clone(),
191            )
192            .get::<PublicNetworkConfig>("config/hotshot")
193            .send()
194            .await
195            {
196                Ok(resp) => {
197                    let config = resp.hotshot_config();
198                    break (config.blocks_per_epoch(), config.epoch_start_block());
199                },
200                Err(e) => {
201                    tracing::error!("Failed to fetch the network config: {e}");
202                    sleep(Duration::from_secs(5));
203                },
204            }
205        };
206        args_builder
207            .blocks_per_epoch(blocks_per_epoch)
208            .epoch_start_block(epoch_start_block);
209    }
210    if opt.deploy_stake_table {
211        if let Some(escrow_period) = opt.exit_escrow_period {
212            args_builder.exit_escrow_period(U256::from(escrow_period.as_secs()));
213        }
214    }
215
216    // then deploy specified contracts
217    let args = args_builder.build()?;
218    if opt.deploy_fee {
219        args.deploy(&mut contracts, Contract::FeeContractProxy)
220            .await?;
221    }
222    if opt.deploy_esp_token {
223        args.deploy(&mut contracts, Contract::EspTokenProxy).await?;
224    }
225    if opt.deploy_light_client_v1 {
226        args.deploy(&mut contracts, Contract::LightClientProxy)
227            .await?;
228    }
229    if opt.upgrade_light_client_v2 {
230        args.deploy(&mut contracts, Contract::LightClientV2).await?;
231    }
232    if opt.deploy_stake_table {
233        args.deploy(&mut contracts, Contract::StakeTableProxy)
234            .await?;
235    }
236
237    // finally print out or persist deployed addresses
238    if let Some(out) = &opt.out {
239        let file = File::options()
240            .create(true)
241            .truncate(true)
242            .write(true)
243            .open(out)?;
244        contracts.write(file)?;
245    } else {
246        contracts.write(stdout())?;
247    }
248
249    Ok(())
250}
