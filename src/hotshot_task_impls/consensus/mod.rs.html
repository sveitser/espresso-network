45/// Task state for the Consensus task.
46pub struct ConsensusTaskState<TYPES: NodeType, I: NodeImplementation<TYPES>, V: Versions> {
47    pub first_epoch: Option<(TYPES::View, TYPES::Epoch)>,
48    /// Our public key
49    pub public_key: TYPES::SignatureKey,
50
51    /// Our Private Key
52    pub private_key: <TYPES::SignatureKey as SignatureKey>::PrivateKey,
53
54    /// Immutable instance state
55    pub instance_state: Arc<TYPES::InstanceState>,
56
57    /// The underlying network
58    pub network: Arc<I::Network>,
59
60    /// Membership for Quorum Certs/votes
61    pub membership_coordinator: EpochMembershipCoordinator<TYPES>,
62
63    /// A map of `QuorumVote` collector tasks.
64    pub vote_collectors: VoteCollectorsMap<TYPES, QuorumVote2<TYPES>, QuorumCertificate2<TYPES>, V>,
65
66    /// A map of `EpochRootQuorumVote` collector tasks.
67    pub epoch_root_vote_collectors: EpochRootVoteCollectorsMap<TYPES, V>,
68
69    /// A map of `QuorumVote` collector tasks. They collect votes from the nodes in the next epoch.
70    pub next_epoch_vote_collectors: VoteCollectorsMap<
71        TYPES,
72        NextEpochQuorumVote2<TYPES>,
73        NextEpochQuorumCertificate2<TYPES>,
74        V,
75    >,
76
77    /// A map of `TimeoutVote` collector tasks.
78    pub timeout_vote_collectors:
79        VoteCollectorsMap<TYPES, TimeoutVote2<TYPES>, TimeoutCertificate2<TYPES>, V>,
80
81    /// The view number that this node is currently executing in.
82    pub cur_view: TYPES::View,
83
84    /// Timestamp this view starts at.
85    pub cur_view_time: i64,
86
87    /// The epoch number that this node is currently executing in.
88    pub cur_epoch: Option<TYPES::Epoch>,
89
90    /// Output events to application
91    pub output_event_stream: async_broadcast::Sender<Event<TYPES>>,
92
93    /// Timeout task handle
94    pub timeout_task: JoinHandle<()>,
95
96    /// View timeout from config.
97    pub timeout: u64,
98
99    /// A reference to the metrics trait.
100    pub consensus: OuterConsensus<TYPES>,
101
102    /// A reference to the storage trait.
103    pub storage: Arc<RwLock<I::Storage>>,
104
105    /// The node's id
106    pub id: u64,
107
108    /// Lock for a decided upgrade
109    pub upgrade_lock: UpgradeLock<TYPES, V>,
110
111    /// Number of blocks in an epoch, zero means there are no epochs
112    pub epoch_height: u64,
113
114    /// The time this view started
115    pub view_start_time: Instant,
116}

118impl<TYPES: NodeType, I: NodeImplementation<TYPES>, V: Versions> ConsensusTaskState<TYPES, I, V> {
119    /// Handles a consensus event received on the event stream
120    #[instrument(skip_all, fields(id = self.id, cur_view = *self.cur_view, cur_epoch = self.cur_epoch.map(|x| *x)), name = "Consensus replica task", level = "error", target = "ConsensusTaskState")]
121    pub async fn handle(
122        &mut self,
123        event: Arc<HotShotEvent<TYPES>>,
124        sender: Sender<Arc<HotShotEvent<TYPES>>>,
125        receiver: Receiver<Arc<HotShotEvent<TYPES>>>,
126    ) -> Result<()> {
127        match event.as_ref() {
128            HotShotEvent::QuorumVoteRecv(ref vote) => {
129                if let Err(e) =
130                    handle_quorum_vote_recv(vote, Arc::clone(&event), &sender, self).await
131                {
132                    tracing::debug!("Failed to handle QuorumVoteRecv event; error = {e}");
133                }
134            },
135            HotShotEvent::EpochRootQuorumVoteRecv(ref vote) => {
136                if let Err(e) =
137                    handle_epoch_root_quorum_vote_recv(vote, Arc::clone(&event), &sender, self)
138                        .await
139                {
140                    tracing::debug!("Failed to handle EpochRootQuorumVoteRecv event; error = {e}");
141                }
142            },
143            HotShotEvent::TimeoutVoteRecv(ref vote) => {
144                if let Err(e) =
145                    handle_timeout_vote_recv(vote, Arc::clone(&event), &sender, self).await
146                {
147                    tracing::debug!("Failed to handle TimeoutVoteRecv event; error = {e}");
148                }
149            },
150            HotShotEvent::SetFirstEpoch(view, epoch) => {
151                self.first_epoch = Some((*view, *epoch));
152            },
153            HotShotEvent::ViewChange(new_view_number, epoch_number) => {
154                if let Some((view, epoch)) = self.first_epoch {
155                    if *new_view_number == view && *epoch_number != Some(epoch) {
156                        broadcast_event(
157                            Arc::new(HotShotEvent::ViewChange(*new_view_number, Some(epoch))),
158                            &sender,
159                        )
160                        .await;
161                    }
162                }
163                if let Err(e) =
164                    handle_view_change(*new_view_number, *epoch_number, &sender, &receiver, self)
165                        .await
166                {
167                    tracing::trace!("Failed to handle ViewChange event; error = {e}");
168                }
169                self.view_start_time = Instant::now();
170            },
171            HotShotEvent::Timeout(view_number, epoch) => {
172                if let Err(e) = handle_timeout(*view_number, *epoch, &sender, self).await {
173                    tracing::debug!("Failed to handle Timeout event; error = {e}");
174                }
175            },
176            HotShotEvent::ExtendedQc2Formed(eqc) => {
177                let cert_view = eqc.view_number();
178                let Some(cert_block_number) = eqc.data.block_number else {
179                    tracing::error!("Received extended QC but no block number");
180                    return Ok(());
181                };
182                let cert_epoch = epoch_from_block_number(cert_block_number, self.epoch_height);
183                tracing::error!(
184                    "Formed Extended QC for view {cert_view:?} and epoch {cert_epoch:?}."
185                );
186                // Transition to the new epoch by sending ViewChange
187                let next_epoch = TYPES::Epoch::new(cert_epoch + 1);
188                broadcast_event(
189                    Arc::new(HotShotEvent::ViewChange(cert_view + 1, Some(next_epoch))),
190                    &sender,
191                )
192                .await;
193                tracing::info!("Entering new epoch: {:?}", next_epoch);
194                tracing::info!(
195                    "Stake table for epoch {:?}:\n\n{:?}",
196                    next_epoch,
197                    self.membership_coordinator
198                        .stake_table_for_epoch(Some(next_epoch))
199                        .await?
200                        .stake_table()
201                        .await
202                );
203                tracing::info!(
204                    "Stake table for epoch {:?}:\n\n{:?}",
205                    next_epoch + 1,
206                    self.membership_coordinator
207                        .stake_table_for_epoch(Some(next_epoch + 1))
208                        .await?
209                        .stake_table()
210                        .await
211                );
212            },
213            HotShotEvent::ExtendedQcRecv(high_qc, next_epoch_high_qc, _) => {
214                if !high_qc
215                    .data
216                    .block_number
217                    .is_some_and(|bn| is_last_block(bn, self.epoch_height))
218                {
219                    tracing::warn!("Received extended QC but we can't verify the leaf is extended");
220                    return Ok(());
221                }
222                if let Err(e) = validate_qc_and_next_epoch_qc(
223                    high_qc,
224                    Some(next_epoch_high_qc),
225                    &self.consensus,
226                    &self.membership_coordinator,
227                    &self.upgrade_lock,
228                    self.epoch_height,
229                )
230                .await
231                {
232                    tracing::error!("Received invalid extended QC: {e}");
233                    return Ok(());
234                }
235
236                let mut consensus_writer = self.consensus.write().await;
237                let high_qc_updated = consensus_writer.update_high_qc(high_qc.clone()).is_ok();
238                let next_high_qc_updated = consensus_writer
239                    .update_next_epoch_high_qc(next_epoch_high_qc.clone())
240                    .is_ok();
241                drop(consensus_writer);
242
243                self.storage
244                    .write()
245                    .await
246                    .update_high_qc2(high_qc.clone())
247                    .await
248                    .map_err(|_| warn!("Failed to update high QC"))?;
249                self.storage
250                    .write()
251                    .await
252                    .update_next_epoch_high_qc2(next_epoch_high_qc.clone())
253                    .await
254                    .map_err(|_| warn!("Failed to update next epoch high QC"))?;
255
256                tracing::debug!(
257                    "Received Extended QC for view {:?} and epoch {:?}.",
258                    high_qc.view_number(),
259                    high_qc.epoch()
260                );
261                if high_qc_updated || next_high_qc_updated {
262                    // Send ViewChange indicating new view and new epoch.
263                    let next_epoch = high_qc.data.epoch().map(|x| x + 1);
264                    tracing::info!("Entering new epoch: {:?}", next_epoch);
265                    broadcast_event(
266                        Arc::new(HotShotEvent::ViewChange(
267                            high_qc.view_number() + 1,
268                            next_epoch,
269                        )),
270                        &sender,
271                    )
272                    .await;
273                }
274            },
275            _ => {},
276        }
277
278        Ok(())
279    }
280}

282#[async_trait]
283impl<TYPES: NodeType, I: NodeImplementation<TYPES>, V: Versions> TaskState
284    for ConsensusTaskState<TYPES, I, V>
285{
286    type Event = HotShotEvent<TYPES>;
287
288    async fn handle_event(
289        &mut self,
290        event: Arc<Self::Event>,
291        sender: &Sender<Arc<Self::Event>>,
292        receiver: &Receiver<Arc<Self::Event>>,
293    ) -> Result<()> {
294        self.handle(event, sender.clone(), receiver.clone()).await
295    }
296
297    /// Joins all subtasks.
298    fn cancel_subtasks(&mut self) {
299        // Cancel the old timeout task
300        std::mem::replace(&mut self.timeout_task, tokio::spawn(async {})).abort();
301    }
