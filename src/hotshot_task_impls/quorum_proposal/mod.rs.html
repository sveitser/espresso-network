43/// The state for the quorum proposal task.
44pub struct QuorumProposalTaskState<TYPES: NodeType, I: NodeImplementation<TYPES>, V: Versions> {
45    /// Latest view number that has been proposed for.
46    pub latest_proposed_view: TYPES::View,
47
48    /// Current epoch
49    pub cur_epoch: Option<TYPES::Epoch>,
50
51    /// Table for the in-progress proposal dependency tasks.
52    pub proposal_dependencies: BTreeMap<TYPES::View, JoinHandle<()>>,
53
54    /// Formed QCs
55    pub formed_quorum_certificates: BTreeMap<TYPES::View, QuorumCertificate2<TYPES>>,
56
57    /// Formed QCs for the next epoch
58    pub formed_next_epoch_quorum_certificates:
59        BTreeMap<TYPES::View, NextEpochQuorumCertificate2<TYPES>>,
60
61    /// Immutable instance state
62    pub instance_state: Arc<TYPES::InstanceState>,
63
64    /// Membership for Quorum Certs/votes
65    pub membership_coordinator: EpochMembershipCoordinator<TYPES>,
66
67    /// Our public key
68    pub public_key: TYPES::SignatureKey,
69
70    /// Our Private Key
71    pub private_key: <TYPES::SignatureKey as SignatureKey>::PrivateKey,
72
73    /// View timeout from config.
74    pub timeout: u64,
75
76    /// This node's storage ref
77    pub storage: I::Storage,
78
79    /// Shared consensus task state
80    pub consensus: OuterConsensus<TYPES>,
81
82    /// The node's id
83    pub id: u64,
84
85    /// The most recent upgrade certificate this node formed.
86    /// Note: this is ONLY for certificates that have been formed internally,
87    /// so that we can propose with them.
88    ///
89    /// Certificates received from other nodes will get reattached regardless of this fields,
90    /// since they will be present in the leaf we propose off of.
91    pub formed_upgrade_certificate: Option<UpgradeCertificate<TYPES>>,
92
93    /// Lock for a decided upgrade
94    pub upgrade_lock: UpgradeLock<TYPES, V>,
95
96    /// Number of blocks in an epoch, zero means there are no epochs
97    pub epoch_height: u64,
98
99    /// Formed light client state update certificates
100    pub formed_state_cert: BTreeMap<TYPES::Epoch, LightClientStateUpdateCertificate<TYPES>>,
101}

106    /// Create an event dependency
107    #[instrument(skip_all, fields(id = self.id, latest_proposed_view = *self.latest_proposed_view), name = "Create event dependency", level = "info")]
108    fn create_event_dependency(
109        &self,
110        dependency_type: ProposalDependency,
111        view_number: TYPES::View,
112        event_receiver: Receiver<Arc<HotShotEvent<TYPES>>>,
113    ) -> EventDependency<Arc<HotShotEvent<TYPES>>> {
114        let id = self.id;
115        EventDependency::new(
116            event_receiver,
117            Box::new(move |event| {
118                let event = event.as_ref();
119                let event_view = match dependency_type {
120                    ProposalDependency::Qc => {
121                        if let HotShotEvent::Qc2Formed(either::Left(qc)) = event {
122                            qc.view_number() + 1
123                        } else if let HotShotEvent::EpochRootQcFormed(root_qc) = event {
124                            root_qc.view_number() + 1
125                        } else {
126                            return false;
127                        }
128                    },
129                    ProposalDependency::TimeoutCert => {
130                        if let HotShotEvent::Qc2Formed(either::Right(timeout)) = event {
131                            timeout.view_number() + 1
132                        } else {
133                            return false;
134                        }
135                    },
136                    ProposalDependency::ViewSyncCert => {
137                        if let HotShotEvent::ViewSyncFinalizeCertificateRecv(view_sync_cert) = event
138                        {
139                            view_sync_cert.view_number()
140                        } else {
141                            return false;
142                        }
143                    },
144                    ProposalDependency::Proposal => {
145                        if let HotShotEvent::QuorumProposalPreliminarilyValidated(proposal) = event
146                        {
147                            proposal.data.view_number() + 1
148                        } else {
149                            return false;
150                        }
151                    },
152                    ProposalDependency::PayloadAndMetadata => {
153                        if let HotShotEvent::SendPayloadCommitmentAndMetadata(
154                            _payload_commitment,
155                            _builder_commitment,
156                            _metadata,
157                            view_number,
158                            _fee,
159                            _auction_result,
160                        ) = event
161                        {
162                            *view_number
163                        } else {
164                            return false;
165                        }
166                    },
167                    ProposalDependency::VidShare => {
168                        if let HotShotEvent::VidDisperseSend(vid_disperse, _) = event {
169                            vid_disperse.data.view_number()
170                        } else {
171                            return false;
172                        }
173                    },
174                };
175                let valid = event_view == view_number;
176                if valid {
177                    tracing::debug!(
178                        "Dependency {dependency_type:?} is complete for view {event_view:?}, my id is {id:?}!",
179                    );
180                }
181                valid
182            }),
183        )
184    }

186    /// Creates the requisite dependencies for the Quorum Proposal task. It also handles any event forwarding.
187    fn create_and_complete_dependencies(
188        &self,
189        view_number: TYPES::View,
190        event_receiver: &Receiver<Arc<HotShotEvent<TYPES>>>,
191        event: Arc<HotShotEvent<TYPES>>,
192    ) -> AndDependency<Vec<Vec<Arc<HotShotEvent<TYPES>>>>> {
193        let mut proposal_dependency = self.create_event_dependency(
194            ProposalDependency::Proposal,
195            view_number,
196            event_receiver.clone(),
197        );
198
199        let mut qc_dependency = self.create_event_dependency(
200            ProposalDependency::Qc,
201            view_number,
202            event_receiver.clone(),
203        );
204
205        let mut view_sync_dependency = self.create_event_dependency(
206            ProposalDependency::ViewSyncCert,
207            view_number,
208            event_receiver.clone(),
209        );
210
211        let mut timeout_dependency = self.create_event_dependency(
212            ProposalDependency::TimeoutCert,
213            view_number,
214            event_receiver.clone(),
215        );
216
217        let mut payload_commitment_dependency = self.create_event_dependency(
218            ProposalDependency::PayloadAndMetadata,
219            view_number,
220            event_receiver.clone(),
221        );
222
223        let mut vid_share_dependency = self.create_event_dependency(
224            ProposalDependency::VidShare,
225            view_number,
226            event_receiver.clone(),
227        );
228
229        let epoch_height = self.epoch_height;
230
231        // Next epoch QC dependency is fulfilled if we get the next epoch QC or
232        // form a current qc that isn't during transition
233        let mut next_epoch_qc_dependency = EventDependency::new(
234            event_receiver.clone(),
235            Box::new(move |event| {
236                if let HotShotEvent::NextEpochQc2Formed(Either::Left(next_epoch_qc)) =
237                    event.as_ref()
238                {
239                    return next_epoch_qc.view_number() + 1 == view_number;
240                }
241                if let HotShotEvent::EpochRootQcFormed(..) = event.as_ref() {
242                    // Epoch root QC is always not in epoch transition
243                    return true;
244                }
245                if let HotShotEvent::Qc2Formed(Either::Left(qc)) = event.as_ref() {
246                    if qc.view_number() + 1 == view_number {
247                        return qc
248                            .data
249                            .block_number
250                            .is_none_or(|bn| !is_epoch_transition(bn, epoch_height));
251                    }
252                }
253                false
254            }),
255        );
256
257        match event.as_ref() {
258            HotShotEvent::SendPayloadCommitmentAndMetadata(..) => {
259                payload_commitment_dependency.mark_as_completed(Arc::clone(&event));
260            },
261            HotShotEvent::QuorumProposalPreliminarilyValidated(..) => {
262                proposal_dependency.mark_as_completed(event);
263            },
264            HotShotEvent::Qc2Formed(quorum_certificate) => match quorum_certificate {
265                Either::Right(_) => timeout_dependency.mark_as_completed(event),
266                Either::Left(qc) => {
267                    if qc
268                        .data
269                        .block_number
270                        .is_none_or(|bn| !is_epoch_transition(bn, epoch_height))
271                    {
272                        next_epoch_qc_dependency.mark_as_completed(event.clone());
273                    }
274                    qc_dependency.mark_as_completed(event);
275                },
276            },
277            HotShotEvent::EpochRootQcFormed(..) => {
278                // Epoch root QC is always not in epoch transition
279                next_epoch_qc_dependency.mark_as_completed(event.clone());
280                qc_dependency.mark_as_completed(event);
281            },
282            HotShotEvent::ViewSyncFinalizeCertificateRecv(_) => {
283                view_sync_dependency.mark_as_completed(event);
284            },
285            HotShotEvent::VidDisperseSend(..) => {
286                vid_share_dependency.mark_as_completed(event);
287            },
288            HotShotEvent::NextEpochQc2Formed(Either::Left(_)) => {
289                next_epoch_qc_dependency.mark_as_completed(event);
290            },
291            _ => {},
292        };
293
294        // We have three cases to consider:
295        let mut secondary_deps = vec![
296            // 1. A timeout cert was received
297            AndDependency::from_deps(vec![timeout_dependency]),
298            // 2. A view sync cert was received.
299            AndDependency::from_deps(vec![view_sync_dependency]),
300        ];
301        // 3. A `Qc2Formed`` event (and `QuorumProposalRecv` event)
302        if *view_number > 1 {
303            secondary_deps.push(AndDependency::from_deps(vec![
304                qc_dependency,
305                proposal_dependency,
306                next_epoch_qc_dependency,
307            ]));
308        } else {
309            secondary_deps.push(AndDependency::from_deps(vec![qc_dependency]));
310        }
311
312        let primary_deps = vec![payload_commitment_dependency, vid_share_dependency];
313
314        AndDependency::from_deps(vec![OrDependency::from_deps(vec![
315            AndDependency::from_deps(vec![
316                OrDependency::from_deps(vec![AndDependency::from_deps(primary_deps)]),
317                OrDependency::from_deps(secondary_deps),
318            ]),
319        ])])
320    }

322    /// Create and store an [`AndDependency`] combining [`EventDependency`]s associated with the
323    /// given view number if it doesn't exist. Also takes in the received `event` to seed a
324    /// dependency as already completed. This allows for the task to receive a proposable event
325    /// without losing the data that it received, as the dependency task would otherwise have no
326    /// ability to receive the event and, thus, would never propose.
327    #[instrument(skip_all, fields(id = self.id, latest_proposed_view = *self.latest_proposed_view), name = "Create dependency task", level = "error")]
328    async fn create_dependency_task_if_new(
329        &mut self,
330        view_number: TYPES::View,
331        epoch_number: Option<TYPES::Epoch>,
332        event_receiver: Receiver<Arc<HotShotEvent<TYPES>>>,
333        event_sender: Sender<Arc<HotShotEvent<TYPES>>>,
334        event: Arc<HotShotEvent<TYPES>>,
335        epoch_transition_indicator: EpochTransitionIndicator,
336    ) -> Result<()> {
337        let epoch_membership = self
338            .membership_coordinator
339            .membership_for_epoch(epoch_number)
340            .await?;
341        let leader_in_current_epoch =
342            epoch_membership.leader(view_number).await? == self.public_key;
343        // If we are in the epoch transition and we are the leader in the next epoch,
344        // we might want to start collecting dependencies for our next epoch proposal.
345
346        let leader_in_next_epoch = !leader_in_current_epoch
347            && epoch_number.is_some()
348            && matches!(
349                epoch_transition_indicator,
350                EpochTransitionIndicator::InTransition
351            )
352            && epoch_membership
353                .next_epoch()
354                .await
355                .context(warn!(
356                    "Missing the randomized stake table for epoch {:?}",
357                    epoch_number.unwrap() + 1
358                ))?
359                .leader(view_number)
360                .await?
361                == self.public_key;
362
363        // Don't even bother making the task if we are not entitled to propose anyway.
364        ensure!(
365            leader_in_current_epoch || leader_in_next_epoch,
366            debug!("We are not the leader of the next view")
367        );
368
369        // Don't try to propose twice for the same view.
370        ensure!(
371            view_number > self.latest_proposed_view,
372            "We have already proposed for this view"
373        );
374
375        tracing::debug!(
376            "Attempting to make dependency task for view {view_number:?} and event {event:?}"
377        );
378
379        ensure!(
380            !self.proposal_dependencies.contains_key(&view_number),
381            "Task already exists"
382        );
383
384        let dependency_chain =
385            self.create_and_complete_dependencies(view_number, &event_receiver, event);
386
387        let dependency_task = DependencyTask::new(
388            dependency_chain,
389            ProposalDependencyHandle {
390                latest_proposed_view: self.latest_proposed_view,
391                view_number,
392                sender: event_sender,
393                receiver: event_receiver,
394                membership: epoch_membership,
395                public_key: self.public_key.clone(),
396                private_key: self.private_key.clone(),
397                instance_state: Arc::clone(&self.instance_state),
398                consensus: OuterConsensus::new(Arc::clone(&self.consensus.inner_consensus)),
399                timeout: self.timeout,
400                formed_upgrade_certificate: self.formed_upgrade_certificate.clone(),
401                upgrade_lock: self.upgrade_lock.clone(),
402                id: self.id,
403                view_start_time: Instant::now(),
404                epoch_height: self.epoch_height,
405            },
406        );
407        self.proposal_dependencies
408            .insert(view_number, dependency_task.run());
409
410        Ok(())
411    }

413    /// Update the latest proposed view number.
414    #[instrument(skip_all, fields(id = self.id, latest_proposed_view = *self.latest_proposed_view), name = "Update latest proposed view", level = "error")]
415    async fn update_latest_proposed_view(&mut self, new_view: TYPES::View) -> bool {
416        if *self.latest_proposed_view < *new_view {
417            tracing::debug!(
418                "Updating latest proposed view from {} to {}",
419                *self.latest_proposed_view,
420                *new_view
421            );
422
423            // Cancel the old dependency tasks.
424            for view in (*self.latest_proposed_view + 1)..=(*new_view) {
425                if let Some(dependency) = self.proposal_dependencies.remove(&TYPES::View::new(view))
426                {
427                    dependency.abort();
428                }
429            }
430
431            self.latest_proposed_view = new_view;
432
433            return true;
434        }
435        false
436    }

438    /// Handles a consensus event received on the event stream
439    #[instrument(skip_all, fields(id = self.id, latest_proposed_view = *self.latest_proposed_view, epoch = self.cur_epoch.map(|x| *x)), name = "handle method", level = "error", target = "QuorumProposalTaskState")]
440    pub async fn handle(
441        &mut self,
442        event: Arc<HotShotEvent<TYPES>>,
443        event_receiver: Receiver<Arc<HotShotEvent<TYPES>>>,
444        event_sender: Sender<Arc<HotShotEvent<TYPES>>>,
445    ) -> Result<()> {
446        let epoch_number = self.cur_epoch;
447        let maybe_high_qc_block_number = self.consensus.read().await.high_qc().data.block_number;
448        let epoch_transition_indicator = if maybe_high_qc_block_number.is_some_and(|bn| {
449            is_epoch_transition(bn, self.epoch_height) && !is_last_block(bn, self.epoch_height)
450        }) {
451            EpochTransitionIndicator::InTransition
452        } else {
453            EpochTransitionIndicator::NotInTransition
454        };
455        match event.as_ref() {
456            HotShotEvent::UpgradeCertificateFormed(cert) => {
457                tracing::debug!(
458                    "Upgrade certificate received for view {}!",
459                    *cert.view_number
460                );
461                // Update our current upgrade_cert as long as we still have a chance of reaching a decide on it in time.
462                if cert.data.decide_by >= self.latest_proposed_view + 3 {
463                    tracing::debug!("Updating current formed_upgrade_certificate");
464
465                    self.formed_upgrade_certificate = Some(cert.clone());
466                }
467            },
468            HotShotEvent::Qc2Formed(cert) => match cert.clone() {
469                either::Right(timeout_cert) => {
470                    let view_number = timeout_cert.view_number + 1;
471                    self.create_dependency_task_if_new(
472                        view_number,
473                        epoch_number,
474                        event_receiver,
475                        event_sender,
476                        Arc::clone(&event),
477                        epoch_transition_indicator,
478                    )
479                    .await?;
480                },
481                either::Left(qc) => {
482                    // Only update if the qc is from a newer view
483                    if qc.view_number <= self.consensus.read().await.high_qc().view_number {
484                        tracing::trace!(
485                            "Received a QC for a view that was not > than our current high QC"
486                        );
487                    }
488
489                    self.formed_quorum_certificates
490                        .insert(qc.view_number(), qc.clone());
491
492                    handle_eqc_formed(
493                        qc.view_number(),
494                        qc.data.leaf_commit,
495                        qc.data.block_number,
496                        self,
497                        &event_sender,
498                    )
499                    .await;
500
501                    let view_number = qc.view_number() + 1;
502                    self.create_dependency_task_if_new(
503                        view_number,
504                        epoch_number,
505                        event_receiver,
506                        event_sender,
507                        Arc::clone(&event),
508                        epoch_transition_indicator,
509                    )
510                    .await?;
511                },
512            },
513
514            HotShotEvent::EpochRootQcFormed(EpochRootQuorumCertificate { qc, state_cert }) => {
515                // Only update if the qc is from a newer view
516                if qc.view_number() <= self.consensus.read().await.high_qc().view_number {
517                    tracing::trace!(
518                        "Received a QC for a view that was not > than our current high QC"
519                    );
520                }
521
522                self.formed_quorum_certificates
523                    .insert(qc.view_number(), qc.clone());
524                self.formed_state_cert
525                    .insert(state_cert.epoch, state_cert.clone());
526
527                self.storage
528                    .update_high_qc2_and_state_cert(qc.clone(), state_cert.clone())
529                    .await
530                    .wrap()
531                    .context(error!(
532                        "Failed to update the epoch root QC and state cert in storage!"
533                    ))?;
534
535                let view_number = qc.view_number() + 1;
536                self.create_dependency_task_if_new(
537                    view_number,
538                    epoch_number,
539                    event_receiver,
540                    event_sender,
541                    Arc::clone(&event),
542                    epoch_transition_indicator,
543                )
544                .await?;
545            },
546            HotShotEvent::SendPayloadCommitmentAndMetadata(
547                _payload_commitment,
548                _builder_commitment,
549                _metadata,
550                view_number,
551                _fee,
552                _auction_result,
553            ) => {
554                let view_number = *view_number;
555
556                self.create_dependency_task_if_new(
557                    view_number,
558                    epoch_number,
559                    event_receiver,
560                    event_sender,
561                    Arc::clone(&event),
562                    epoch_transition_indicator,
563                )
564                .await?;
565            },
566            HotShotEvent::ViewSyncFinalizeCertificateRecv(certificate) => {
567                let epoch_number = certificate.data.epoch;
568                let epoch_membership = self
569                    .membership_coordinator
570                    .stake_table_for_epoch(epoch_number)
571                    .await
572                    .context(warn!("No Stake Table for Epoch = {epoch_number:?}"))?;
573
574                let membership_stake_table = epoch_membership.stake_table().await;
575                let membership_success_threshold = epoch_membership.success_threshold().await;
576
577                certificate
578                    .is_valid_cert(
579                        &StakeTableEntries::<TYPES>::from(membership_stake_table).0,
580                        membership_success_threshold,
581                        &self.upgrade_lock,
582                    )
583                    .await
584                    .context(|e| {
585                        warn!(
586                            "View Sync Finalize certificate {:?} was invalid: {}",
587                            certificate.data(),
588                            e
589                        )
590                    })?;
591
592                let view_number = certificate.view_number;
593
594                self.create_dependency_task_if_new(
595                    view_number,
596                    epoch_number,
597                    event_receiver,
598                    event_sender,
599                    event,
600                    epoch_transition_indicator,
601                )
602                .await?;
603            },
604            HotShotEvent::QuorumProposalPreliminarilyValidated(proposal) => {
605                let view_number = proposal.data.view_number();
606                // All nodes get the latest proposed view as a proxy of `cur_view` of old.
607                if !self.update_latest_proposed_view(view_number).await {
608                    tracing::trace!("Failed to update latest proposed view");
609                }
610
611                self.create_dependency_task_if_new(
612                    view_number + 1,
613                    epoch_number,
614                    event_receiver,
615                    event_sender,
616                    Arc::clone(&event),
617                    epoch_transition_indicator,
618                )
619                .await?;
620            },
621            HotShotEvent::QuorumProposalSend(proposal, _) => {
622                let view = proposal.data.view_number();
623
624                ensure!(
625                    self.update_latest_proposed_view(view).await,
626                    "Failed to update latest proposed view"
627                );
628            },
629            HotShotEvent::VidDisperseSend(vid_disperse, _) => {
630                let view_number = vid_disperse.data.view_number();
631                self.create_dependency_task_if_new(
632                    view_number,
633                    epoch_number,
634                    event_receiver,
635                    event_sender,
636                    Arc::clone(&event),
637                    epoch_transition_indicator,
638                )
639                .await?;
640            },
641            HotShotEvent::ViewChange(view, epoch) => {
642                if epoch > &self.cur_epoch {
643                    self.cur_epoch = *epoch;
644                }
645                let keep_view = TYPES::View::new(view.saturating_sub(1));
646                self.cancel_tasks(keep_view);
647            },
648            HotShotEvent::Timeout(view, ..) => {
649                let keep_view = TYPES::View::new(view.saturating_sub(1));
650                self.cancel_tasks(keep_view);
651            },
652            HotShotEvent::NextEpochQc2Formed(Either::Left(next_epoch_qc)) => {
653                // Only update if the qc is from a newer view
654                let current_next_epoch_qc =
655                    self.consensus.read().await.next_epoch_high_qc().cloned();
656                ensure!(current_next_epoch_qc.is_none() ||
657                    next_epoch_qc.view_number > current_next_epoch_qc.unwrap().view_number,
658                    debug!("Received a next epoch QC for a view that was not > than our current next epoch high QC")
659                );
660
661                self.formed_next_epoch_quorum_certificates
662                    .insert(next_epoch_qc.view_number(), next_epoch_qc.clone());
663
664                handle_eqc_formed(
665                    next_epoch_qc.view_number(),
666                    next_epoch_qc.data.leaf_commit,
667                    next_epoch_qc.data.block_number,
668                    self,
669                    &event_sender,
670                )
671                .await;
672
673                let view_number = next_epoch_qc.view_number() + 1;
674                self.create_dependency_task_if_new(
675                    view_number,
676                    epoch_number,
677                    event_receiver,
678                    event_sender,
679                    Arc::clone(&event),
680                    epoch_transition_indicator,
681                )
682                .await?;
683            },
684            _ => {},
685        }
686        Ok(())
687    }
688
689    /// Cancel all tasks the consensus tasks has spawned before the given view
690    pub fn cancel_tasks(&mut self, view: TYPES::View) {
691        let keep = self.proposal_dependencies.split_off(&view);
692        while let Some((_, task)) = self.proposal_dependencies.pop_first() {
693            task.abort();
694        }
695        self.proposal_dependencies = keep;
696    }
697}

712    }
