53/// Trigger a request to the network for a proposal for a view and wait for the response or timeout.
54#[instrument(skip_all)]
55#[allow(clippy::too_many_arguments)]
56pub(crate) async fn fetch_proposal<TYPES: NodeType, V: Versions>(
57    qc: &QuorumCertificate2<TYPES>,
58    event_sender: Sender<Arc<HotShotEvent<TYPES>>>,
59    event_receiver: Receiver<Arc<HotShotEvent<TYPES>>>,
60    membership_coordinator: EpochMembershipCoordinator<TYPES>,
61    consensus: OuterConsensus<TYPES>,
62    sender_public_key: TYPES::SignatureKey,
63    sender_private_key: <TYPES::SignatureKey as SignatureKey>::PrivateKey,
64    upgrade_lock: &UpgradeLock<TYPES, V>,
65    epoch_height: u64,
66) -> Result<(Leaf2<TYPES>, View<TYPES>)> {
67    let view_number = qc.view_number();
68    let leaf_commit = qc.data.leaf_commit;
69    // We need to be able to sign this request before submitting it to the network. Compute the
70    // payload first.
71    let signed_proposal_request = ProposalRequestPayload {
72        view_number,
73        key: sender_public_key,
74    };
75
76    // Finally, compute the signature for the payload.
77    let signature = TYPES::SignatureKey::sign(
78        &sender_private_key,
79        signed_proposal_request.commit().as_ref(),
80    )
81    .wrap()
82    .context(error!("Failed to sign proposal. This should never happen."))?;
83
84    tracing::info!("Sending proposal request for view {}", view_number);
85
86    // First, broadcast that we need a proposal to the current leader
87    broadcast_event(
88        HotShotEvent::QuorumProposalRequestSend(signed_proposal_request, signature).into(),
89        &event_sender,
90    )
91    .await;
92
93    let mut rx = event_receiver.clone();
94    // Make a background task to await the arrival of the event data.
95    let Ok(Some(proposal)) =
96        // We want to explicitly timeout here so we aren't waiting around for the data.
97        timeout(REQUEST_TIMEOUT, async move {
98            // We want to iterate until the proposal is not None, or until we reach the timeout.
99            while let Ok(event) = rx.recv_direct().await {
100                if let HotShotEvent::QuorumProposalResponseRecv(quorum_proposal) = event.as_ref() {
101                    let leaf = Leaf2::from_quorum_proposal(&quorum_proposal.data);
102                    if leaf.view_number() == view_number && leaf.commit() == leaf_commit {
103                        return Some(quorum_proposal.clone());
104                    }
105                }
106            }
107            None
108        })
109        .await
110    else {
111        bail!("Request for proposal failed");
112    };
113
114    let view_number = proposal.data.view_number();
115    let justify_qc = proposal.data.justify_qc().clone();
116
117    let justify_qc_epoch = justify_qc.data.epoch();
118
119    let epoch_membership = membership_coordinator
120        .stake_table_for_epoch(justify_qc_epoch)
121        .await?;
122    let membership_stake_table = epoch_membership.stake_table().await;
123    let membership_success_threshold = epoch_membership.success_threshold().await;
124
125    justify_qc
126        .is_valid_cert(
127            &StakeTableEntries::<TYPES>::from(membership_stake_table).0,
128            membership_success_threshold,
129            upgrade_lock,
130        )
131        .await
132        .context(|e| warn!("Invalid justify_qc in proposal for view {view_number}: {e}"))?;
133
134    let mut consensus_writer = consensus.write().await;
135    let leaf = Leaf2::from_quorum_proposal(&proposal.data);
136    let state = Arc::new(
137        <TYPES::ValidatedState as ValidatedState<TYPES>>::from_header(proposal.data.block_header()),
138    );
139
140    if let Err(e) = consensus_writer.update_leaf(leaf.clone(), Arc::clone(&state), None) {
141        tracing::trace!("{e:?}");
142    }
143    let view = View {
144        view_inner: ViewInner::Leaf {
145            leaf: leaf.commit(),
146            state,
147            delta: None,
148            epoch: leaf.epoch(epoch_height),
149        },
150    };
151    Ok((leaf, view))
152}
153pub async fn handle_drb_result<TYPES: NodeType, I: NodeImplementation<TYPES>>(
154    membership: &Arc<RwLock<TYPES::Membership>>,
155    epoch: TYPES::Epoch,
156    storage: &I::Storage,
157    consensus: &OuterConsensus<TYPES>,
158    drb_result: DrbResult,
159) {
160    let mut consensus_writer = consensus.write().await;
161    consensus_writer.drb_results.store_result(epoch, drb_result);
162    drop(consensus_writer);
163    tracing::debug!("Calling add_drb_result for epoch {epoch}");
164    if let Err(e) = storage.add_drb_result(epoch, drb_result).await {
165        tracing::error!("Failed to store drb result for epoch {epoch}: {e}");
166    }
167
168    membership.write().await.add_drb_result(epoch, drb_result)
169}
170/// Start the DRB computation task for the next epoch.
171fn start_drb_task<TYPES: NodeType, I: NodeImplementation<TYPES>>(
172    seed: DrbSeedInput,
173    epoch: TYPES::Epoch,
174    membership: &Arc<RwLock<TYPES::Membership>>,
175    storage: &I::Storage,
176    consensus: &OuterConsensus<TYPES>,
177) {
178    let membership = membership.clone();
179    let storage = storage.clone();
180    let consensus = consensus.clone();
181    tokio::spawn(async move {
182        let drb_result = tokio::task::spawn_blocking(move || {
183            hotshot_types::drb::compute_drb_result::<TYPES>(seed)
184        })
185        .await
186        .unwrap();
187
188        handle_drb_result::<TYPES, I>(&membership, epoch, &storage, &consensus, drb_result).await;
189        drb_result
190    });
191}
192/// Handles calling add_epoch_root and sync_l1 on Membership if necessary.
193async fn decide_epoch_root<TYPES: NodeType, I: NodeImplementation<TYPES>>(
194    decided_leaf: &Leaf2<TYPES>,
195    epoch_height: u64,
196    membership: &Arc<RwLock<TYPES::Membership>>,
197    storage: &I::Storage,
198    consensus: &OuterConsensus<TYPES>,
199) {
200    let decided_block_number = decided_leaf.block_header().block_number();
201
202    // Skip if this is not the expected block.
203    if epoch_height != 0 && is_epoch_root(decided_block_number, epoch_height) {
204        let next_epoch_number =
205            TYPES::Epoch::new(epoch_from_block_number(decided_block_number, epoch_height) + 2);
206
207        let mut start = Instant::now();
208        if let Err(e) = storage
209            .add_epoch_root(next_epoch_number, decided_leaf.block_header().clone())
210            .await
211        {
212            tracing::error!("Failed to store epoch root for epoch {next_epoch_number}: {e}");
213        }
214        tracing::info!("Time taken to store epoch root: {:?}", start.elapsed());
215
216        start = Instant::now();
217        let write_callback = {
218            tracing::debug!("Calling add_epoch_root for epoch {next_epoch_number}");
219            let membership_reader = membership.read().await;
220            membership_reader
221                .add_epoch_root(next_epoch_number, decided_leaf.block_header().clone())
222                .await
223        };
224        if let Some(write_callback) = write_callback {
225            let mut membership_writer = membership.write().await;
226            write_callback(&mut *membership_writer);
227        }
228        tracing::info!("Time taken to add epoch root: {:?}", start.elapsed());
229
230        let mut consensus_writer = consensus.write().await;
231        consensus_writer
232            .drb_results
233            .garbage_collect(next_epoch_number);
234        drop(consensus_writer);
235
236        let Ok(drb_seed_input_vec) = bincode::serialize(&decided_leaf.justify_qc().signatures)
237        else {
238            tracing::error!("Failed to serialize the QC signature.");
239            return;
240        };
241
242        let mut drb_seed_input = [0u8; 32];
243        let len = drb_seed_input_vec.len().min(32);
244        drb_seed_input[..len].copy_from_slice(&drb_seed_input_vec[..len]);
245
246        start_drb_task::<TYPES, I>(
247            drb_seed_input,
248            next_epoch_number,
249            membership,
250            storage,
251            consensus,
252        );
253    }
254}

256/// Helper type to give names and to the output values of the leaf chain traversal operation.
257#[derive(Debug)]
258pub struct LeafChainTraversalOutcome<TYPES: NodeType> {
259    /// The new locked view obtained from a 2 chain starting from the proposal's parent.
260    pub new_locked_view_number: Option<TYPES::View>,
261
262    /// The new decided view obtained from a 3 chain starting from the proposal's parent.
263    pub new_decided_view_number: Option<TYPES::View>,
264
265    /// The qc for the decided chain.
266    pub new_decide_qc: Option<QuorumCertificate2<TYPES>>,
267
268    /// The decided leaves with corresponding validated state and VID info.
269    pub leaf_views: Vec<LeafInfo<TYPES>>,
270
271    /// The transactions in the block payload for each leaf.
272    pub included_txns: Option<HashSet<Commitment<<TYPES as NodeType>::Transaction>>>,
273
274    /// The most recent upgrade certificate from one of the leaves.
275    pub decided_upgrade_cert: Option<UpgradeCertificate<TYPES>>,
276}

278/// We need Default to be implemented because the leaf ascension has very few failure branches,
279/// and when they *do* happen, we still return intermediate states. Default makes the burden
280/// of filling values easier.
281impl<TYPES: NodeType + Default> Default for LeafChainTraversalOutcome<TYPES> {
282    /// The default method for this type is to set all of the returned values to `None`.
283    fn default() -> Self {
284        Self {
285            new_locked_view_number: None,
286            new_decided_view_number: None,
287            new_decide_qc: None,
288            leaf_views: Vec::new(),
289            included_txns: None,
290            decided_upgrade_cert: None,
291        }
292    }
293}

295/// calculate the new decided leaf chain based on the rules of HotStuff 2
296///
297/// # Panics
298/// If the leaf chain contains no decided leaf while reaching a decided view, which should be
299/// impossible.
300pub async fn decide_from_proposal_2<TYPES: NodeType, I: NodeImplementation<TYPES>, V: Versions>(
301    proposal: &QuorumProposalWrapper<TYPES>,
302    consensus: OuterConsensus<TYPES>,
303    existing_upgrade_cert: Arc<RwLock<Option<UpgradeCertificate<TYPES>>>>,
304    public_key: &TYPES::SignatureKey,
305    with_epochs: bool,
306    membership: &EpochMembershipCoordinator<TYPES>,
307    storage: &I::Storage,
308) -> LeafChainTraversalOutcome<TYPES> {
309    let mut res = LeafChainTraversalOutcome::default();
310    let consensus_reader = consensus.read().await;
311    let proposed_leaf = Leaf2::from_quorum_proposal(proposal);
312    res.new_locked_view_number = Some(proposed_leaf.justify_qc().view_number());
313
314    // If we don't have the proposals parent return early
315    let Some(parent_info) = consensus_reader
316        .parent_leaf_info(&proposed_leaf, public_key)
317        .await
318    else {
319        return res;
320    };
321    // Get the parents parent and check if it's consecutive in view to the parent, if so we can decided
322    // the grandparents view.  If not we're done.
323    let Some(grand_parent_info) = consensus_reader
324        .parent_leaf_info(&parent_info.leaf, public_key)
325        .await
326    else {
327        return res;
328    };
329    if grand_parent_info.leaf.view_number() + 1 != parent_info.leaf.view_number() {
330        return res;
331    }
332    res.new_decide_qc = Some(parent_info.leaf.justify_qc().clone());
333    let decided_view_number = grand_parent_info.leaf.view_number();
334    res.new_decided_view_number = Some(decided_view_number);
335    // We've reached decide, now get the leaf chain all the way back to the last decided view, not including it.
336    let old_anchor_view = consensus_reader.last_decided_view();
337    let mut current_leaf_info = Some(grand_parent_info);
338    let existing_upgrade_cert_reader = existing_upgrade_cert.read().await;
339    let mut txns = HashSet::new();
340    while current_leaf_info
341        .as_ref()
342        .is_some_and(|info| info.leaf.view_number() > old_anchor_view)
343    {
344        // unwrap is safe, we just checked that he option is some
345        let info = &mut current_leaf_info.unwrap();
346        // Check if there's a new upgrade certificate available.
347        if let Some(cert) = info.leaf.upgrade_certificate() {
348            if info.leaf.upgrade_certificate() != *existing_upgrade_cert_reader {
349                if cert.data.decide_by < decided_view_number {
350                    tracing::warn!("Failed to decide an upgrade certificate in time. Ignoring.");
351                } else {
352                    tracing::info!("Reached decide on upgrade certificate: {cert:?}");
353                    res.decided_upgrade_cert = Some(cert.clone());
354                }
355            }
356        }
357
358        // If the block payload is available for this leaf, include it in
359        // the leaf chain that we send to the client.
360        if let Some(payload) = consensus_reader
361            .saved_payloads()
362            .get(&info.leaf.view_number())
363        {
364            info.leaf
365                .fill_block_payload_unchecked(payload.as_ref().payload.clone());
366        }
367
368        if let Some(ref payload) = info.leaf.block_payload() {
369            for txn in payload.transaction_commitments(info.leaf.block_header().metadata()) {
370                txns.insert(txn);
371            }
372        }
373
374        current_leaf_info = consensus_reader
375            .parent_leaf_info(&info.leaf, public_key)
376            .await;
377        res.leaf_views.push(info.clone());
378    }
379
380    if !txns.is_empty() {
381        res.included_txns = Some(txns);
382    }
383
384    if with_epochs && res.new_decided_view_number.is_some() {
385        let Some(first_leaf) = res.leaf_views.first() else {
386            return res;
387        };
388        let epoch_height = consensus_reader.epoch_height;
389        consensus_reader
390            .metrics
391            .last_synced_block_height
392            .set(usize::try_from(first_leaf.leaf.height()).unwrap_or(0));
393        drop(consensus_reader);
394
395        for decided_leaf_info in &res.leaf_views {
396            decide_epoch_root::<TYPES, I>(
397                &decided_leaf_info.leaf,
398                epoch_height,
399                membership.membership(),
400                storage,
401                &consensus,
402            )
403            .await;
404        }
405    }
406
407    res
408}

410/// Ascends the leaf chain by traversing through the parent commitments of the proposal. We begin
411/// by obtaining the parent view, and if we are in a chain (i.e. the next view from the parent is
412/// one view newer), then we begin attempting to form the chain. This is a direct impl from
413/// [HotStuff](https://arxiv.org/pdf/1803.05069) section 5:
414///
415/// > When a node b* carries a QC that refers to a direct parent, i.e., b*.justify.node = b*.parent,
416/// > we say that it forms a One-Chain. Denote by b'' = b*.justify.node. Node b* forms a Two-Chain,
417/// > if in addition to forming a One-Chain, b''.justify.node = b''.parent.
418/// > It forms a Three-Chain, if b'' forms a Two-Chain.
419///
420/// We follow this exact logic to determine if we are able to reach a commit and a decide. A commit
421/// is reached when we have a two chain, and a decide is reached when we have a three chain.
422///
423/// # Example
424/// Suppose we have a decide for view 1, and we then move on to get undecided views 2, 3, and 4. Further,
425/// suppose that our *next* proposal is for view 5, but this leader did not see info for view 4, so the
426/// justify qc of the proposal points to view 3. This is fine, and the undecided chain now becomes
427/// 2-3-5.
428///
429/// Assuming we continue with honest leaders, we then eventually could get a chain like: 2-3-5-6-7-8. This
430/// will prompt a decide event to occur (this code), where the `proposal` is for view 8. Now, since the
431/// lowest value in the 3-chain here would be 5 (excluding 8 since we only walk the parents), we begin at
432/// the first link in the chain, and walk back through all undecided views, making our new anchor view 5,
433/// and out new locked view will be 6.
434///
435/// Upon receipt then of a proposal for view 9, assuming it is valid, this entire process will repeat, and
436/// the anchor view will be set to view 6, with the locked view as view 7.
437///
438/// # Panics
439/// If the leaf chain contains no decided leaf while reaching a decided view, which should be
440/// impossible.
441#[allow(clippy::too_many_arguments)]
442pub async fn decide_from_proposal<TYPES: NodeType, I: NodeImplementation<TYPES>, V: Versions>(
443    proposal: &QuorumProposalWrapper<TYPES>,
444    consensus: OuterConsensus<TYPES>,
445    existing_upgrade_cert: Arc<RwLock<Option<UpgradeCertificate<TYPES>>>>,
446    public_key: &TYPES::SignatureKey,
447    with_epochs: bool,
448    membership: &Arc<RwLock<TYPES::Membership>>,
449    storage: &I::Storage,
450    epoch_height: u64,
451) -> LeafChainTraversalOutcome<TYPES> {
452    let consensus_reader = consensus.read().await;
453    let existing_upgrade_cert_reader = existing_upgrade_cert.read().await;
454    let view_number = proposal.view_number();
455    let parent_view_number = proposal.justify_qc().view_number();
456    let old_anchor_view = consensus_reader.last_decided_view();
457
458    let mut last_view_number_visited = view_number;
459    let mut current_chain_length = 0usize;
460    let mut res = LeafChainTraversalOutcome::default();
461
462    if let Err(e) = consensus_reader.visit_leaf_ancestors(
463        parent_view_number,
464        Terminator::Exclusive(old_anchor_view),
465        true,
466        |leaf, state, delta| {
467            // This is the core paper logic. We're implementing the chain in chained hotstuff.
468            if res.new_decided_view_number.is_none() {
469                // If the last view number is the child of the leaf we've moved to...
470                if last_view_number_visited == leaf.view_number() + 1 {
471                    last_view_number_visited = leaf.view_number();
472
473                    // The chain grows by one
474                    current_chain_length += 1;
475
476                    // We emit a locked view when the chain length is 2
477                    if current_chain_length == 2 {
478                        res.new_locked_view_number = Some(leaf.view_number());
479                        // The next leaf in the chain, if there is one, is decided, so this
480                        // leaf's justify_qc would become the QC for the decided chain.
481                        res.new_decide_qc = Some(leaf.justify_qc().clone());
482                    } else if current_chain_length == 3 {
483                        // And we decide when the chain length is 3.
484                        res.new_decided_view_number = Some(leaf.view_number());
485                    }
486                } else {
487                    // There isn't a new chain extension available, so we signal to the callback
488                    // owner that we can exit for now.
489                    return false;
490                }
491            }
492
493            // Now, if we *have* reached a decide, we need to do some state updates.
494            if let Some(new_decided_view) = res.new_decided_view_number {
495                // First, get a mutable reference to the provided leaf.
496                let mut leaf = leaf.clone();
497
498                // Update the metrics
499                if leaf.view_number() == new_decided_view {
500                    consensus_reader
501                        .metrics
502                        .last_synced_block_height
503                        .set(usize::try_from(leaf.height()).unwrap_or(0));
504                }
505
506                // Check if there's a new upgrade certificate available.
507                if let Some(cert) = leaf.upgrade_certificate() {
508                    if leaf.upgrade_certificate() != *existing_upgrade_cert_reader {
509                        if cert.data.decide_by < view_number {
510                            tracing::warn!(
511                                "Failed to decide an upgrade certificate in time. Ignoring."
512                            );
513                        } else {
514                            tracing::info!("Reached decide on upgrade certificate: {cert:?}");
515                            res.decided_upgrade_cert = Some(cert.clone());
516                        }
517                    }
518                }
519                // If the block payload is available for this leaf, include it in
520                // the leaf chain that we send to the client.
521                if let Some(payload) = consensus_reader.saved_payloads().get(&leaf.view_number()) {
522                    leaf.fill_block_payload_unchecked(payload.as_ref().payload.clone());
523                }
524
525                // Get the VID share at the leaf's view number, corresponding to our key
526                // (if one exists)
527                let vid_share = consensus_reader
528                    .vid_shares()
529                    .get(&leaf.view_number())
530                    .and_then(|key_map| key_map.get(public_key))
531                    .and_then(|epoch_map| epoch_map.get(&leaf.epoch(epoch_height)))
532                    .map(|prop| prop.data.clone());
533
534                let state_cert = if leaf.with_epoch
535                    && is_epoch_root(
536                        leaf.block_header().block_number(),
537                        consensus_reader.epoch_height,
538                    ) {
539                    match consensus_reader.state_cert() {
540                        // Sanity check that the state cert is for the same view as the decided leaf
541                        Some(state_cert)
542                            if state_cert.light_client_state.view_number
543                                == leaf.view_number().u64() =>
544                        {
545                            Some(state_cert.clone())
546                        },
547                        _ => None,
548                    }
549                } else {
550                    None
551                };
552
553                // Add our data into a new `LeafInfo`
554                res.leaf_views.push(LeafInfo::new(
555                    leaf.clone(),
556                    Arc::clone(&state),
557                    delta.clone(),
558                    vid_share,
559                    state_cert,
560                ));
561                if let Some(ref payload) = leaf.block_payload() {
562                    res.included_txns = Some(
563                        payload
564                            .transaction_commitments(leaf.block_header().metadata())
565                            .into_iter()
566                            .collect::<HashSet<_>>(),
567                    );
568                }
569            }
570            true
571        },
572    ) {
573        tracing::debug!("Leaf ascension failed; error={e}");
574    }
575
576    let epoch_height = consensus_reader.epoch_height;
577    drop(consensus_reader);
578
579    if with_epochs && res.new_decided_view_number.is_some() {
580        for decided_leaf_info in &res.leaf_views {
581            decide_epoch_root::<TYPES, I>(
582                &decided_leaf_info.leaf,
583                epoch_height,
584                membership,
585                storage,
586                &consensus,
587            )
588            .await;
589        }
590    }
591
592    res
593}

595/// Gets the parent leaf and state from the parent of a proposal, returning an [`utils::anytrace::Error`] if not.
596#[instrument(skip_all)]
597#[allow(clippy::too_many_arguments)]
598pub(crate) async fn parent_leaf_and_state<TYPES: NodeType, V: Versions>(
599    event_sender: &Sender<Arc<HotShotEvent<TYPES>>>,
600    event_receiver: &Receiver<Arc<HotShotEvent<TYPES>>>,
601    membership: EpochMembershipCoordinator<TYPES>,
602    public_key: TYPES::SignatureKey,
603    private_key: <TYPES::SignatureKey as SignatureKey>::PrivateKey,
604    consensus: OuterConsensus<TYPES>,
605    upgrade_lock: &UpgradeLock<TYPES, V>,
606    parent_qc: &QuorumCertificate2<TYPES>,
607    epoch_height: u64,
608) -> Result<(Leaf2<TYPES>, Arc<<TYPES as NodeType>::ValidatedState>)> {
609    let consensus_reader = consensus.read().await;
610    let vsm_contains_parent_view = consensus_reader
611        .validated_state_map()
612        .contains_key(&parent_qc.view_number());
613    drop(consensus_reader);
614
615    if !vsm_contains_parent_view {
616        let _ = fetch_proposal(
617            parent_qc,
618            event_sender.clone(),
619            event_receiver.clone(),
620            membership,
621            consensus.clone(),
622            public_key.clone(),
623            private_key.clone(),
624            upgrade_lock,
625            epoch_height,
626        )
627        .await
628        .context(info!("Failed to fetch proposal"))?;
629    }
630
631    let consensus_reader = consensus.read().await;
632    let parent_view = consensus_reader.validated_state_map().get(&parent_qc.view_number()).context(
633        debug!("Couldn't find parent view in state map, waiting for replica to see proposal; parent_view_number: {}", *parent_qc.view_number())
634    )?;
635
636    let (leaf_commitment, state) = parent_view.leaf_and_state().context(
637        info!("Parent of high QC points to a view without a proposal; parent_view_number: {}, parent_view {:?}", *parent_qc.view_number(), parent_view)
638    )?;
639
640    if leaf_commitment != consensus_reader.high_qc().data().leaf_commit {
641        // NOTE: This happens on the genesis block
642        tracing::debug!(
643            "They don't equal: {:?}   {:?}",
644            leaf_commitment,
645            consensus_reader.high_qc().data().leaf_commit
646        );
647    }
648
649    let leaf = consensus_reader
650        .saved_leaves()
651        .get(&leaf_commitment)
652        .context(info!("Failed to find high QC of parent"))?;
653
654    Ok((leaf.clone(), Arc::clone(state)))
655}

657pub(crate) async fn update_high_qc<TYPES: NodeType, I: NodeImplementation<TYPES>, V: Versions>(
658    proposal: &Proposal<TYPES, QuorumProposalWrapper<TYPES>>,
659    validation_info: &ValidationInfo<TYPES, I, V>,
660) -> Result<()> {
661    let in_transition_epoch = proposal
662        .data
663        .justify_qc()
664        .data
665        .block_number
666        .is_some_and(|bn| {
667            !is_transition_block(bn, validation_info.epoch_height)
668                && is_epoch_transition(bn, validation_info.epoch_height)
669                && bn % validation_info.epoch_height != 0
670        });
671    let justify_qc = proposal.data.justify_qc();
672    let maybe_next_epoch_justify_qc = proposal.data.next_epoch_justify_qc();
673    if !in_transition_epoch {
674        tracing::debug!(
675            "Storing high QC for view {:?} and height {:?}",
676            justify_qc.view_number(),
677            justify_qc.data.block_number
678        );
679        if let Err(e) = validation_info
680            .storage
681            .update_high_qc2(justify_qc.clone())
682            .await
683        {
684            bail!("Failed to store High QC, not voting; error = {e:?}");
685        }
686        if justify_qc
687            .data
688            .block_number
689            .is_some_and(|bn| is_epoch_root(bn, validation_info.epoch_height))
690        {
691            let Some(state_cert) = proposal.data.state_cert() else {
692                bail!("Epoch root QC has no state cert, not voting!");
693            };
694            if let Err(e) = validation_info
695                .storage
696                .update_state_cert(state_cert.clone())
697                .await
698            {
699                bail!("Failed to store the light client state update certificate, not voting; error = {:?}", e);
700            }
701            validation_info
702                .consensus
703                .write()
704                .await
705                .update_state_cert(state_cert.clone())?;
706        }
707        if let Some(ref next_epoch_justify_qc) = maybe_next_epoch_justify_qc {
708            if let Err(e) = validation_info
709                .storage
710                .update_next_epoch_high_qc2(next_epoch_justify_qc.clone())
711                .await
712            {
713                bail!("Failed to store next epoch High QC, not voting; error = {e:?}");
714            }
715        }
716    }
717    let mut consensus_writer = validation_info.consensus.write().await;
718    if let Some(ref next_epoch_justify_qc) = maybe_next_epoch_justify_qc {
719        if justify_qc
720            .data
721            .block_number
722            .is_some_and(|bn| is_transition_block(bn, validation_info.epoch_height))
723        {
724            consensus_writer.reset_high_qc(justify_qc.clone(), next_epoch_justify_qc.clone())?;
725            consensus_writer
726                .update_transition_qc(justify_qc.clone(), next_epoch_justify_qc.clone());
727            return Ok(());
728        }
729        consensus_writer.update_next_epoch_high_qc(next_epoch_justify_qc.clone())?;
730    }
731    consensus_writer.update_high_qc(justify_qc.clone())?;
732
733    Ok(())
734}

750pub(crate) async fn validate_epoch_transition_qc<
751    TYPES: NodeType,
752    I: NodeImplementation<TYPES>,
753    V: Versions,
754>(
755    proposal: &Proposal<TYPES, QuorumProposalWrapper<TYPES>>,
756    validation_info: &ValidationInfo<TYPES, I, V>,
757) -> Result<()> {
758    let proposed_qc = proposal.data.justify_qc();
759    let Some(qc_block_number) = proposed_qc.data().block_number else {
760        bail!("Justify QC has no block number");
761    };
762    if !is_epoch_transition(qc_block_number, validation_info.epoch_height)
763        || qc_block_number % validation_info.epoch_height == 0
764    {
765        return Ok(());
766    }
767    let Some(next_epoch_qc) = proposal.data.next_epoch_justify_qc() else {
768        bail!("Next epoch justify QC is not present");
769    };
770    ensure!(
771        next_epoch_qc.data.leaf_commit == proposed_qc.data().leaf_commit,
772        "Next epoch QC has different leaf commit to justify QC"
773    );
774
775    if is_transition_block(qc_block_number, validation_info.epoch_height) {
776        // Height is epoch height - 2
777        ensure!(
778            transition_qc(validation_info).await.is_none_or(
779                |(qc, _)| qc.view_number() <= proposed_qc.view_number()
780            ),
781            "Proposed transition qc must have view number greater than or equal to previous transition QC"
782        );
783
784        validation_info
785            .consensus
786            .write()
787            .await
788            .update_transition_qc(proposed_qc.clone(), next_epoch_qc.clone());
789        // reset the high qc to the transition qc
790        update_high_qc(proposal, validation_info).await?;
791    } else {
792        // Height is either epoch height - 1 or epoch height
793        ensure!(
794            transition_qc(validation_info)
795                .await
796                .is_none_or(|(qc, _)| qc.view_number() < proposed_qc.view_number()),
797            "Transition block must have view number greater than previous transition QC"
798        );
799        ensure!(
800            proposal.data.view_change_evidence().is_none(),
801            "Second to last block and last block of epoch must directly extend previous block, Qc Block number: {qc_block_number}, Proposal Block number: {}",
802            proposal.data.block_header().block_number()
803        );
804        ensure!(
805            proposed_qc.view_number() + 1 == proposal.data.view_number()
806            || transition_qc(validation_info).await.is_some_and(|(qc, _)| &qc == proposed_qc),
807            "Transition proposals must extend the previous view directly, or extend the previous transition block"
808        );
809    }
810    Ok(())
811}

813/// Validate the state and safety and liveness of a proposal then emit
814/// a `QuorumProposalValidated` event.
815///
816///
817/// # Errors
818/// If any validation or state update fails.
819#[allow(clippy::too_many_lines)]
820#[instrument(skip_all, fields(id = validation_info.id, view = *proposal.data.view_number()))]
821pub async fn validate_proposal_safety_and_liveness<
822    TYPES: NodeType,
823    I: NodeImplementation<TYPES>,
824    V: Versions,
825>(
826    proposal: Proposal<TYPES, QuorumProposalWrapper<TYPES>>,
827    parent_leaf: Leaf2<TYPES>,
828    validation_info: &ValidationInfo<TYPES, I, V>,
829    event_stream: Sender<Arc<HotShotEvent<TYPES>>>,
830    sender: TYPES::SignatureKey,
831) -> Result<()> {
832    let view_number = proposal.data.view_number();
833
834    let mut valid_epoch_transition = false;
835    if validation_info
836        .upgrade_lock
837        .version(proposal.data.justify_qc().view_number())
838        .await
839        .is_ok_and(|v| v >= V::Epochs::VERSION)
840    {
841        let Some(block_number) = proposal.data.justify_qc().data.block_number else {
842            bail!("Quorum Proposal has no block number but it's after the epoch upgrade");
843        };
844        if is_epoch_transition(block_number, validation_info.epoch_height) {
845            validate_epoch_transition_qc(&proposal, validation_info).await?;
846            valid_epoch_transition = true;
847        }
848    }
849
850    let proposed_leaf = Leaf2::from_quorum_proposal(&proposal.data);
851    ensure!(
852        proposed_leaf.parent_commitment() == parent_leaf.commit(),
853        "Proposed leaf does not extend the parent leaf."
854    );
855    let proposal_epoch = option_epoch_from_block_number::<TYPES>(
856        validation_info
857            .upgrade_lock
858            .epochs_enabled(view_number)
859            .await,
860        proposed_leaf.height(),
861        validation_info.epoch_height,
862    );
863
864    let state = Arc::new(
865        <TYPES::ValidatedState as ValidatedState<TYPES>>::from_header(proposal.data.block_header()),
866    );
867
868    {
869        let mut consensus_writer = validation_info.consensus.write().await;
870        if let Err(e) = consensus_writer.update_leaf(proposed_leaf.clone(), state, None) {
871            tracing::trace!("{e:?}");
872        }
873
874        // Update our internal storage of the proposal. The proposal is valid, so
875        // we swallow this error and just log if it occurs.
876        if let Err(e) = consensus_writer.update_proposed_view(proposal.clone()) {
877            tracing::debug!("Internal proposal update failed; error = {e:#}");
878        };
879    }
880
881    UpgradeCertificate::validate(
882        proposal.data.upgrade_certificate(),
883        &validation_info.membership,
884        proposal_epoch,
885        &validation_info.upgrade_lock,
886    )
887    .await?;
888
889    // Validate that the upgrade certificate is re-attached, if we saw one on the parent
890    proposed_leaf
891        .extends_upgrade(
892            &parent_leaf,
893            &validation_info.upgrade_lock.decided_upgrade_certificate,
894        )
895        .await?;
896
897    let justify_qc = proposal.data.justify_qc().clone();
898    // Create a positive vote if either liveness or safety check
899    // passes.
900
901    {
902        let consensus_reader = validation_info.consensus.read().await;
903        // Epoch safety check:
904        // The proposal is safe if
905        // 1. the proposed block and the justify QC block belong to the same epoch or
906        // 2. the justify QC is the eQC for the previous block
907        let justify_qc_epoch = option_epoch_from_block_number::<TYPES>(
908            validation_info
909                .upgrade_lock
910                .epochs_enabled(view_number)
911                .await,
912            parent_leaf.height(),
913            validation_info.epoch_height,
914        );
915        ensure!(
916            proposal_epoch == justify_qc_epoch
917                || consensus_reader.check_eqc(&proposed_leaf, &parent_leaf),
918            {
919                error!("Failed epoch safety check \n Proposed leaf is {proposed_leaf:?} \n justify QC leaf is {parent_leaf:?}")
920            }
921        );
922
923        // Make sure that the epoch transition proposal includes the next epoch QC
924        if is_epoch_transition(parent_leaf.height(), validation_info.epoch_height)
925            && validation_info
926                .upgrade_lock
927                .epochs_enabled(view_number)
928                .await
929        {
930            ensure!(proposal.data.next_epoch_justify_qc().is_some(),
931            "Epoch transition proposal does not include the next epoch justify QC. Do not vote!");
932        }
933
934        // Liveness check.
935        let liveness_check =
936            justify_qc.view_number() > consensus_reader.locked_view() || valid_epoch_transition;
937
938        // Safety check.
939        // Check if proposal extends from the locked leaf.
940        let outcome = consensus_reader.visit_leaf_ancestors(
941            justify_qc.view_number(),
942            Terminator::Inclusive(consensus_reader.locked_view()),
943            false,
944            |leaf, _, _| {
945                // if leaf view no == locked view no then we're done, report success by
946                // returning true
947                leaf.view_number() != consensus_reader.locked_view()
948            },
949        );
950        let safety_check = outcome.is_ok();
951
952        ensure!(safety_check || liveness_check, {
953            if let Err(e) = outcome {
954                broadcast_event(
955                    Event {
956                        view_number,
957                        event: EventType::Error { error: Arc::new(e) },
958                    },
959                    &validation_info.output_event_stream,
960                )
961                .await;
962            }
963
964            error!("Failed safety and liveness check \n High QC is {:?}  Proposal QC is {:?}  Locked view is {:?}", consensus_reader.high_qc(), proposal.data.clone(), consensus_reader.locked_view())
965        });
966    }
967
968    // We accept the proposal, notify the application layer
969    broadcast_event(
970        Event {
971            view_number,
972            event: EventType::QuorumProposal {
973                proposal: proposal.clone(),
974                sender,
975            },
976        },
977        &validation_info.output_event_stream,
978    )
979    .await;
980
981    // Notify other tasks
982    broadcast_event(
983        Arc::new(HotShotEvent::QuorumProposalValidated(
984            proposal.clone(),
985            parent_leaf,
986        )),
987        &event_stream,
988    )
989    .await;
990
991    Ok(())
992}

994/// Validates, from a given `proposal` that the view that it is being submitted for is valid when
995/// compared to `cur_view` which is the highest proposed view (so far) for the caller. If the proposal
996/// is for a view that's later than expected, that the proposal includes a timeout or view sync certificate.
997///
998/// # Errors
999/// If any validation or view number check fails.
1000pub(crate) async fn validate_proposal_view_and_certs<
1001    TYPES: NodeType,
1002    I: NodeImplementation<TYPES>,
1003    V: Versions,
1004>(
1005    proposal: &Proposal<TYPES, QuorumProposalWrapper<TYPES>>,
1006    validation_info: &ValidationInfo<TYPES, I, V>,
1007) -> Result<()> {
1008    let view_number = proposal.data.view_number();
1009    ensure!(
1010        view_number >= validation_info.consensus.read().await.cur_view(),
1011        "Proposal is from an older view {:?}",
1012        proposal.data
1013    );
1014
1015    // Validate the proposal's signature. This should also catch if the leaf_commitment does not equal our calculated parent commitment
1016    let mut membership = validation_info.membership.clone();
1017    proposal.validate_signature(&membership).await?;
1018
1019    // Verify a timeout certificate OR a view sync certificate exists and is valid.
1020    if proposal.data.justify_qc().view_number() != view_number - 1 {
1021        let received_proposal_cert =
1022            proposal.data.view_change_evidence().clone().context(debug!(
1023                "Quorum proposal for view {view_number} needed a timeout or view sync certificate, but did not have one",
1024        ))?;
1025
1026        match received_proposal_cert {
1027            ViewChangeEvidence2::Timeout(timeout_cert) => {
1028                ensure!(
1029                    timeout_cert.data().view == view_number - 1,
1030                    "Timeout certificate for view {view_number} was not for the immediately preceding view"
1031                );
1032                let timeout_cert_epoch = timeout_cert.data().epoch();
1033                membership = membership.get_new_epoch(timeout_cert_epoch).await?;
1034
1035                let membership_stake_table = membership.stake_table().await;
1036                let membership_success_threshold = membership.success_threshold().await;
1037
1038                timeout_cert
1039                    .is_valid_cert(
1040                        &StakeTableEntries::<TYPES>::from(membership_stake_table).0,
1041                        membership_success_threshold,
1042                        &validation_info.upgrade_lock,
1043                    )
1044                    .await
1045                    .context(|e| {
1046                        warn!("Timeout certificate for view {view_number} was invalid: {e}")
1047                    })?;
1048            },
1049            ViewChangeEvidence2::ViewSync(view_sync_cert) => {
1050                ensure!(
1051                    view_sync_cert.view_number == view_number,
1052                    "View sync cert view number {:?} does not match proposal view number {:?}",
1053                    view_sync_cert.view_number,
1054                    view_number
1055                );
1056
1057                let view_sync_cert_epoch = view_sync_cert.data().epoch();
1058                membership = membership.get_new_epoch(view_sync_cert_epoch).await?;
1059
1060                let membership_stake_table = membership.stake_table().await;
1061                let membership_success_threshold = membership.success_threshold().await;
1062
1063                // View sync certs must also be valid.
1064                view_sync_cert
1065                    .is_valid_cert(
1066                        &StakeTableEntries::<TYPES>::from(membership_stake_table).0,
1067                        membership_success_threshold,
1068                        &validation_info.upgrade_lock,
1069                    )
1070                    .await
1071                    .context(|e| warn!("Invalid view sync finalize cert provided: {e}"))?;
1072            },
1073        }
1074    }
1075
1076    // Validate the upgrade certificate -- this is just a signature validation.
1077    // Note that we don't do anything with the certificate directly if this passes; it eventually gets stored as part of the leaf if nothing goes wrong.
1078    {
1079        let epoch = option_epoch_from_block_number::<TYPES>(
1080            proposal.data.epoch().is_some(),
1081            proposal.data.block_header().block_number(),
1082            validation_info.epoch_height,
1083        );
1084        UpgradeCertificate::validate(
1085            proposal.data.upgrade_certificate(),
1086            &validation_info.membership,
1087            epoch,
1088            &validation_info.upgrade_lock,
1089        )
1090        .await?;
1091    }
1092
1093    Ok(())
1094}

1096/// Helper function to send events and log errors

1100        Ok(Some(overflowed)) => {

1111/// Validates qc's signatures and, if provided, validates next_epoch_qc's signatures and whether it
1112/// corresponds to the provided high_qc.
1113pub async fn validate_qc_and_next_epoch_qc<TYPES: NodeType, V: Versions>(
1114    qc: &QuorumCertificate2<TYPES>,
1115    maybe_next_epoch_qc: Option<&NextEpochQuorumCertificate2<TYPES>>,
1116    consensus: &OuterConsensus<TYPES>,
1117    membership_coordinator: &EpochMembershipCoordinator<TYPES>,
1118    upgrade_lock: &UpgradeLock<TYPES, V>,
1119    epoch_height: u64,
1120) -> Result<()> {
1121    let mut epoch_membership = membership_coordinator
1122        .stake_table_for_epoch(qc.data.epoch)
1123        .await?;
1124
1125    let membership_stake_table = epoch_membership.stake_table().await;
1126    let membership_success_threshold = epoch_membership.success_threshold().await;
1127
1128    if let Err(e) = qc
1129        .is_valid_cert(
1130            &StakeTableEntries::<TYPES>::from(membership_stake_table).0,
1131            membership_success_threshold,
1132            upgrade_lock,
1133        )
1134        .await
1135    {
1136        consensus.read().await.metrics.invalid_qc.update(1);
1137        return Err(warn!("Invalid certificate: {e}"));
1138    }
1139
1140    if upgrade_lock.epochs_enabled(qc.view_number()).await {
1141        ensure!(
1142            qc.data.block_number.is_some(),
1143            "QC for epoch {:?} has no block number",
1144            qc.data.epoch
1145        );
1146    }
1147
1148    if qc
1149        .data
1150        .block_number
1151        .is_some_and(|b| is_epoch_transition(b, epoch_height))
1152    {
1153        ensure!(
1154            maybe_next_epoch_qc.is_some(),
1155            error!("Received High QC for the transition block but not the next epoch QC")
1156        );
1157    }
1158
1159    if let Some(next_epoch_qc) = maybe_next_epoch_qc {
1160        // If the next epoch qc exists, make sure it's equal to the qc
1161        if qc.view_number() != next_epoch_qc.view_number() || qc.data != *next_epoch_qc.data {
1162            bail!("Next epoch qc exists but it's not equal with qc.");
1163        }
1164        epoch_membership = epoch_membership.next_epoch_stake_table().await?;
1165        let membership_next_stake_table = epoch_membership.stake_table().await;
1166        let membership_next_success_threshold = epoch_membership.success_threshold().await;
1167
1168        // Validate the next epoch qc as well
1169        next_epoch_qc
1170            .is_valid_cert(
1171                &StakeTableEntries::<TYPES>::from(membership_next_stake_table).0,
1172                membership_next_success_threshold,
1173                upgrade_lock,
1174            )
1175            .await
1176            .context(|e| warn!("Invalid next epoch certificate: {e}"))?;
1177    }
1178    Ok(())
1179}

1181/// Validates the light client state update certificate
1182pub async fn validate_light_client_state_update_certificate<TYPES: NodeType>(
1183    state_cert: &LightClientStateUpdateCertificate<TYPES>,
1184    membership_coordinator: &EpochMembershipCoordinator<TYPES>,
1185) -> Result<()> {
1186    tracing::debug!("Validating light client state update certificate");
1187
1188    let epoch_membership = membership_coordinator
1189        .membership_for_epoch(state_cert.epoch())
1190        .await?;
1191
1192    let membership_stake_table = epoch_membership.stake_table().await;
1193    let membership_success_threshold = epoch_membership.success_threshold().await;
1194
1195    let mut state_key_map = HashMap::new();
1196    membership_stake_table.into_iter().for_each(|config| {
1197        state_key_map.insert(
1198            config.state_ver_key.clone(),
1199            config.stake_table_entry.stake(),
1200        );
1201    });
1202
1203    let mut accumulated_stake = U256::from(0);
1204    for (key, sig) in state_cert.signatures.iter() {
1205        if let Some(stake) = state_key_map.get(key) {
1206            accumulated_stake += *stake;
1207            if !key.verify_state_sig(
1208                sig,
1209                &state_cert.light_client_state,
1210                &state_cert.next_stake_table_state,
1211            ) {
1212                bail!("Invalid light client state update certificate signature");
1213            }
1214        } else {
1215            bail!("Invalid light client state update certificate signature");
1216        }
1217    }
1218    if accumulated_stake < membership_success_threshold {
1219        bail!("Light client state update certificate does not meet the success threshold");
1220    }
1221
1222    Ok(())
1223}

1225pub(crate) fn check_qc_state_cert_correspondence<TYPES: NodeType>(
1226    qc: &QuorumCertificate2<TYPES>,
1227    state_cert: &LightClientStateUpdateCertificate<TYPES>,
1228    epoch_height: u64,
1229) -> bool {
1230    qc.data
1231        .block_number
1232        .is_some_and(|bn| is_epoch_root(bn, epoch_height))
1233        && Some(state_cert.epoch) == qc.data.epoch()
1234        && qc.view_number().u64() == state_cert.light_client_state.view_number
1235}

1237/// Gets the second VID share, the current or the next epoch accordingly, from the shared consensus state;
1238/// makes sure it corresponds to the given DA certificate;
1239/// if it's not yet available, waits for it with the given timeout.
1240pub async fn wait_for_second_vid_share<TYPES: NodeType>(
1241    target_epoch: Option<TYPES::Epoch>,
1242    vid_share: &Proposal<TYPES, VidDisperseShare<TYPES>>,
1243    da_cert: &DaCertificate2<TYPES>,
1244    consensus: &OuterConsensus<TYPES>,
1245    receiver: &Receiver<Arc<HotShotEvent<TYPES>>>,
1246) -> Result<Proposal<TYPES, VidDisperseShare<TYPES>>> {
1247    tracing::debug!("getting the second VID share for epoch {:?}", target_epoch);
1248    let maybe_second_vid_share = consensus
1249        .read()
1250        .await
1251        .vid_shares()
1252        .get(&vid_share.data.view_number())
1253        .and_then(|key_map| key_map.get(vid_share.data.recipient_key()))
1254        .and_then(|epoch_map| epoch_map.get(&target_epoch))
1255        .cloned();
1256    if let Some(second_vid_share) = maybe_second_vid_share {
1257        if (target_epoch == da_cert.epoch()
1258            && second_vid_share.data.payload_commitment() == da_cert.data().payload_commit)
1259            || (target_epoch != da_cert.epoch()
1260                && Some(second_vid_share.data.payload_commitment())
1261                    == da_cert.data().next_epoch_payload_commit)
1262        {
1263            return Ok(second_vid_share);
1264        }
1265    }
1266
1267    let receiver = receiver.clone();
1268    let da_cert_clone = da_cert.clone();
1269    let Some(event) = EventDependency::new(
1270        receiver,
1271        Box::new(move |event| {
1272            let event = event.as_ref();
1273            if let HotShotEvent::VidShareValidated(second_vid_share) = event {
1274                if target_epoch == da_cert_clone.epoch() {
1275                    second_vid_share.data.payload_commitment()
1276                        == da_cert_clone.data().payload_commit
1277                } else {
1278                    Some(second_vid_share.data.payload_commitment())
1279                        == da_cert_clone.data().next_epoch_payload_commit
1280                }
1281            } else {
1282                false
1283            }
1284        }),
1285    )
1286    .completed()
1287    .await
1288    else {
1289        return Err(warn!("Error while waiting for the second VID share."));
1290    };
1291    let HotShotEvent::VidShareValidated(second_vid_share) = event.as_ref() else {
1292        // this shouldn't happen
1293        return Err(warn!("Received event is not VidShareValidated but we checked it earlier. Shouldn't be possible."));
1294    };
1295    Ok(second_vid_share.clone())
1296}
