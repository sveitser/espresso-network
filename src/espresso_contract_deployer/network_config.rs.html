20/// Returns both genesis light client state and stake table state
21pub async fn light_client_genesis(
22    sequencer_url: &Url,
23    stake_table_capacity: usize,
24) -> anyhow::Result<(LightClientStateSol, StakeTableStateSol)> {
25    let st = fetch_stake_table_from_sequencer(sequencer_url, None)
26        .await
27        .with_context(|| "Failed to initialize stake table")?;
28    light_client_genesis_from_stake_table(&st, stake_table_capacity)
29}

31/// Fetch the stake table from a sequencer node given the epoch number
32///
33/// Does not error, runs until the stake table is provided.
34pub async fn fetch_stake_table_from_sequencer(
35    sequencer_url: &Url,
36    epoch: Option<<SeqTypes as NodeType>::Epoch>,
37) -> Result<HSStakeTable<SeqTypes>> {
38    tracing::info!("Initializing stake table from node for epoch {epoch:?}");
39
40    match epoch {
41        Some(epoch) => loop {
42            match surf_disco::Client::<tide_disco::error::ServerError, StaticVersion<0, 1>>::new(
43                sequencer_url.clone(),
44            )
45            .get::<Vec<PeerConfig<SeqTypes>>>(&format!("node/stake-table/{}", epoch.u64()))
46            .send()
47            .await
48            {
49                Ok(resp) => break Ok(resp.into()),
50                Err(e) => {
51                    tracing::error!("Failed to fetch the stake table: {e}");
52                    sleep(Duration::from_secs(5)).await;
53                },
54            }
55        },
56        None => loop {
57            match surf_disco::Client::<tide_disco::error::ServerError, StaticVersion<0, 1>>::new(
58                sequencer_url.clone(),
59            )
60            .get::<PublicNetworkConfig>("config/hotshot")
61            .send()
62            .await
63            {
64                Ok(resp) => break Ok(resp.hotshot_config().known_nodes_with_stake().into()),
65                Err(e) => {
66                    tracing::error!("Failed to fetch the network config: {e}");
67                    sleep(Duration::from_secs(5)).await;
68                },
69            }
70        },
71    }
72}

75/// derive the genesis light client state and stake table state from initial set of `PeerConfig`
76pub fn light_client_genesis_from_stake_table(
77    st: &HSStakeTable<SeqTypes>,
78    stake_table_capacity: usize,
79) -> anyhow::Result<(LightClientStateSol, StakeTableStateSol)> {
80    let st_state = st.commitment(stake_table_capacity)?;
81    Ok((
82        LightClientStateSol {
83            viewNum: 0,
84            blockHeight: 0,
85            blockCommRoot: U256::from(0u32),
86        },
87        StakeTableStateSol {
88            blsKeyComm: field_to_u256(st_state.bls_key_comm),
89            schnorrKeyComm: field_to_u256(st_state.schnorr_key_comm),
90            amountComm: field_to_u256(st_state.amount_comm),
91            threshold: field_to_u256(st_state.threshold),
92        },
93    ))
94}

96/// Get the epoch-related  from the sequencer's `PublicHotShotConfig` struct
97/// return (blocks_per_epoch, epoch_start_block)
98pub async fn fetch_epoch_config_from_sequencer(sequencer_url: &Url) -> anyhow::Result<(u64, u64)> {
99    // Request the configuration until it is successful
100    let epoch_config = loop {
101        match surf_disco::Client::<tide_disco::error::ServerError, StaticVersion<0, 1>>::new(
102            sequencer_url.clone(),
103        )
104        .get::<PublicNetworkConfig>("config/hotshot")
105        .send()
106        .await
107        {
108            Ok(resp) => {
109                let config = resp.hotshot_config();
110                break (config.blocks_per_epoch(), config.epoch_start_block());
111            },
112            Err(e) => {
113                tracing::error!("Failed to fetch the network config: {e}");
114                sleep(Duration::from_secs(5)).await;
115            },
116        }
117    };
118    Ok(epoch_config)
