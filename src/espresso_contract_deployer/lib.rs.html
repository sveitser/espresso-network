34/// a handy thin wrapper around wallet builder and provider builder that directly
35/// returns an instantiated `Provider` with default fillers with wallet, ready to send tx
36pub fn build_provider(mnemonic: String, account_index: u32, url: Url) -> HttpProviderWithWallet {
37    let signer = build_signer(mnemonic, account_index);
38    let wallet = EthereumWallet::from(signer);
39    ProviderBuilder::new().wallet(wallet).on_http(url)
40}

42pub fn build_signer(mnemonic: String, account_index: u32) -> PrivateKeySigner {
43    MnemonicBuilder::<English>::default()
44        .phrase(mnemonic)
45        .index(account_index)
46        .expect("wrong mnemonic or index")
47        .build()
48        .expect("fail to build signer")
49}

201    pub fn address(&self, contract: Contract) -> Option<Address> {
202        self.0.get(&contract).copied()

205    /// Deploy a contract (with logging and cached deployments)
206    ///
207    /// The deployment `tx` will be sent only if contract `name` is not already deployed;
208    /// otherwise this function will just return the predeployed address.
209    pub async fn deploy<T, P>(
210        &mut self,
211        name: Contract,
212        tx: RawCallBuilder<T, P>,
213    ) -> Result<Address>
214    where
215        P: Provider,
216    {
217        if let Some(addr) = self.0.get(&name) {
218            tracing::info!("skipping deployment of {name}, already deployed at {addr:#x}");
219            return Ok(*addr);
220        }
221        tracing::info!("deploying {name}");
222        let addr = tx.deploy().await?;
223        tracing::info!("deployed {name} at {addr:#x}");
224
225        self.0.insert(name, addr);
226        Ok(addr)
227    }
228
229    /// Write a .env file.
230    pub fn write(&self, mut w: impl Write) -> Result<()> {
231        for (contract, address) in &self.0 {
232            writeln!(w, "{contract}={address:#x}")?;
233        }
234        Ok(())
235    }
236}

238/// Default deployment function `LightClient.sol` or `LightClientMock.sol` with `mock: true`.
239///
240/// # NOTE:
241/// In most cases, you only need to use [`deploy_light_client_proxy()`]
242///
243/// # NOTE:
244/// currently, `LightClient.sol` follows upgradable contract, thus a follow-up
245/// call to `.initialize()` with proper genesis block (and other constructor args)
246/// are expected to be *delegatecall-ed through the proxy contract*.
247pub(crate) async fn deploy_light_client_contract(
248    provider: impl Provider,
249    contracts: &mut Contracts,
250    mock: bool,
251) -> Result<Address> {
252    // Deploy library contracts.
253    let plonk_verifier_addr = contracts
254        .deploy(
255            Contract::PlonkVerifier,
256            PlonkVerifier::deploy_builder(&provider),
257        )
258        .await?;
259
260    assert!(is_contract(&provider, plonk_verifier_addr).await?);
261
262    // when generate alloy's bindings, we supply a placeholder address, now we modify the actual
263    // bytecode with deployed address of the library.
264    let target_lc_bytecode = if mock {
265        LightClientMock::BYTECODE.encode_hex()
266    } else {
267        LightClient::BYTECODE.encode_hex()
268    };
269    let lc_linked_bytecode = {
270        match target_lc_bytecode
271            .matches(LIBRARY_PLACEHOLDER_ADDRESS)
272            .count()
273        {
274            0 => return Err(anyhow!("lib placeholder not found")),
275            1 => Bytes::from_hex(target_lc_bytecode.replacen(
276                LIBRARY_PLACEHOLDER_ADDRESS,
277                &plonk_verifier_addr.encode_hex(),
278                1,
279            ))?,
280            _ => {
281                return Err(anyhow!(
282                    "more than one lib placeholder found, consider using a different value"
283                ))
284            },
285        }
286    };
287
288    // Deploy the light client
289    let light_client_addr = if mock {
290        // for mock, we don't populate the `contracts` since it only track production-ready deployments
291        let addr = LightClientMock::deploy_builder(&provider)
292            .map(|req| req.with_deploy_code(lc_linked_bytecode))
293            .deploy()
294            .await?;
295        tracing::info!("deployed LightClientMock at {addr:#x}");
296        addr
297    } else {
298        contracts
299            .deploy(
300                Contract::LightClient,
301                LightClient::deploy_builder(&provider)
302                    .map(|req| req.with_deploy_code(lc_linked_bytecode)),
303            )
304            .await?
305    };
306    Ok(light_client_addr)
307}

309/// The primary logic for deploying and initializing an upgradable light client contract.
310///
311/// Deploy the upgradable proxy contract, point to an already deployed light client contract as its implementation, and invoke `initialize()` on it.
312/// This is run after `deploy_light_client_contract()`, returns the proxy address.
313/// This works for both mock and production light client proxy.
314pub async fn deploy_light_client_proxy(
315    provider: impl Provider,
316    contracts: &mut Contracts,
317    mock: bool,
318    genesis_state: LightClientStateSol,
319    genesis_stake: StakeTableStateSol,
320    admin: Address,
321    prover: Option<Address>,
322) -> Result<Address> {
323    // deploy the light client implementation contract
324    let impl_addr = deploy_light_client_contract(&provider, contracts, mock).await?;
325    let lc = LightClient::new(impl_addr, &provider);
326
327    // prepare the input arg for `initialize()`
328    let init_data = lc
329        .initialize(
330            genesis_state,
331            genesis_stake,
332            MAX_HISTORY_RETENTION_SECONDS,
333            admin,
334        )
335        .calldata()
336        .to_owned();
337    // deploy proxy and initialize
338    let lc_proxy_addr = contracts
339        .deploy(
340            Contract::LightClientProxy,
341            ERC1967Proxy::deploy_builder(&provider, impl_addr, init_data),
342        )
343        .await?;
344
345    // sanity check
346    if !is_proxy_contract(&provider, lc_proxy_addr).await? {
347        panic!("LightClientProxy detected not as a proxy, report error!");
348    }
349
350    // instantiate a proxy instance, cast as LightClient's ABI interface
351    let lc_proxy = LightClient::new(lc_proxy_addr, &provider);
352
353    // set permissioned prover
354    if let Some(prover) = prover {
355        tracing::info!(%lc_proxy_addr, %prover, "Set permissioned prover ");
356        lc_proxy
357            .setPermissionedProver(prover)
358            .send()
359            .await?
360            .get_receipt()
361            .await?;
362    }
363
364    // post deploy verification checks
365    assert_eq!(lc_proxy.getVersion().call().await?.majorVersion, 1);
366    assert_eq!(lc_proxy.owner().call().await?._0, admin);
367    if let Some(prover) = prover {
368        assert_eq!(lc_proxy.permissionedProver().call().await?._0, prover);
369    }
370    assert_eq!(
371        lc_proxy.stateHistoryRetentionPeriod().call().await?._0,
372        864000
373    );
374    assert_eq!(
375        lc_proxy.currentBlockNumber().call().await?._0,
376        U256::from(provider.get_block_number().await?)
377    );
378
379    Ok(lc_proxy_addr)
380}

382/// Upgrade the light client proxy to use LightClientV2.
383/// Internally, first detect existence of proxy, then deploy LCV2, then upgrade and initializeV2.
384/// Internal to "deploy LCV2", we deploy PlonkVerifierV2 whose address will be used at LCV2 init time.
385pub async fn upgrade_light_client_v2(
386    provider: impl Provider,
387    contracts: &mut Contracts,
388    is_mock: bool,
389    blocks_per_epoch: u64,
390    epoch_start_block: u64,
391) -> Result<TransactionReceipt> {
392    match contracts.address(Contract::LightClientProxy) {
393        // check if proxy already exists
394        None => Err(anyhow!("LightClientProxy not found, can't upgrade")),
395        Some(proxy_addr) => {
396            let proxy = LightClient::new(proxy_addr, &provider);
397            let state_history_retention_period =
398                proxy.stateHistoryRetentionPeriod().call().await?._0;
399            // first deploy PlonkVerifierV2.sol
400            let pv2_addr = contracts
401                .deploy(
402                    Contract::PlonkVerifierV2,
403                    PlonkVerifierV2::deploy_builder(&provider),
404                )
405                .await?;
406
407            assert!(is_contract(&provider, pv2_addr).await?);
408
409            // then deploy LightClientV2.sol
410            let target_lcv2_bytecode = if is_mock {
411                LightClientV2Mock::BYTECODE.encode_hex()
412            } else {
413                LightClientV2::BYTECODE.encode_hex()
414            };
415            let lcv2_linked_bytecode = {
416                match target_lcv2_bytecode
417                    .matches(LIBRARY_PLACEHOLDER_ADDRESS)
418                    .count()
419                {
420                    0 => return Err(anyhow!("lib placeholder not found")),
421                    1 => Bytes::from_hex(target_lcv2_bytecode.replacen(
422                        LIBRARY_PLACEHOLDER_ADDRESS,
423                        &pv2_addr.encode_hex(),
424                        1,
425                    ))?,
426                    _ => {
427                        return Err(anyhow!(
428                            "more than one lib placeholder found, consider using a different value"
429                        ))
430                    },
431                }
432            };
433            let lcv2_addr = if is_mock {
434                let addr = LightClientV2Mock::deploy_builder(&provider)
435                    .map(|req| req.with_deploy_code(lcv2_linked_bytecode))
436                    .deploy()
437                    .await?;
438                tracing::info!("deployed LightClientV2Mock at {addr:#x}");
439                addr
440            } else {
441                contracts
442                    .deploy(
443                        Contract::LightClientV2,
444                        LightClientV2::deploy_builder(&provider)
445                            .map(|req| req.with_deploy_code(lcv2_linked_bytecode)),
446                    )
447                    .await?
448            };
449
450            // prepare init calldata
451            let lcv2 = LightClientV2::new(lcv2_addr, &provider);
452            let init_data = lcv2
453                .initializeV2(blocks_per_epoch, epoch_start_block)
454                .calldata()
455                .to_owned();
456            // invoke upgrade on proxy
457            let receipt = proxy
458                .upgradeToAndCall(lcv2_addr, init_data)
459                .send()
460                .await?
461                .get_receipt()
462                .await?;
463            if receipt.inner.is_success() {
464                // post deploy verification checks
465                let proxy_as_v2 = LightClientV2::new(proxy_addr, &provider);
466                assert_eq!(proxy_as_v2.getVersion().call().await?.majorVersion, 2);
467                assert_eq!(
468                    proxy_as_v2.blocksPerEpoch().call().await?._0,
469                    blocks_per_epoch
470                );
471                assert_eq!(
472                    proxy_as_v2.epochStartBlock().call().await?._0,
473                    epoch_start_block
474                );
475                assert_eq!(
476                    proxy_as_v2.stateHistoryRetentionPeriod().call().await?._0,
477                    state_history_retention_period
478                );
479                assert_eq!(
480                    proxy_as_v2.currentBlockNumber().call().await?._0,
481                    U256::from(provider.get_block_number().await?)
482                );
483
484                tracing::info!(%lcv2_addr, "LightClientProxy successfully upgrade to: ")
485            } else {
486                tracing::error!("LightClientProxy upgrade failed: {:?}", receipt);
487            }
488
489            Ok(receipt)
490        },
491    }
492}

494/// The primary logic for deploying and initializing an upgradable fee contract.
495///
496/// Deploy the upgradable proxy contract, point to a deployed fee contract as its implementation, and invoke `initialize()` on it.
497/// - `admin`: is the new owner (e.g. a multisig address) of the proxy contract
498///
499/// Return the proxy address.
500pub async fn deploy_fee_contract_proxy(
501    provider: impl Provider,
502    contracts: &mut Contracts,
503    admin: Address,
504) -> Result<Address> {
505    // deploy the fee implementation contract
506    let fee_addr = contracts
507        .deploy(
508            Contract::FeeContract,
509            FeeContract::deploy_builder(&provider),
510        )
511        .await?;
512    let fee = FeeContract::new(fee_addr, &provider);
513
514    // prepare the input arg for `initialize()`
515    let init_data = fee.initialize(admin).calldata().to_owned();
516    // deploy proxy and initialize
517    let fee_proxy_addr = contracts
518        .deploy(
519            Contract::FeeContractProxy,
520            ERC1967Proxy::deploy_builder(&provider, fee_addr, init_data),
521        )
522        .await?;
523    // sanity check
524    if !is_proxy_contract(&provider, fee_proxy_addr).await? {
525        panic!("FeeContractProxy detected not as a proxy, report error!");
526    }
527
528    // post deploy verification checks
529    let fee_proxy = FeeContract::new(fee_proxy_addr, &provider);
530    assert_eq!(fee_proxy.getVersion().call().await?.majorVersion, 1);
531    assert_eq!(fee_proxy.owner().call().await?._0, admin);
532
533    Ok(fee_proxy_addr)
534}

628/// Common logic for any Ownable contract to transfer ownership
629pub async fn transfer_ownership(
630    provider: impl Provider,
631    target: Contract,
632    addr: Address,
633    new_owner: Address,
634) -> Result<TransactionReceipt> {
635    let receipt = match target {
636        Contract::LightClient | Contract::LightClientProxy => {
637            tracing::info!(%addr, %new_owner, "Transfer LightClient ownership");
638            let lc = LightClient::new(addr, &provider);
639            lc.transferOwnership(new_owner)
640                .send()
641                .await?
642                .get_receipt()
643                .await?
644        },
645        Contract::FeeContract | Contract::FeeContractProxy => {
646            tracing::info!(%addr, %new_owner, "Transfer FeeContract ownership");
647            let fee = FeeContract::new(addr, &provider);
648            fee.transferOwnership(new_owner)
649                .send()
650                .await?
651                .get_receipt()
652                .await?
653        },
654        Contract::EspToken | Contract::EspTokenProxy => {
655            tracing::info!(%addr, %new_owner, "Transfer EspToken ownership");
656            let token = EspToken::new(addr, &provider);
657            token
658                .transferOwnership(new_owner)
659                .send()
660                .await?
661                .get_receipt()
662                .await?
663        },
664        Contract::StakeTable | Contract::StakeTableProxy => {
665            tracing::info!(%addr, %new_owner, "Transfer StakeTable ownership");
666            let stake_table = StakeTable::new(addr, &provider);
667            stake_table
668                .transferOwnership(new_owner)
669                .send()
670                .await?
671                .get_receipt()
672                .await?
673        },
674        _ => return Err(anyhow!("Not Ownable, can't transfer ownership!")),
675    };
676    Ok(receipt)
677}

679/// helper function to decide if the contract at given address `addr` is a proxy contract
680pub async fn is_proxy_contract(provider: impl Provider, addr: Address) -> Result<bool> {
681    // confirm that the proxy_address is a proxy
682    // using the implementation slot, 0x360894a13ba1a3210667c828492db98dca3e2076cc3735a920a3ca505d382bbc, which is the keccak-256 hash of "eip1967.proxy.implementation" subtracted by 1
683    let impl_slot = U256::from_str_radix(
684        "360894a13ba1a3210667c828492db98dca3e2076cc3735a920a3ca505d382bbc",
685        16,
686    )?;
687    let storage = provider.get_storage_at(addr, impl_slot).await?;
688    let impl_address = Address::from_slice(&storage.to_be_bytes_vec()[12..]);
689
690    // when the implementation address is not equal to zero, it's a proxy
691    Ok(impl_address != Address::default())
692}

694pub async fn is_contract(provider: impl Provider, address: Address) -> Result<bool> {
695    if address == Address::ZERO {
696        return Ok(false);
697    }
698
699    let code = provider.get_code_at(address).await?;
700    if code.is_empty() {
701        return Ok(false);
702    }
703
704    Ok(true)
705}

707/// Deploy and initialize a Timelock contract
708///
709/// Parameters:
710/// - `min_delay`: The minimum delay for operations
711/// - `proposers`: The list of addresses that can propose
712/// - `executors`: The list of addresses that can execute
713/// - `admin`: The address that can perform admin actions
714pub async fn deploy_timelock(
715    provider: impl Provider,
716    contracts: &mut Contracts,
717    min_delay: U256,
718    proposers: Vec<Address>,
719    executors: Vec<Address>,
720    admin: Address,
721) -> Result<Address> {
722    let timelock_addr = contracts
723        .deploy(
724            Contract::Timelock,
725            Timelock::deploy_builder(
726                &provider,
727                min_delay,
728                proposers.clone(),
729                executors.clone(),
730                admin,
731            ),
732        )
733        .await?;
734
735    // Verify deployment
736    let timelock = Timelock::new(timelock_addr, &provider);
737
738    // Verify initialization parameters
739    assert_eq!(timelock.getMinDelay().call().await?._0, min_delay);
740    assert!(
741        timelock
742            .hasRole(timelock.PROPOSER_ROLE().call().await?._0, proposers[0])
743            .call()
744            .await?
745            ._0
746    );
747    assert!(
748        timelock
749            .hasRole(timelock.EXECUTOR_ROLE().call().await?._0, executors[0])
750            .call()
751            .await?
752            ._0
753    );
754
755    // test that the admin is in the default admin role where DEFAULT_ADMIN_ROLE = 0x00
756    let default_admin_role = U256::ZERO;
757    assert!(
758        timelock
759            .hasRole(default_admin_role.into(), admin)
760            .call()
761            .await?
762            ._0
763    );
764
765    Ok(timelock_addr)
766}
